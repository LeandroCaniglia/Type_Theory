\section{Quasi-Inverses}

\begin{lem}\label{lem:X->X-id-homotopy}
    Let\/ $X$ be a type. Given a function\/ $u\colon X\to X$ and a homotopy\/ $\theta\colon u\sim\id_X$, there is a homotopy
    \[
        \theta\rwct u\sim u\lwct\theta.
    \]
\end{lem}

\begin{proof}
    Let $p\colon x\eq Xy$ be a path. The naturality identity \eqref{eq:nat-transformation} of $\theta$ evaluated on $p$ yields
    \[
        \theta(x)\ct p\eq{u(x)\eq Xy}\fun{ap}_u(p)\ct\theta(y).
    \]
    By choosing $p\defeq\theta(x)\colon u(x)\eq Xx$, we obtain
    \[
        \theta(u(x))\ct\theta(x)
        \eq{u(u(x))\eq Xx}
        \fun{ap}_u(\theta(x))\ct\theta(x).
    \]
    Right-canceling $\theta(x)$ yields the stated homotopy.
\end{proof}

\begin{prop}\label{prop:coherence-swap}
    Given two functions\/ $f\colon A\to B$ and\/ $g\colon B\to A$ and two homotopies\/ $\eta\colon f\circ g\sim\id_B$ and\/ $\vartheta\colon g\circ f\sim\id_A$, the types
    \[
        (f\lwct\vartheta\sim\eta\rwct f)
            \to(g\lwct\eta\sim\vartheta\rwct g)
        \quad\text{and\/}\quad
        (g\lwct\eta\sim\vartheta\rwct g)
            \to(f\lwct\vartheta\sim\eta\rwct f)
    \]
    are inhabited.
\end{prop}

\begin{proof}
    By symmetry, it suffices to construct a map from the first to the second homotopy.

    Fix a homotopy $\tau\colon f\lwct\vartheta\sim\eta\rwct f$. We must construct $\rho\colon g\lwct\eta\sim\vartheta\rwct g$.

    Applying Lemma~\ref{lem:X->X-id-homotopy} to $\eta\colon f\circ g\sim\id_B$ and $\vartheta\colon g\circ f\sim\id_A$, we obtain two homotopies:
    \begin{align*}
        N_\eta&\colon\eta\rwct(f\circ g)
            \sim(f\circ g)\lwct\eta,\\
        N_\vartheta&\colon\vartheta\rwct(g\circ f)
            \sim(g\circ f)\lwct\vartheta.
    \end{align*}
    By whiskering $\tau$ with $g$ on both sides, we obtain
    \[
        g\lwct\tau\rwct g\colon(g\circ f)\lwct\vartheta\rwct g\sim g\lwct\eta\rwct(f\circ g).
    \]
    Then
    \begin{align*}
        \vartheta\rwct(g\circ f\circ g)
        &\sim(g\circ f)\lwct\vartheta\rwct g && \text{; via $N_\vartheta\rwct g$}\\
        &\sim g\lwct\eta\rwct(f\circ g) && \text{; via $g\lwct\tau\rwct g$}\\
        &\sim(g\circ f\circ g)\lwct\eta && \text{; via $g\lwct N_\eta$},
    \end{align*}
    yielding a homotopy $\alpha\colon\vartheta\rwct(g\circ f\circ g)\sim(g\circ f\circ g)\lwct\eta$.

    Now consider the naturality of $\vartheta\colon g\circ f\sim\id_A$ evaluated on the homotopy $g\lwct\eta\colon g\circ f\circ g\sim g$, as described in \S\,\ref{homotopy-evaluated-on-homotopy}. In this case, diagram~\eqref{tik:homotopy-evaluated-on-homotopy} yields the following commutative square of homotopies:
    \[
        \begin{tikzcd}[column sep=huge, row sep=huge]
            g\circ f\circ g\circ f\circ g
                \arrow[r, "\vartheta\rwct(g\circ f\circ g)"]
                \arrow[d, "(g\circ f\circ g)\lwct\eta"'] &
            g\circ f\circ g
                \arrow[d, "g\lwct\eta"] \\
            g\circ f\circ g
                \arrow[r, "\vartheta\rwct g"'] &
            g
        \end{tikzcd}
    \]
    Since $\alpha$ establishes that the top edge is homotopic to the left edge, we obtain the required homotopy
    \[
        g\lwct\eta \sim \vartheta\rwct g.
    \]
\end{proof}

\begin{cor}\label{cor:coherent-quasi-inverse-2}
    If\/ $f\colon A\to B$ is an equivalence with quasi-inverse\/ $\dep{g,(\eta,\vartheta)}$, there is a homotopy\/ $\vartheta'\colon g\circ f\sim\id_A$ such that\/ $\dep{g,(\eta,\vartheta')}$ is a coherent quasi-inverse.
\end{cor}

\begin{proof}
    The hypothesis implies that $\dep{f,(\vartheta,\eta)}$ is a quasi-inverse of $g$.
    By Theorem~\ref{thm:coherent-quasi-inverse}, there is a homotopy $\vartheta'$ such that $\dep{f,(\vartheta',\eta)}$ is a coherent quasi-inverse.
    This means that we have $\tau\colon g\lwct\eta\sim\vartheta'\rwct g$. Applying Proposition~\ref{prop:coherence-swap}, we deduce that $f\lwct\vartheta'\sim\eta\rwct f$ is inhabited, as desired.
\end{proof}

\begin{thm}\label{thm:equivalence-fibers-are-contractible}
    If\/ $f\colon A\to B$ is an equivalence with quasi-inverse\/ $\dep{g,(\eta,\vartheta)}$, then for all\/ $y\colon B$ the fiber\/ $\fun{fib}_f(y)$ is contractible with center\/ $\dep{g(y),\eta(y)}$.
\end{thm}

\begin{proof}
    By Corollary~\ref{cor:coherent-quasi-inverse-2}, after replacing $\vartheta$ with $\vartheta'$, we can assume that there exists a homotopy
    \[
        \tau\colon f\lwct\vartheta\sim\eta\rwct f.
    \]
    Given $y\colon B$, note that
    \[
        \dep{g(y),\eta(y)}\colon\fun{fib}_f(y)
    \]
    is well-typed since $\eta(y)\colon f(g(y))\eq By$.

    Fix $\dep{x,p}\colon\fun{fib}_f(y)$, where $p\colon f(x)\eq By$. According to Definition~\ref{defn:fiber-over-y}, we must provide a path $\gamma\colon g(y)\eq Ax$ such that
    \[
        \fun{transport}^{z\mapsto f(z)\eq By}(\gamma,\eta(y))\eq{f(x)\eq By}p.
    \]
    Using Lemma~\ref{lem:post-composition-transport}, the \lhs equals
    \[
        \fun{transport}^{z\mapsto z\eq By}\bigl(\fun{ap}_f(\gamma),\eta(y)\bigr).
    \]
    Therefore, by Proposition~\ref{prop:transport-decomposition-for-paths}, our goal becomes
    \[
        \fun{ap}_f(\gamma)^{-1}\ct\eta(y)\eq{f(x)\eq By}p,
    \]
    which is equivalent to
    \[
        \fun{ap}_f(\gamma)\ct p\eq{f(g(y))\eq By}\eta(y).\tag{$\dag$}
    \]
    Let
    \[
        \gamma\defeq\fun{ap}_g(p)^{-1}\ct\vartheta_x.
    \]
    Since $\fun{ap}_g(p)\colon g(f(x))\eq Ag(y)$ and $\vartheta_x\colon g(f(x))\eq Ax$, we see that $\gamma$ is of the required type. Moreover,
    \[
        \fun{ap}_f(\gamma)\ct p
            \eq{f(g(y))\eq By}
                \fun{ap}_f(\fun{ap}_g(p))^{-1}
                \ct\fun{ap}_f(\vartheta_x)
                \ct p.
    \]
    Using the coherence condition $\tau_x\colon\fun{ap}_f(\vartheta_x)
            \eq{f(g(f(x)))\eq Bf(x)}\eta_{f(x)}$, we obtain
    \[\tag{$\ddag$}
        \fun{ap}_f(\gamma)\ct p
        \eq{f(g(y))\eq By}
        \fun{ap}_f(\fun{ap}_g(p))^{-1}\ct\eta_{f(x)}\ct p.
    \]
    The naturality path diagram \eqref{tik:nat-transformation} applied to $p\colon f(x)\eq By$ and $\eta\colon f\circ g\sim\id_B$ yields
    \[
        \begin{tikzcd}[column sep=large]
            f(g(f(x)))
                    \arrow[r,"\fun{ap}_{f\circ g}(p)"]
                    \arrow[d,"\eta(f(x))"']
                &f(g(y))
                    \arrow[d,"\eta(y)"]\\
            f(x)
                    \arrow[r,"\fun{ap}_{\id_B}(p)"']
                &y
        \end{tikzcd}
    \]
    Using that $\fun{ap}_{f\circ g}\equiv\fun{ap}_f\circ\fun{ap}_g$ and $\fun{ap}_{\id_B}\equiv\id$, the diagram translates into
    \[
        \fun{ap}_f(\fun{ap}_g(p))^{-1}\ct\eta(f(x))\ct p
        \eq{f(g(y))\eq By}
        \eta(y).
    \]
    Substituting this into ($\ddag$), we arrive at ($\dag$).
\end{proof}

\begin{defn}\label{defn:left-right-inverses}
    Given a function $f\colon A\to B$, we define the types
    \begin{align*}
        \fun{linv}(f)
            &\defeq\sum_{g\colon B\to A}g\circ f\sim \id_A\\
        \fun{rinv}(f)
            &\defeq\sum_{g\colon B\to A}f\circ g\sim\id_B
    \end{align*}
    of \textsl{left} and \textsl{right inverses} to $f$, respectively. We call $f$ \textsl{left invertible} if $\fun{linv}(f)$ is inhabited, and \textsl{right invertible} if $\fun{rinv}(f)$ is inhabited.
\end{defn}

\begin{lem}\label{lem:transport-Q}
    Let\/ $B\colon\univ$ be a type and\/ $P\colon B\to\univ$ a type family. Consider two functions\/ $f,g\colon B\to\Sigma_BP$, a path\/ $p\colon f=g$, and the type family
    \begin{align*}
        Q\colon(B\to\Sigma_BP)&\to\univ\\
        h&\mapsto\pi_1\circ h\sim\id_B.
    \end{align*}
    Then, for any\/ $\eta\colon Q(f)$ and any\/ $b\colon B$, we have
    \[
        \fun{transport}^Q(p,\eta)(b)
        \eq{\pi_1(g(b))\eq Bb}\fun{transport}^{Q_b}(p,\eta(b)),
    \]
    where\/ $Q_b\defeq\lambda h.\,\pi_1(h(b))\eq Bb$.
\end{lem}

\begin{proof}
    By path induction on $p$, it suffices to consider the case where $g\equiv f$ and $p\equiv\fun{refl}_f$. Since transport along reflexivity is the identity, the goal reduces to
    \[
        \eta(b)\eq{\pi_1(f(b))\eq Bb}\eta(b),
    \]
    which is witnessed by $\fun{refl}_{\eta(b)}$.
\end{proof}

\begin{thm}\fe\label{thm:pi-sections}
    Let $B\colon\univ$ be a type and $P\colon B\to\univ$ a type family. The type of dependent functions $\prod_{b\colon B}P(b)$ is equivalent to the type of right inverses of the first projection $\pi_1\colon\Sigma_BP\to B$. That is, there is an equivalence
    \[
        \Bigl(
            \prod_{b\colon B}P(b)
        \Bigr)
        \simeq
        \fun{rinv}(\pi_1).
    \]
\end{thm}

\begin{proof}
    We define the forward map by
    \begin{align*}
        \Phi\colon
            \Bigl(
                \prod_{b\colon B}P(b)
            \Bigr)
            &\to\fun{rinv}(\pi_1)\\
        \sigma&\mapsto\dep{
            \explain[B\to\Sigma_BP]{\lambda b.\,
            \dep{b,\sigma(b)}},\lambda b.\,\fun{refl}_b},
    \intertext{and the backward map by}
        \Psi\colon
            \fun{rinv}(\pi_1)&\to\prod_{b\colon B}P(b)\\
        \dep{g,\eta}&\mapsto\lambda b.\,
            \fun{transport}^P(\eta(b),\pi_2(g(b))).
    \end{align*}
    This definition is well-typed because $\eta(b)\colon\pi_1(g(b))\eq{B}b$, and so the transport map takes $P(\pi_1(g(b)))$ to $P(b)$.
    
    We must show that these maps are mutually inverse. For the forward-backward composition, we have
    \begin{align*}
        \Psi(\Phi(\sigma))(b)
            &\equiv\fun{transport}^P(\fun{refl}_b,\pi_2(\dep{b,\sigma(b)}))\\
            &\equiv\fun{transport}^P(\fun{refl}_b,\sigma(b))\\
            &\equiv \sigma(b).
    \end{align*}
    Hence, $\Psi(\Phi(\sigma))\equiv\sigma$, which implies $\Psi\circ\Phi\sim\id$.

    For the backward-forward composition, we obtain
    \[
        \Phi(\Psi(\dep{g,\eta}))
            \equiv\bigdep{
                \lambda b.\,\dep{b,
                    \fun{transport}^P(\eta(b),\pi_2(g(b)))},
                \lambda b.\,\fun{refl}_b
            }.
    \]
    Let
    \[
        f\defeq\lambda b.\,
            \dep{b,\fun{transport}^P(\eta(b),\pi_2(g(b)))}
    \]
    denote the function in the first component of the \rhs. We must show that
    \[
        \dep{g,\eta}
        \eq{\fun{rinv}(\pi_1)}\dep{f,\lambda b.\,\fun{refl}_b}.
    \]
    To establish a path $g\eq{B\to\Sigma_BP}f$, we will construct a homotopy $\vartheta\colon g\sim f$ and then apply function extensionality to get $\fun{funext}(\vartheta)$.
    
    Fix $b\colon B$. To construct a path $\vartheta(b)\colon g(b)\eq{\Sigma_BP}f(b)$, we need paths for both components:
    \begin{itemize}
        \item a path in the first component given by
        \[
            \eta(b)\colon\pi_1(g(b))\eq{B}b,
        \]
        
        \item a path in the second component between the transport of $\pi_2(g(b))$ along $\eta(b)$ and $\pi_2(f(b))$. Since $\pi_2(f(b))\equiv\fun{transport}^P(\eta(b),\pi_2(g(b)))$, this path is trivially produced by reflexivity.
    \end{itemize}
    This establishes the path $\vartheta(b)$, yielding $\fun{funext}(\vartheta)\colon g\eq{B\to\Sigma_BP}f$.
    
    To complete the proof, we must show that transporting the second component $\eta$ along $\fun{funext}(\vartheta)$ in the family
    \begin{align*}
        Q\colon(B\to\Sigma_BP)&\to\univ\\
        h&\mapsto \pi_1\circ h\sim\id_B
    \end{align*}
    yields $\lambda b.\,\fun{refl}_b$. By Lemma~\ref{lem:transport-Q}, we deduce
    \begin{align*}
        \fun{transport}^Q(\fun{funext}(\vartheta),\eta)(b)
            &\eq{Q_b(f)}\fun{transport}^{Q_b}
                (\fun{funext}(\vartheta),\eta(b)),
    \end{align*}
    where $Q_b\defeq\lambda h.\,\pi_1(h(b))\eq{B}b$.
    
    Applying Transport in Identity Types (Lemma~\ref{lem:transport-in-identity-types}) to the map $\lambda h.\,\pi_1(h(b))$ and the constant $\lambda h.\,b$, the \rhs becomes
    \begin{align*}
        \rhs
            &\eq{Q_b(f)}
                \fun{ap}_{\lambda h.\,\pi_1(h(b))}(
                    \fun{funext}(\vartheta)
                )^{-1}
                \ct\eta(b)
                \ct\fun{ap}_{\lambda h.\,b}(
                    \fun{funext}(\vartheta)
                )\\
            &\eq{Q_b(f)}
                \bigl(
                    \fun{ap}_{\pi_1}\circ\fun{ap}_{\ev b}
                        (\fun{funext}(\vartheta))
                \bigr)^{-1}
                \ct\eta(b)
                \ct\cancel{\fun{refl}_b}\\
            &\eq{Q_b(f)}
                \fun{ap}_{\pi_1}(\vartheta(b))^{-1}\ct\eta(b).
    \end{align*}
    Since the path in the first component of $\vartheta(b)$ is $\eta(b)$, Lemma~\ref{lem:ap_pi1-pair=} implies that $\fun{ap}_{\pi_1}(\vartheta(b))\eq{}\eta(b)$. Thus, we conclude that
    \[
        \rhs\eq{Q_b(f)}
            \eta(b)^{-1}\ct\eta(b)\eq{}\fun{refl}_b.
    \]
    By function extensionality, the transported homotopy is equal to $\lambda b.\,\fun{refl}_b$. This completes the proof.
    %
    
\end{proof}

\begin{prop}\fe\label{prop:side-inverses-are-contractible}
    If\/ $f\colon A\to B$ has a quasi-inverse\/ $\dep{g,(\eta,\vartheta)}$, then the types\/ $\fun{rinv}(f)$ and\/ $\fun{linv}(f)$ are contractible with respective centers\/ $\dep{g,\eta}$ and\/ $\dep{g,\vartheta}$.
\end{prop}

\begin{proof}
    Consider the maps
    \begin{align*}
        (f\circ\dummy)&\colon(C\to A)\to(C\to B)\\
        (\dummy\circ f)&\colon(B\to C)\to(A\to C).
    \end{align*}
    If\/ $g$ is a quasi-inverse of\/ $f$, then\/ $(g\circ\dummy)$ and\/ $(\dummy\circ g)$ are quasi-inverses of\/ $(f\circ\dummy)$ and\/ $(\dummy\circ f)$, respectively. Thus, they are equivalences and, by Theorem~\ref{thm:equivalence-fibers-are-contractible}, their fibers are contractible. 
    
    Therefore, the same theorem implies that\/ $\fun{rinv}(f)$ and\/ $\fun{linv}(f)$ are contractible because, by function extensionality, we have
    \begin{align*}
        \fun{rinv}(f)
            &\simeq\Bigl(
                \sum_{g\colon B\to A}
                    f\circ g\eq{B\to B}\id_B
            \Bigr)
            \equiv\fun{fib}_{f\circ\dummy}(\id_B)\\
        \fun{linv}(f)
            &\simeq\Bigl(
                \sum_{g\colon B\to A}
                    g\circ f\eq{A\to A}\id_A
            \Bigr)
            \equiv\fun{fib}_{\dummy\circ f}(\id_A).
    \end{align*}
    Since\/ $\dep{g,\fun{funext}(\eta)}$ and\/ $\dep{g,\fun{funext}(\vartheta)}$ are the respective centers of\/ $\fun{fib}_{f\circ\dummy}(\id_B)$ and\/ $\fun{fib}_{\dummy\circ f}(\id_A)$, the statement on the centers of\/ $\fun{rinv}(f)$ and\/ $\fun{linv}(f)$ follows.
\end{proof}

\begin{lem}\label{lem:qinv-reassociation}
    Let\/ $f\colon A\to B$ be a function. Then there is an equivalence
    \[
        \fun{qinv}(f)\simeq
            \sum_{\omega\colon\fun{rinv}(f)}\pi_1(\omega)\circ f\sim\id_A.
    \]
\end{lem}

\begin{proof}
    By definition, the type $\fun{qinv}(f)$ is given by
    \[
        \fun{qinv}(f)\equiv\sum_{h\colon B\to A}
            (f\circ h\sim\id_B)\times(h\circ f\sim\id_A).
    \]
    By expanding the cartesian product into a non-dependent $\Sigma$-type, this becomes definitionally equivalent to
    \[
        \sum_{h\colon B\to A}\;\sum_{\theta\colon f\circ h\sim\id_B}h\circ f\sim\id_A.
    \]
    Using $\Sigma$-associativity (see Proposition~\ref{prop:sigma-associativity}), this nested $\Sigma$-type is equivalent to
    \[
        \sum_{\omega\colon\sum_{h\colon B\to A}f\circ h\sim\id_B}\pi_1(\omega)\circ f\sim\id_A.
    \]
    This completes the proof because the index type is precisely $\fun{rinv}(f)$.
\end{proof}

\begin{lem}\label{lem:contractible-sigma-product}
    Let\/ $X\colon\univ$ be a type, and let\/ $A,B\colon X\to\univ$ be two type families over\/ $X$. If\/ $\Sigma_XA$ is contractible with center\/ $\dep{z,c}$, then there is an equivalence
    \[
        \sum_{x\colon X}A(x)\times B(x)\simeq B(z).
    \]
\end{lem}

\begin{proof}
    By rewriting the cartesian product as a non-dependent $\Sigma$-type, the left-hand side equals
    \[
        \sum_{x\colon X}\;\sum_{u\colon A(x)}B(x).
    \]
    Using $\Sigma$-associativity (see Proposition~\ref{prop:sigma-associativity}), this total space is equivalent to
    \[
        \sum_{\omega\colon\Sigma_XA}B(\pi_1(\omega)).
    \]
    By part b) of Theorem~\ref{thm:contractible-data-ignored}, we obtain
    \[
        \Bigl(
            \sum_{\omega\colon\Sigma_XA}B(\pi_1(\omega))
        \Bigr)
        \simeq B(\pi_1(\dep{z,c}))\equiv B(z).
    \]
\end{proof}

\begin{prop}\fe\label{prop:qinv-id-homotopy-equivalence}
    If\/ $f\colon A\to B$ is such that\/ $\fun{qinv}(f)$ is inhabited, then
    \[
        \fun{qinv}(f)\simeq(\id_A\sim\id_A).
    \]
\end{prop}

\begin{proof}
    Since $\fun{qinv}(f)$ is inhabited, $f$ has a quasi-inverse $\dep{g,(\eta,\vartheta)}$. Define
    \begin{align*}
        \sigma\colon(\id_A\sim\id_A)&\to(g\circ f\sim\id_A)\\
        \epsilon&\mapsto\vartheta\ct\epsilon.
    \intertext{The function $\sigma$ is an equivalence with inverse}
        (g\circ f\sim\id_A)&\to(\id_A\sim\id_A)\\
        \delta&\mapsto\vartheta^{-1}\ct\delta.
    \end{align*}
    By Proposition~\ref{prop:side-inverses-are-contractible}, $\fun{rinv}(f)$ is contractible with center $\dep{g,\eta}$. Thus, applying Lemma~\ref{lem:contractible-sigma-product} to the definition
    \[
        \fun{qinv}(f)
            \equiv\sum_{h\colon B\to A}
                (f\circ h\sim\id_B)\times(h\circ f\sim\id_A)
    \]
    yields an equivalence $\fun{qinv}(f)\simeq(g\circ f\sim\id_A)$, which, combined with the equivalence induced by $\sigma$, completes the proof.
\end{proof}

\begin{lem}\fe\label{lem:conditioned-id-id-homotopy}
    Suppose we have a type\/ $A$ with\/ $a\colon A$ and\/ $q\colon a\eq Aa$ such that
    \begin{enumerate}[a),font=\upshape]
        \item The type\/ $a\eq Aa$ is a set.
        \item For all\/ $x\colon A$ we have\/ $\lVert a\eq Ax\rVert$.
        \item For all\/ $p\colon a\eq Aa$ we have\/ $p\ct q\eq{a\eq Aa}q\ct p$.
    \end{enumerate}
    Then there exists a homotopy\/ $\eta\colon\id_A\sim\id_A$ with\/ $\eta(a)\equiv q$.
\end{lem}

\begin{proof}
    The proof consists of three parts.

    \textsc{Part 1.} {\itshape Given\/ $x,y\colon A$, the type\/ $x\eq Ay$ is a set.}

    We proceed by defining a map
    \[
        \gamma\colon
            \bigl(
                (a\eq Ax)\times(a\eq Ay)
            \bigr)
            \to\fun{isSet}(x\eq Ay).
    \]
    Given $r\colon a\eq Ax$ and $s\colon a\eq Ay$, we define
    \begin{align*}
        g_{r,s}\colon(a\eq Aa)&\to(x\eq Ay)\\
        p&\mapsto r^{-1}\ct p\ct s.
    \end{align*}
    Since this map has an inverse, it induces an equivalence
    \[
        (a\eq Aa)\simeq(x\eq Ay).
    \]
    By Exercise~\ref{exr:equivalent-to-set-is-set}, condition a) implies the existence of a witness 
    \[
        \omega_{r,s}\colon\fun{isSet}(x\eq Ay).
    \]
    Hence, we can define $\gamma(r,s)\defeq\omega_{r,s}$.

    Given that $\fun{isSet}(x\eq Ay)$ is a mere proposition [cf.~Remark~\ref{rem:ids-in-a-set-are-mere-propositions}], the map $\gamma$ factors through
    \[
        \fun{rec}_{\lVert\cdots\rVert}(\fun{isSet}(x\eq Ay),\gamma)
        \colon
            \lVert(a\eq Ax)\times(a\eq Ay)\rVert
            \to\fun{isSet}(x\eq Ay).
    \]
    By the lemma included in the solution to Exercise~\ref{exr:finite-axiom-of-choice}, condition b) implies that this truncated type is inhabited. Consequently, the type $\fun{isSet}(x\eq Ay)$ is inhabited as well.

    \textsc{Part 2.} {\itshape Given\/ $x\colon A$, the type
    \[
        B(x)\defeq\sum_{r\colon x\eq Ax}\;\prod_{s\colon a\eq Ax}
            r\eq{x\eq Ax}s^{-1}\ct q\ct s
    \]
    is a mere proposition.}
    
    To prove this, it suffices to construct a map
    \[
        (a\eq Ax)\to\fun{isProp}(B(x)).
    \]
    Let $t\colon a\eq Ax$. If $\dep{r,u}$ and $\dep{r',u'}$ are elements of $B(x)$, we must construct from $t$ a path between them.

    Since
    \[
    u(t)\colon r\eq{x\eq Ax}t^{-1}\ct q\ct t
    \quad\text{and}\quad
    u'(t)\colon r'\eq{x\eq Ax}t^{-1}\ct q\ct t,
    \]
    we obtain the path
    \[
        e\defeq u(t)\ct u'(t)^{-1}\colon r\eq{x\eq Ax}r'.
    \]
    To establish the equality $\dep{r,u}\eq{B(x)}\dep{r',u'}$, we must show that for the type family
    \[
        S(p)\defeq\prod_{w\colon a\eq Ax}
            p\eq{x\eq Ax}w^{-1}\ct q\ct w,
    \]
    we have $\fun{transport}^S(e,u)\eq{}u'$.
    By function extensionality, this reduces to showing that for all $w\colon a\eq Ax$,
    \[
        \fun{transport}^S(e,u)(w)\eq{}u'(w)
    \]
    in $r'\eq{x\eq Ax}w^{-1}\ct q\ct w$. But, since $x\eq Ax$ is a set, this identity type is a mere proposition, and the equality holds automatically.

    \textsc{Part 3.} {\itshape There is a homotopy\/ $\eta\colon\id_A\sim\id_A$ with\/ $\eta(a)\equiv q$.}
    
    Because $B(x)$ is a mere proposition for any $x\colon A$, to prove that $B(x)$ is inhabited, it suffices to construct a map $(a\eq Ax)\to B(x)$, and then invoke condition b).
    
    Consider
    \begin{align*}
        D&\colon\prod_{x\colon A}(a\eq Ax)\to\univ\\
        D(x,p)&\defeq B(x).
    \end{align*}
    By \textsc{\nameref{lpar:based-path-induction}}, to define a map
    \[
        f\colon\prod_{x\colon A}\;\prod_{p\colon a\eq Ax}B(x),
    \]
    it suffices to construct an element of $D(a,\fun{refl}_a)\equiv B(a)$.
    
    By definition,
    \[
        B(a)\equiv\sum_{r\colon a\eq Aa}\;\prod_{s\colon a\eq Aa}
            r\eq{a\eq Aa}s^{-1}\ct q\ct s.
    \]
    Condition c) implies that for $r\defeq q$, we have a path
    \[
        \beta(s)\colon q\eq{a\eq Aa}s^{-1}\ct q\ct s.
    \]
    Thus, $\dep{q,\beta}\colon B(a)$, as required. As a result, for every $x\colon A$, we obtain a map $f_x\colon(a\eq Ax)\to B(x)$. Given that $B(x)$ is a mere proposition, $f_x$ factors through $\lVert a\eq Ax\rVert$. And since this type is inhabited, we obtain a witness $b_x\colon B(x)$. Moreover, $b_a\equiv f_a(\fun{refl}_a)\equiv\dep{q,\beta}$.
    
    Therefore, we can define the homotopy $\eta\defeq\lambda x.\,\pi_1(b_x)$ that satisfies
    \[
        \eta(a)
            \equiv\pi_1(b_a)
            \equiv\pi_1(\dep{q,\beta})
            \equiv q.
    \]
\end{proof}

\begin{thm}\fe\label{thm:qinv-is-not-mere-proposition}
    There exist types\/ $A$ and\/ $B$ and a function\/ $f\colon A\to B$ such that\/ $\fun{qinv}(f)$ is not a mere proposition. 
\end{thm}

\begin{proof}
    By Proposition~\ref{prop:qinv-id-homotopy-equivalence}, it suffices to construct a type $A$ and a homotopy $\eta\colon\id_A\sim\id_A$ with $\eta\ne(\lambda x.\,\fun{refl}_x)$.

    For the type, we set
    \[
        A\defeq\sum_{X\colon\univ}\lVert\fun2\eq\univ X\rVert.
    \]
    Let $a\defeq\dep{\fun2,\lvert\fun{refl}_{\fun2}\rvert}\colon A$, and consider the equivalence $e\colon\fun2\simeq\fun2$ induced by the map $\fun{not}$ introduced in Exercise~\ref{exr:not}. By the lemma included in the solution to Exercise~\ref{exr:2-simeq-2-simeq-2}, $\fun{not}\ne\id_{\fun2}$. Consequently, $e$ is not the identity equivalence, meaning that the path $q\defeq\fun{ua}(e)$ satisfies $q\ne\fun{refl}_{\fun2}$.

    Since we intend to use Lemma~\ref{lem:conditioned-id-id-homotopy}, we need to prove the three conditions stated there.

    \textsc{Condition}~a)
    By Lemma~\ref{lem:total-space-with-mere-proposition-fibers} and the univalence axiom, there is an equivalence
    \[
        (a\eq Aa)\simeq(\fun2\simeq\fun2).
    \]
    Since $(\fun2\simeq\fun2)\simeq\fun2$ by Exercise~\ref{exr:2-simeq-2-simeq-2}, and $\fun2$ is a set by Theorem~\ref{thm:boolean-type-is-set}, condition~a) holds.

    \textsc{Condition}~b) Given $x\defeq\dep{X,u}$ with $u\colon\lVert\fun2\eq\univ X\rVert$, we need to verify that the truncated type
    \[
        \lVert\dep{\fun2,\lvert\fun{refl}_{\fun2}\rvert}
        \eq A\dep{X,u}\rVert
    \]
    is inhabited.
    
    We proceed by recursion on the truncated type $\lVert\fun2\eq\univ X\rVert$. Let $P\colon\univ\to\univ$ be the type family $X\mapsto\lVert\fun2\eq\univ X\rVert$. Given a path $p\colon\fun2\eq\univ X$, we can apply Lemma~\ref{lem:total-space-with-mere-proposition-fibers} to obtain a path $\epsilon_p\colon a\eq Ax$, because $P(X)$ is a mere proposition.
    
    It follows that $p\mapsto\lvert\epsilon_p\rvert$ is a map of type $(\fun2\eq\univ X)\to\lVert a\eq Ax\rVert$. Since the codomain is a mere proposition, this map factors through $\lVert\fun2\eq\univ X\rVert$. Evaluating this factored map at the given $u$ provides the desired witness in $\lVert a\eq Ax\rVert$.

    \textsc{Condition}~c) As established in the verification of condition~a), we know that $(a\eq Aa)\simeq\fun2$ [cf.~Proposition~\ref{prop:equivalence-is-an-equivalence-relation}~c)]. Thus, the identity type has exactly two inhabitants. Besides $q$, which trivially commutes with itself, the only other element corresponds to $\fun{ua}(\id_{\fun2})$, which is the reflexivity path and also commutes with $q$. Hence, condition~c) is satisfied.
    
    Since all three conditions hold, Lemma~\ref{lem:conditioned-id-id-homotopy} provides a homotopy $\eta\colon\id_A\sim\id_A$ such that $\eta(a)\equiv q$. Since $q\ne\fun{refl}_{\fun2}$, it follows that $\eta\ne(\lambda x.\,\fun{refl}_x)$.

    To complete the theorem, define $B\defeq A$ and $f\defeq\id_A$. By Proposition~\ref{prop:qinv-id-homotopy-equivalence}, the existence of the non-trivial self-homotopy $\eta$ of the identity map on $A$ implies that the type $\fun{qinv}(\id_A)$ is not a mere proposition.
    %

\end{proof}

\section{Half-Adjoint Equivalences}

Definition~\ref{defn:half-adjoint-equivalence} introduces the concept of a coherent quasi-inverse of an equivalence $f\colon A\to B$. Then, Theorem~\ref{thm:coherent-quasi-inverse} shows that such a quasi-inverse always exists. Here, we continue the study of this condition, introducing the type
\begin{equation}\label{eq:half-adjoint-homotopy}
    \fun{ishae}(f)\defeq
        \sum_{g\colon B\to A}\;
        \sum_{\eta\colon f\circ g\sim\id_B}\;
        \sum_{\vartheta\colon g\circ f\sim\id_A}
        f\lwct\vartheta\sim\eta\rwct f.
\end{equation}

The need for this definition arises from the failure of $\fun{qinv}(f)$ to be a mere proposition. The type $\fun{qinv}(f)$ is formed by three pieces of data: a map $g$, a right homotopy $\eta$, and a left homotopy $\vartheta$. As stated in Proposition~\ref{prop:side-inverses-are-contractible}, when $f$ is an equivalence, the space $\fun{linv}(f)$ of its left inverses is contractible. As a consequence, the total space $\fun{qinv}(f)$ reduces to the space of right homotopies $\eta$ [cf.~Theorem~\ref{thm:contractible-types}~b)].

As established in Proposition~\ref{prop:qinv-id-homotopy-equivalence}, the space $\fun{qinv}(f)$ is equivalent to $\id_A\sim\id_A$. If the space $A$ is not a set, there are multiple distinct, non-homotopic loops at each point, meaning there are multiple valid choices for $\eta$. Consequently, $\fun{qinv}(f)$ can have multiple inhabitants.

To ensure that the property of being an equivalence is a mere proposition, we must eliminate these extra degrees of freedom. This is achieved by introducing the coherence datum in \eqref{eq:half-adjoint-homotopy}, which tethers the right homotopy $\eta$ to the left homotopy $\vartheta$. While the space of all possible right homotopies is potentially large, the space of pairs $\dep{\eta,\tau}$, where $\tau\colon f\lwct\vartheta\sim\eta\rwct f$, is contractible. By pairing the contractible space of left inverses $\dep{g,\vartheta}$ with the contractible space of coherent right homotopies $\dep{\eta,\tau}$, all higher degrees of freedom are eliminated, and $\fun{ishae}(f)$ turns out to be a mere proposition.

On a practical level, if $\fun{isequiv}(f)$ were not a mere proposition, every time we used the fact that $f$ is an equivalence, we would have to explicitly keep track of which specific proof we were using.

By constructing $\fun{ishae}(f)$ as a mere proposition, we attain \textit{proof irrelevance}. We can construct an equivalence and immediately forget the specific construction of the right/left homotopies, because any two inhabitants of the mere proposition are propositionally equal. This saves the theory from collapsing into an unmanageable infinite tower of coherence conditions.


\begin{rem}\label{rem:why-half}
    As we discussed at the beginning of this section, the type of quasi-inverses reduces to the space of choices for the right homotopy $\eta$. This leaves $\eta$ as an unbound ``loose end,'' meaning the type $\fun{qinv}(f)$ is not generally a mere proposition.
    
    The left coherence condition $\tau\colon f\lwct\vartheta\sim\eta\rwct f$ fixes this loose end by tethering $\eta$ to $\vartheta$. Since the space of such coherent pairs $\dep{\eta,\tau}$ is contractible, the entire structure becomes a sequence of contractible extensions, leaving no degrees of freedom. Thus, the type $\fun{ishae}(f)$ is a mere proposition.
    
    If we wanted to reproduce the standard categorical adjunction by demanding both coherence conditions, we would have to define a type containing the data $\dep{g,\vartheta,\eta,\tau,\nu}$, where $\nu\colon g\lwct\eta\sim\vartheta\rwct g$.
    
    Since the base space of half-adjoint data $\dep{g,\vartheta,\eta,\tau}$ is contractible, the total space of this two-sided structure would reduce entirely to the space of choices for~$\nu$.
    
    However, $\nu$ is a homotopy between homotopies (a 2-path). Just as with $\eta$ in the case of quasi-inverses, there can be multiple choices for $\nu$, which re-introduces degrees of freedom in higher homotopy levels. In other words, the type of full two-sided adjoint equivalences generally fails to be a mere proposition.
    
    This is why the type-theoretic definition specifies only one coherence condition, rather than redundantly including both.
\end{rem}

\begin{thm}\label{thm:qinv->ishae}
    Given\/ $f\colon A\to B$, there is a map\/ $\fun{qinv}(f)\to\fun{ishae}(f)$.
\end{thm}

\begin{proof}
    This is Theorem~\ref{thm:coherent-quasi-inverse} restated after the introduction of $\fun{ishae}(f)$.
\end{proof}

\begin{lem}\label{lem:fib-equality}
    Let\/ $f\colon A\to B$ be a function. Then, given\/ $y\colon B$, and elements\/ $\dep{x,p},\dep{x',p'}\colon\fun{fib}_f(y)$, we have an equivalence
    \[
        \bigl(
            \dep{x,p}\eq{\fun{fib}_f(y)}\dep{x',p'}
        \bigr)
        \simeq
        \sum_{q\colon x=x'}
            \fun{ap}_f(q)\ct p'\eq{f(x)\eq By}p.
    \]
\end{lem}

\begin{proof}
    By the characterization of equality in $\Sigma$-types [cf.~\S\,\ref{equality-in-Sigma-types}], we have an equivalence
    \[
        \bigl(\dep{x,p}\eq{\fun{fib}_f(y)}\dep{x',p'}\bigr)
        \simeq
        \sum_{q\colon x=x'} \fun{transport}^P(q,p)\eq{f(x')\eq By}p',
    \]
    where $P(x)\defeq f(x)\eq By$.

    Since $P\equiv(z\mapsto z\eq By)\circ f$, we can use Lemma~\ref{lem:post-composition-transport} and Proposition~\ref{prop:transport-decomposition-for-paths} to compute the transport as
    \[
        \fun{transport}^P(q,p)
        \eq{f(x')\eq By}
        \fun{ap}_f(q)^{-1}\ct p.
    \]
    Thus, for any fixed $q\colon x\eq Ax'$, the identity above involving the transport is equivalent to
    \[
        \fun{ap}_f(q)^{-1}\ct p\eq{f(x')\eq By}p'.
    \]
    Left-concatenating both sides with $\fun{ap}_f(q)$ yields an equivalence to
    \[
        p\eq{f(x)\eq By}\fun{ap}_f(q)\ct p'.
    \]
    Taking the path inverse provides a further equivalence to
    \[
        \fun{ap}_f(q)\ct p'\eq{f(x)\eq By}p.
    \]
    Since this establishes an equivalence between the fibers for each $q\colon x\eq Ax'$, the induced map on the total spaces is an equivalence as well.
\end{proof}

\begin{rem}\label{rem:equivalence-fib-is-contractible}
    By Theorem~\ref{thm:equivalence-fibers-are-contractible}, if $f\colon A\to B$ is a half-adjoint equivalence, then for any $y\colon B$, the fiber $\fun{fib}_f(y)$ is contractible.
\end{rem}

\begin{defn}\label{def:coherences}
    For any function\/ $f\colon A\to B$, a right inverse\/ $\dep{g,\eta}\colon\fun{rinv}(f)$, and a left inverse\/ $\dep{g,\vartheta}\colon\fun{linv}(f)$, we define the types of \textsl{right coherences} and \textsl{left coherences} as
    \begin{align*}
        \fun{rcoh}_f(\dep{g,\eta})
            &\defeq \sum_{\vartheta\colon g\circ f\sim\id_A} f\lwct\vartheta\sim\eta\rwct f,\\
        \fun{lcoh}_f(\dep{g,\vartheta})
            &\defeq \sum_{\eta\colon f\circ g\sim\id_B} g\lwct\eta\sim\vartheta\rwct g.
    \end{align*}
\end{defn}

\begin{ntn}\label{ntn:fib-pullback}
    Given\/ $u\colon X\to Z$, $v\colon Z\to Y$, $w\colon X\to Y$, and\/ $\theta\colon v\circ u\sim w$, we introduce the function
    \begin{align*}
        \dep{u,\theta}
            &\colon\prod_{x\colon X}\fun{fib}_v(w(x))\\
        x&\mapsto\dep{u(x),\theta(x)}.
    \end{align*}
    Representing this in a homotopy diagram, the total space of the fibration acts as the categorical pullback\/ $X\times_Y Z$. The homotopy\/ $\theta$ witnesses the commutativity of the quadrangle, and\/ $\dep{u,\theta}$ is the map induced by the pair\/ $(\id_X,u)$:
    \[
        \begin{tikzcd}[arrows={|->}, column sep=1.1cm]
            x
                \arrow[rd, dashed, "\dep{u,\theta}"]
                \arrow[rdd, "\id_X"', bend right=30]
                \arrow[rrd, "u", bend left=20]\\
            & \dep{x,\dep{u(x),\theta(x)}}
                \arrow[r, "\pi_1\circ\pi_2"]
                \arrow[d, "\pi_1"']
                \arrow[rd, phantom, "\scriptstyle\theta(x)"]
            & u(x)
                \arrow[d, "v"]\\
            & x
                \arrow[r, "w"']
            & w(x)
        \end{tikzcd}
    \]
\end{ntn}

\begin{lem}\label{lem:side-coherences-equivalence}
    Let\/ $f\colon A\to B$, $g\colon B\to A$, $\eta\colon f\circ g\sim\id_B$, and\/ $\vartheta\colon g\circ f\sim\id_A$. Using the preceding notation we have
    \begin{align*}
        \fun{rcoh}_f(\dep{g,\eta})
            &\simeq
                \bigl(
                    \dep{g\circ f,\eta\rwct f}
                        \sim\dep{\id_A,\fun{refl}\rwct f}
                \bigr),\\
        \fun{lcoh}_f(\dep{g,\vartheta})
            &\simeq
                \bigl(
                    \dep{f\circ g,\vartheta\rwct g}
                        \sim\dep{\id_B,\fun{refl}\rwct g}
                \bigr).
    \end{align*}
\end{lem}

\begin{proof}
    By symmetry, it suffices to prove the first equivalence.

    First note that the expressions above are well-typed:
    \begin{enumerate}[a),font=\upshape]
        \item For $\dep{g\circ f,\eta\rwct f}$, take $u\defeq g\circ f$, $\theta\defeq\eta\rwct f$, $v\defeq f$, and $w\defeq f$.
        
        \item For $\dep{\id_A,\fun{refl}\rwct f}$, take $u\defeq\id_A$, $\theta\defeq\fun{refl}\rwct f$, $v\defeq f$, and $w\defeq f$.
    \end{enumerate}
    
    For the forward map of the equivalence, fix $\dep{\alpha,\beta}\colon\fun{rcoh}_f(\dep{g,\eta})$. We have to construct the corresponding homotopy.

    By definition, $\alpha\colon g\circ f\sim\id_A$ and $\beta\colon f\lwct\alpha\sim\eta\rwct f$.
    
    Given $x\colon A$, we need a path
    \[
        \dep{g(f(x)),\eta_{f(x)}}
            \eq{\fun{fib}_f(f(x))}
            \dep{x,\fun{refl}_{f(x)}}.
    \]
    Let $p\defeq\alpha(x)$ and $q\defeq\beta(x)$. Then 
    \[
        p\colon g(f(x))\eq Ax
        \quad\text{and }\quad
        q\colon\fun{ap}_f(p)
            \eq{f(g(f(x)))\eq Bf(x)}\eta_{f(x)}.
    \]
    By Lemma~\ref{lem:fib-equality}, constructing the required path in $\fun{fib}_f(f(x))$ is equivalent to providing an element of the $\Sigma$-type
    \[
        \sum_{r\colon g(f(x))\eq Ax}
            \fun{ap}_f(r)\ct\fun{refl}_{f(x)}
            \eq{f(g(f(x)))\eq Bf(x)}
            \eta_{f(x)}.
    \]
    Since $\fun{ru}(\fun{ap}_f(p))\colon\fun{ap}_f(p)\ct\fun{refl}_{f(x)}\eq{f(g(f(x)))\eq Bf(x)}\fun{ap}_f(p)$, we can take $r\defeq p$, form the required dependent pair $\dep{p,\fun{ru}(\fun{ap}_f(p))\ct q}$.

    For the backward map, consider $\tau\colon\dep{g\circ f,\eta\rwct f}\sim\dep{\id_A,\fun{refl}\rwct f}$, and let us construct a witness of $\fun{rcoh}_f(\dep{g,\eta})$.

    According to the definition, we need two homotopies 
    \[
        \varepsilon\colon g\circ f\sim\id_A
        \quad\text{and}\quad
        \nu\colon f\lwct\varepsilon\sim\eta\rwct f.
    \]
    Evaluating at $x\colon A$ and following Notation~\ref{ntn:fib-pullback}, we obtain
    \[
        \tau(x)\colon\dep{g(f(x)),\eta_{f(x)}}
            \eq{\fun{fib}_f(f(x))}
            \dep{x,\fun{refl}_{f(x)}}.
    \]
    Applying the forward direction of Lemma~\ref{lem:fib-equality} to $\tau(x)$, we obtain a pair $\dep{p,q}$ consisting of two paths
    \[
        p\colon g(f(x))\eq Ax
        \quad\text{and}\quad
        q\colon\fun{ap}_f(p)\ct\fun{refl}_{f(x)}
        \eq{f(g(f(x)))\eq Bf(x)}\eta_{f(x)}.
    \]
    Thus, we can define
    \[
        \varepsilon(x)\defeq p
        \quad\text{and}\quad
        \nu(x)\defeq\fun{ru}(\fun{ap}_f(p))^{-1}\ct q.
    \]

    Verifying that these maps are quasi-inverses of each other is straightforward.
\end{proof}

\begin{lem}\fe\label{lem:rcoh-is-contractible}
    If\/ $f\colon A\to B$ is an equivalence, then for any\/ $\dep{g,\eta}\colon\fun{rinv}(f)$, the type\/ $\fun{rcoh}_f(\dep{g,\eta})$ is contractible.
\end{lem}

\begin{proof}
    According to Notation~\ref{ntn:fib-pullback}, we have
    \[
        \dep{g\circ f,\eta}\colon
            \prod_{x\colon A}\fun{fib}_f(f(x)).
    \]
    Since the fibers $\fun{fib}_f$ of $f$ are contractible by Theorem~\ref{thm:equivalence-fibers-are-contractible}, we deduce from part a) of Theorem~\ref{thm:contractible-types} that the $\Pi$-type above is contractible.

    For the same reasons, $\dep{\id_A,\fun{refl}\rwct f}$ is contractible. Hence, by part g) of the same theorem, we deduce that $\dep{g\circ f,\eta}\sim\dep{\id_A,\fun{refl}\rwct f}$ is contractible. The conclusion follows from Lemma~\ref{lem:side-coherences-equivalence}.
\end{proof}

\begin{thm}\fe\label{thm:ishae-is-mere-proposition}
    For any\/ $f\colon A\to B$, the type\/ $\fun{ishae}(f)$ is a mere proposition.
\end{thm}

\begin{proof}
    By $\Sigma$-associativity, Proposition~\ref{prop:sigma-associativity}, we have
    \begin{align*}
        \fun{ishae}(f)
            &\defeq
                \sum_{g\colon B\to A}\;
                \sum_{\eta\colon f\circ g\sim\id_B}
                    \fun{rcoh}_f(\dep{\fun{g,\eta}})\\
            &\eq{}
                \sum_{\omega\colon\fun{rinv}(f)}\fun{rcoh}_f(\omega).
    \end{align*}
    If $\fun{ishae}(f)$ is inhabited, then $\fun{rinv}(f)$ is contractible by Proposition~\ref{prop:side-inverses-are-contractible}, and $\fun{rcoh}(f)$ is contractible by Lemma~\ref{lem:rcoh-is-contractible}. Thus, the conclusion follows from part~d) of Theorem~\ref{thm:contractible-types}.
\end{proof}

\section{Bi-invertible Maps}

\begin{defn}
    We say that $f\colon A\to B$ is \textsl{bi-invertible} if it has both a left inverse and a right inverse, i.e., if the type
    \[
        \fun{biinv}(f)\defeq\fun{linv}(f)\times\fun{rinv}(f)
    \]
    is inhabited.
\end{defn}

\begin{thm}\fe\label{thm:biinv-is-mere-proposition}
    For any\/ $f\colon A\to B$, the type\/ $\fun{biinv}(f)$ is a mere proposition.
\end{thm}

\begin{proof}
    It suffices to show that, in the case where $\fun{biinv}(f)$ is inhabited, it is contractible.
    If $\fun{biinv}(f)$ is inhabited, $f$ has both a left and a right inverse. This implies that $f$ is quasi-invertible.
    Hence, by Proposition~\ref{prop:side-inverses-are-contractible}, both $\fun{rinv}(f)$ and $\fun{linv}(f)$ are contractible.
    Thus, the conclusion follows from part c) of Theorem~\ref{thm:contractible-types}.
    
    %
    
\end{proof}

\begin{rem}
    While\/ $\fun{ishae}(f)$ requires the addition of a $2$-dimensional coherence homotopy to reach the status of a mere proposition, $\fun{biinv}(f)$ achieves the same using only $1$-dimensional homotopies.
\end{rem}

\begin{cor}\fe
    Let\/ $f\colon A\to B$ be a function. Then\/ $\fun{biinv}(f)\simeq\fun{ishae}(f)$.
\end{cor}

\begin{proof}
    Given that both types are mere propositions, by Lemma~\ref{lem:mere-proposition-equivalence}, it suffices to show a forward and a backward map.

    By Theorem~\ref{thm:qinv->ishae}, there is a map $\fun{qinv}(f)\to\fun{ishae}(f)$. Composing this with the trivial map $\fun{biinv}(f)\to\fun{qinv}(f)$, one obtains a forward map.
    %
    
\end{proof}

\section{Contractible Fibers}
Recall from Definition~\ref{defn:fiber-over-y} that a map is contractible when its fibers are contractible. As observed in Remark~\ref{rem:equivalence-fib-is-contractible}, the fibers of a half-adjoint equivalence are contractible, which means that there is a map $\fun{ishae}(f)\to\fun{isContrMap}(f)$.

\begin{thm}\label{thm:isContr->ishae}
    For any\/ $f\colon A\to B$, we have\/ $\fun{isContrMap}(f)\to\fun{ishae}(f)$.
\end{thm}

\begin{proof}
    By abuse of notation,
    \[
        \fun{isContrMap}(f)\equiv
            \prod_{y\colon B}\;
            \sum_{\dep{a,p}\colon\fun{fib}_f(y)}\;
            \prod_{\dep{x,q}\colon\fun{fib}_f(y)}
            \dep{a,p}\eq{\fun{fib}_f(y)}\dep{x,q}.
    \]
    On the other hand,
    \[
        \fun{ishae}(f)\equiv
            \sum_{g\colon B\to A}\;
            \sum_{\eta\colon f\circ g\sim\id_B}\;
            \sum_{\vartheta\colon g\circ f\sim\id_A}
            f\lwct\vartheta\sim\eta\rwct f.
    \]
    Therefore, we can define the map as
    \begin{align*}
        \fun{isContrMap}(f)&\to\fun{ishae}(f)\\
        \alpha&\mapsto\dep{g,\dep{\eta,\dep{\vartheta,\tau}}},
    \end{align*}
    where
    \begin{align*}
        g(y)
            &\defeq\pi_1(\pi_1(\alpha(y)))\\
            &\equiv\pi_1(\dep{a,p})\\
            &\equiv a,\\[1ex]
        \eta(y)
            &\defeq\pi_2(\pi_1(\alpha(y)))\\
            &\equiv\pi_2(\dep{a,p})\\
            &\equiv p.
    \end{align*}
    To construct $\vartheta(x)\colon g(f(x))\eq A x$ and $\tau(x)\colon\fun{ap}_f(\vartheta(x))\eq{f(g(f(x)))\eq B f(x)}\eta(f(x))$, set 
    \[
        a\defeq\pi_1(\pi_1(\alpha(f(x))))
        \quad\text{and}\quad
        p\defeq\pi_2(\pi_1(\alpha(f(x)))),
    \]
    and observe that $a\equiv g(f(x))$ and $p\equiv\eta(f(x))$.
    
    Evaluating the contraction homotopy $\pi_2(\alpha(f(x)))$ at the canonical element $\dep{x,\fun{refl}_{f(x)}}$ yields a path 
    \[
        s\colon
            \dep{a,p}
            \eq{\fun{fib}_f(f(x))}
            \dep{x,\fun{refl}_{f(x)}}.
    \]
    By Lemma~\ref{lem:fib-equality}, we can use $s$ to construct a dependent pair $\dep{r,q}$, where
    \[
        r\colon a\eq A x
        \quad\text{and}\quad
        q\colon
            \fun{ap}_f(r)\ct\fun{refl}_{f(x)}
            \eq{f(a)\eq B f(x)}
            p.
    \]
    After replacing $q$ with $\fun{ru}(\fun{ap}_f(r))^{-1}\ct q$, we may assume that
    \[
        r\colon g(f(x))\eq A x
        \quad\text{and}\quad
        q\colon\fun{ap}_f(r)\eq{f(a)\eq B f(x)}\eta(f(x)),
    \]
    which we can use to define $\vartheta(x)\defeq r$ and $\tau(x)\defeq q$.
    %
    
\end{proof}

\begin{lem}\label{lem:mere-prop-iff-inhabitable->contractibe}
    For any type\/ $A\colon\univ$, the types\/ $\fun{isProp}(A)$ and\/ $A\to\fun{isContr}(A)$ are logically equivalent.
\end{lem}

\begin{proof}
    For the forward map, consider the assignment
    \begin{align*}
        \fun{isProp}(A)&\to(A\to\fun{isContr}(A))\\
        p&\mapsto\lambda x.\,\dep{x,p(x)},
    \end{align*}
    which is well-defined because
    \[
        p\colon\prod_{x\colon A}\prod_{y\colon A}x\eq Ay.
    \]
    For the backward map, fix $f\colon A\to\fun{isContr}(A)$. To produce a witness of $\fun{isProp}(A)$, we must construct a path $x\eq Ay$ for any given elements $x,y\colon A$. 
    
    Since $x\colon A$, evaluating $f$ at $x$ yields a proof that $A$ is contractible. Therefore, $c\defeq\pi_1(f(x))$ is a center of contraction and
    \[
        \pi_2(f(x))\colon\prod_{z\colon A}c\eq Az.
    \]
    Hence,
    \[
        \explain[x\eq Ac]{\pi_2(f(x))(x)^{-1}}
        \ct
        \explain[c\eq Ay]{\pi_2(f(x))(y)}\colon x\eq Ay
    \]
    is the required path.
\end{proof}

\begin{lem}\we\label{lem:isContrMap-is-mere-proposition}
    For any\/ $f\colon A\to B$, the type\/ $\fun{isContrMap}(f)$ is a mere proposition.
\end{lem}

\begin{proof}
    By Lemma~\ref{lem:mere-prop-iff-inhabitable->contractibe}, to prove that $\fun{isContrMap}(f)$ is a mere proposition, we assume that there is an element
    \[
        c\colon\fun{isContrMap}(f).
    \]
    By definition,
    \[
        \fun{isContrMap}(f)\equiv\prod_{y\colon B}\fun{isContr}(\fun{fib}_f(y)).
    \]
    Hence, $c(y)$ is a proof that $\fun{fib}_f(y)$ is contractible for every $y\colon B$. 
    
    By Theorem~\ref{thm:isContr-is-a-mere-proposition}, the type $\fun{isContr}(\fun{fib}_f(y))$ is a mere proposition for each $y\colon B$. 
    
    Since $\fun{isContr}(\fun{fib}_f(y))$ is both a mere proposition and inhabited by $c(y)$, it is contractible for every $y\colon B$. 
    
    We can now apply the weak function extensionality axiom to the family of types $P(y)\equiv\fun{isContr}(\fun{fib}_f(y))$. Since each fiber $P(y)$ is contractible, the product type $\prod_{y\colon B}P(y)$ is contractible.
    
    Because this product type is definitionally equal to $\fun{isContrMap}(f)$, we conclude that $\fun{isContrMap}(f)$ is contractible, hence, is a mere proposition.
    
\end{proof}

\begin{thm}\we\label{thm:isContr-ishae-equivalence}
    For any\/ $f\colon A\to B$, we have\/ $\fun{isContrMap}(f)\simeq\fun{ishae}(f)$.
\end{thm}

\begin{proof}
    This follows from the previous lemma, Lemma~\ref{lem:mere-proposition-equivalence}, and the fact that we have maps $\fun{isContrMap}(f)\to\fun{ishae}(f)$ and $\fun{ishae}(f)\to\fun{isContrMap}(f)$.
    %
    
\end{proof}

\begin{cor}\fe\label{cor:map-followed-by-map-to-isequiv}
    Let\/ $f\colon A\to B$ be a map, and suppose we have another map\/ $B\to\fun{isequiv}(f)$. Then\/ $f$ is an equivalence.
\end{cor}

\begin{proof}
    By the theorem, to prove that $f$ is an equivalence, it suffices to construct a witness of
    \[
        \fun{isContrMap}(f)\equiv\prod_{y\colon B}\fun{isContr}(\fun{fib}_f(y)).
    \]
    Fix $y\colon B$. If $h\colon B\to\fun{isContrMap}(f)$ is the given map, then $h(y)\colon\fun{isContrMap}(f)$. In particular,
    \[
        h(y)(y)\colon\fun{isContr}(\fun{fib}_f(y)).
    \]
    Thus, $\lambda y.\,h(y)(y)$ is exactly the required witness of $\fun{isContrMap}(f)$.
    %
    
\end{proof}

\section{Surjections and Embeddings}

\begin{defn}
    Let\/ $f\colon A\to B$. We say that\/ $f$ is
    \begin{enumerate}[-]
        \item \textsl{surjective} (or a \textsl{surjection}) if\/ $\lVert\fun{fib}_f(b)\rVert$ is inhabited for every\/ $b\colon B$;
        \item an \textsl{embedding} if for every\/ $x,y\colon A$ the function
        \[
            \fun{ap}_f\colon(x\eq Ay)\to(f(x)\eq Bf(y))
        \]
        induces an equivalence.
    \end{enumerate}
\end{defn}

\begin{rems} Suppose that $A$ and $B$ are sets, and let $f\colon A\to B$ be a function.
    \begin{enumerate}[-]
        \item If $f$ is a surjection, the hypothesis that every fiber is merely inhabited is expressed by the type $\prod_{b\colon B}\lVert\fun{fib}_f(b)\rVert$. In contrast, a retraction requires an explicit, constructive selection of preimages, corresponding to the untruncated product $\prod_{b\colon B}\fun{fib}_f(b)$.
    
        Since we cannot extract the explicit data required to build a retraction from a truncated type, AC (for sets) bridges this exact gap by asserting the implication
        \[
            \Bigl(
                \prod_{b\colon B}\lVert\fun{fib}_f(b)\rVert
            \Bigr)
            \to\Bigl\lVert
                \prod_{b\colon B}\fun{fib}_f(b)
            \Bigr\rVert.
        \]
        This states precisely that if every fiber of a function between sets is merely inhabited, then the type of its retractions is merely inhabited as well.
    
        It should be noted that for every $y\colon B$, the fiber $\fun{fib}_f(y)$ is a set. This follows from part~e) of Theorem~\ref{thm:sets}, because $A$ is a set and, since $B$ is a set, the identity type $f(x)\eq B y$ is a mere proposition (and hence a set) for any $x\colon A$.

        \item By Lemma~\ref{lem:mere-proposition-equivalence}, $f$ is an embedding if, and only if, the function type
        \[
            (f(x)\eq Bf(y))\to(x\eq Ay)
        \]
        is inhabited.
    \end{enumerate}
\end{rems}

\begin{lem}\label{lem:embedding-fibers-are-mere-propositions}
    If\/ $f\colon A\to B$ is an embedding, then its fibers are mere propositions. 
\end{lem}

\begin{proof}
    Let $b\colon B$, and let $\dep{x,p}$ and $\dep{y,q}$ be two elements in $\fun{fib}_f(b)$.
    To show that $\fun{fib}_f(b)$ is a mere proposition, it suffices to prove that the identity type $\dep{x,p}\eq{\fun{fib}_f(b)}\dep{y,q}$ is contractible.
    By Lemma~\ref{lem:fib-equality}, we have an equivalence
    \[
        \bigl(
            \dep{x,p}\eq{\fun{fib}_f(b)}\dep{y,q}
        \bigr)
        \simeq
        \sum_{r\colon x\eq Ay}
            \fun{ap}_f(r)\ct q\eq{f(x)\eq Bb}p.
    \]
    Therefore,
    \begin{align*}
        \bigl(
            \dep{x,p}\eq{\fun{fib}_f(b)}\dep{y,q}
        \bigr)
        &\simeq
        \sum_{r\colon x\eq Ay}
            \fun{ap}_f(r)\eq{f(x)\eq Bf(y)}p\ct q^{-1}\\
        &\equiv\fun{fib}_{\fun{ap}_f}(p\ct q^{-1}).
    \end{align*}    
    Since $\fun{ap}_f$ is an equivalence, it has contractible fibers.
    %
    
\end{proof}

\begin{thm}\fe
    A function\/ $f\colon A\to B$ induces an equivalence if, and only if, it is both surjective and an embedding.
\end{thm}

\begin{proof}
    If $f$ induces an equivalence, it is a surjection because, by Remark~\ref{rem:equivalence-fib-is-contractible}, its fibers are contractible, hence $\lVert\fun{fib}_f(b)\rVert$ is contractible and inhabited by its center. It is an embedding because, by Proposition~\ref{prop:ap-properties}, $\fun{ap}_{f^{-1}}$ is a quasi-inverse of $\fun{ap}_f$.

    To prove that $f$ is an equivalence, by Theorem~\ref{thm:isContr-ishae-equivalence}, it suffices to show that its fibers are contractible. By Lemma~\ref{lem:embedding-fibers-are-mere-propositions}, the fibers of $f$ are mere propositions. Therefore, the canonical map $\fun{squash}\colon\fun{fib}_f(b)\to\lVert\fun{fib}_f(b)\rVert$ has a quasi-inverse. Since $f$ is a surjection, it follows that $\fun{fib}_f(b)$ is inhabited, hence contractible by Lemma~\ref{lem:contractible-equivalences}.
    %
    
\end{proof}

\section{Closure Properties of Equivalences}

\begin{defn}\label{defn:function-retract}
    A function $g\colon A\to B$ is said to be a \textsl{retract} of a function $f\colon X\to Y$ if there is a diagram
    \begin{equation}\label{tik:retraction-1}
        \begin{tikzcd}
            A
                    \arrow[rr,"\scriptstyle\eta"',phantom,bend left=30]
                    \arrow[rr,"\id_A",bend left=50]
                    \arrow[r,"s"]
                    \arrow[d,"g"']
                    \arrow[rd,
                        phantom,
                        "\scriptstyle\vartheta"]
                &X
                    \arrow[rd,
                        phantom,
                        "\scriptstyle\varepsilon"]
                    \arrow[r,"r"]
                    \arrow[d,"f"]
                &A
                    \arrow[d,"g"]\\
            B
                    \arrow[r,"s'"']
                    \arrow[rr,
                        "\scriptstyle\eta'"',
                        bend right=25,
                        phantom]
                    \arrow[rr,"\id_B"',bend right=50]
                &Y
                    \arrow[r,"r'"']
                &B
        \end{tikzcd}
        \qquad
        \begin{aligned}
            &\eta\hphantom'\colon r\circ s\sim\id_A\\
            &\eta'\colon r'\circ s'\sim\id_B\\
            &\vartheta\hphantom'\colon f\circ s\sim s'\circ g\\
            &\varepsilon\hphantom'\colon g\circ r\sim r'\circ f
        \end{aligned}        
    \end{equation}
    and a homotopy $\tau$ witnessing the commutativity of the square
    \begin{equation}\label{tik:retraction-2}
        \begin{tikzcd}
            g\circ r\circ s
                    \arrow[r,"\varepsilon\rwct s"]
                    \arrow[d,"g\lwct\eta"']
                    \arrow[rd,
                        phantom,
                        "\scriptstyle\tau"]
                &r'\circ f\circ s
                    \arrow[d,"r'\lwct\vartheta"]\\
            g
                    \arrow[r,"(\eta'\rwct g)^{-1}"']
                &r'\circ s'\circ g.
        \end{tikzcd}
    \end{equation}
\end{defn}

\begin{prop}
    Let\/ $A$ and\/ $B$ be types. Then\/ $B$ is a retract of\/ $A$ if, and only if, the constant map\/ $B\to\fun1$ is a retract of\/ $A\to\fun1$.
\end{prop}

\begin{proof}
    Suppose that $B$ is a retract of $A$. By Definition~\ref{defn:type-retract}, there exist a section $s\colon B\to A$, a retraction $r\colon A\to B$, and a homotopy $\eta\colon r\circ s\sim\id_B$.

    The data for the map retraction translates into the following diagram, which commutes trivially:
    \[
        \begin{tikzcd}
            B
                \arrow[rr, "\scriptstyle\eta"', phantom, bend left=30]
                \arrow[rr, "\id_B", bend left=50]
                \arrow[r, "s"]
                \arrow[d]
                \arrow[rd, phantom, "\scriptstyle\vartheta"]
            & A
                \arrow[rd, phantom, "\scriptstyle\varepsilon"]
                \arrow[r, "r"]
                \arrow[d]
            & B
                \arrow[d] \\
            \fun1
                \arrow[r, "\id_{\fun1}"']
                \arrow[rr, "\scriptstyle\fun{refl}"', phantom, bend right=25]
                \arrow[rr, "\id_{\fun1}"', bend right=50]
            & \fun1
                \arrow[r, "\id_{\fun1}"']
            & \fun1
        \end{tikzcd}
    \]
    where $\vartheta$ and $\varepsilon$ are the constant homotopies defined by $b\mapsto\fun{refl}_\ast$ and $a\mapsto\fun{refl}_\ast$, respectively.

    The coherence condition \eqref{tik:retraction-2} is satisfied automatically since all paths in $\fun1$ are trivially equal by contractibility. Thus, by Definition~\ref{defn:function-retract}, the map $B\to\fun1$ is a retract of $A\to\fun1$.

    Conversely, if $B\to\fun1$ is a retract of $A\to\fun1$, the retraction data inherently provides maps $s\colon B\to A$ and $r\colon A\to B$, along with a homotopy $\eta\colon r\circ s\sim\id_B$. This is precisely the data required by Definition~\ref{defn:type-retract} to establish that $B$ is a retract of $A$.
\end{proof}

\begin{lem}\label{lem:retract-fib}
    Suppose that a function\/ $g\colon A\to B$ is a retract of a function\/ $f\colon X\to Y$, and let\/ $s'\colon B\to Y$ be the section provided by\/ {\upshape Definition~\ref{defn:function-retract}}. Then for every\/ $b\colon B$, the fiber\/ $\fun{fib}_g(b)$ is a retract of the fiber\/ $\fun{fib}_f(s'(b))$.
\end{lem}

\begin{proof}
    Fix $b\colon B$. To define the forward function between the fibers, consider
    \begin{alignat*}4
        \sigma&\colon\fun{fib}_g(b)\to\fun{fib}_f(s'(b)),
            &\quad\dep{a,p}
            &\mapsto\dep{
                s(a),
                \vartheta(a)\ct\fun{ap}_{s'}(p)
            }.
    \intertext{This is well-defined because $\vartheta(a)\colon f(s(a))\eq{Y} s'(g(a))$ and $p\colon g(a)\eq{B} b$. For the backward map, consider}
        \rho&\colon\fun{fib}_f(s'(b))\to\fun{fib}_g(b),
            &\quad\dep{x,q}
            &\mapsto\dep{
                r(x),
                \varepsilon(x)\ct\fun{ap}_{r'}(q)\ct\eta'(b)
            },
    \intertext{which is well-defined because}
        q&\colon f(x)\eq{Y} s'(b),\\
        \fun{ap}_{r'}(q)&\colon r'(f(x))\eq{B} r'(s'(b)),\\
        \eta'(b)&\colon r'(s'(b))\eq{B} b,\\
        \varepsilon(x)&\colon g(r(x))\eq{B} r'(f(x)).
    \end{alignat*}
    It remains to be shown that $\rho\circ\sigma\sim\id_{\fun{fib}_g(b)}$.
    
    Take $\dep{a,p}\colon\fun{fib}_g(b)$. The first component of $\rho\circ\sigma(\dep{a,p})$ is $r(s(a))$. The second component, which we will call $p'$, yields the following computation:
    \begin{align*}
        p' &\equiv
                \varepsilon(s(a))
                    \ct\fun{ap}_{r'}(
                        \vartheta(a)
                        \ct\fun{ap}_{s'}(p)
                    )
                    \ct\eta'(b)\\
            &\eq{g(r(s(a)))\eq{B} b}
                \varepsilon\rwct s(a)
                    \ct r'\lwct\vartheta(a)
                    \ct\fun{ap}_{r'}(
                        \fun{ap}_{s'}(p)
                    )\ct\eta'(b)\\
            &\eq{g(r(s(a)))\eq{B} b}
                (
                    \varepsilon\rwct s
                    \ct r'\lwct\vartheta
                )(a)
                    \ct\fun{ap}_{r'\circ s'}(p)
                    \ct\eta'(b)\\
            &\eq{g(r(s(a)))\eq{B} b}
                (
                    g\lwct\eta
                    \ct(\eta'\rwct g)^{-1}
                )(a)
                    \ct\fun{ap}_{r'\circ s'}(p)
                    \ct\eta'(b)
                    &&\text{; by }\tau\\
            &\eq{g(r(s(a)))\eq{B} b} 
                \fun{ap}_g(\eta(a))
                    \ct\eta'(g(a))^{-1}
                    \ct\fun{ap}_{r'\circ s'}(p)
                    \ct\eta'(b).
    \end{align*}
    The naturality equation \eqref{eq:nat-transformation} of the homotopy $\eta'\colon r'\circ s'\sim\id_B$ applied to $p\colon g(a)\eq{B} b$ yields
    \[
        \fun{ap}_{r'\circ s'}(p)\ct\eta'(b)
        \eq{r'(s'(g(a)))\eq{B} b}\eta'(g(a))\ct p.
    \]
    Substituting this into the expression for $p'$ above gives:
    \[
        p' \eq{g(r(s(a)))\eq{B} b} \fun{ap}_g(\eta(a))\ct p.
    \]
    Given that $\eta(a)\colon r(s(a))\eq{A} a$, the equality $\dep{r(s(a)),p'}\eq{\fun{fib}_g(b)}\dep{a,p}$ is a direct consequence of Lemma~\ref{lem:fib-equality}. Hence, $\rho\circ\sigma(\dep{a,p})\eq{\fun{fib}_g(b)}\dep{a,p}$.
    %
    
\end{proof}

\begin{thm}
    A retract of an equivalence is an equivalence.
\end{thm}

\begin{proof}
    Let $g$ be a retract of an equivalence $f$. By Theorem~\ref{thm:equivalence-fibers-are-contractible}, every fiber of $f$ is contractible. Furthermore, Lemma~\ref{lem:retract-fib} establishes that each fiber of $g$ is a retract of a corresponding fiber of $f$. Since retracts of contractible types are themselves contractible by Proposition~\ref{prop:contractible-retracts}, it follows that all fibers of $g$ are contractible. Thus, $g$ is an equivalence by Theorem~\ref{thm:isContr->ishae}.
    %

\end{proof}

\begin{defn}\label{defn:total-map}
    A type family $P\colon A\to\univ$ defines the \textsl{fibration} over $A$ given by the projection
    \[
        \pi_1\colon\Bigl(\sum_{x\colon A}P(x)\Bigr)\to A.
    \]
    From this point of view, given two type families $P,Q\colon A\to\univ$, we refer to a function 
    \[
        f\colon\prod_{x\colon A}P(x)\to Q(x)
    \]
    as a \textsl{fiberwise map} or \textsl{fiberwise transformation}. Furthermore, the \textsl{total} function of such a map is defined as
    \[
        \fun{total}(f)\defeq
            \lambda w.\,
                \dep{
                    \pi_1(w),
                    f(\pi_1(w),\pi_2(w))
                }
                \colon
                    \Bigl(
                        \sum_{x\colon A}P(x)
                    \Bigr)\to
                    \Bigl(
                        \sum_{x\colon A}Q(x)
                    \Bigr).
    \]
    The map $f$ is a \textsl{fiberwise equivalence} if $f(x)\colon P(x)\to Q(x)$ is an equivalence for every $x\colon A$.
\end{defn}

\begin{thm}\label{thm:total-fib-equivalence}
    Suppose that\/ $f$ is a fiberwise transformation between families\/ $P$ and\/ $Q$ over a type\/ $A$, and let\/ $x\colon A$ and\/ $v\colon Q(x)$. Then we have an equivalence
    \[
        \fun{fib}_{\fun{total}(f)}(\dep{x, v}) \simeq \fun{fib}_{f(x)}(v).
    \]
\end{thm}

\begin{proof}
    By definition, the fiber of $\fun{total}(f)$ over $\dep{x,v}$ is
    \[
        \fun{fib}_{\fun{total}(f)}(\dep{x,v})
        \equiv\sum_{w\colon\Sigma_AP}
            \fun{total}(f)(w)\eq{\Sigma_AQ}\dep{x,v}.
    \]
    By expanding $w$ as a pair $\dep{y, u}$ and using the characterization of equality in $\Sigma$-types, we have
    \begin{align*}
        \fun{fib}_{\fun{total}(f)}(\dep{x,v}) 
            &\simeq\sum_{y\colon A}\;\sum_{u\colon P(y)} 
            \dep{y,f(y,u)}\eq{\Sigma_AQ}\dep{x,v}\\
            &\simeq
                \sum_{y\colon A}\;
                \sum_{u\colon P(y)}\;
                \sum_{p\colon y\eq Ax}p_*(f(y,u))\eq{Q(x)}v.
    \end{align*}
    Applying $\Sigma$-commutativity and -associativity (Propositions~\ref{prop:sigma-commutativity} and~\ref{prop:sigma-associativity}), we obtain
    \[
        \fun{fib}_{\fun{total}(f)}(\dep{x, v})
        \simeq
        \sum_{\dep{y, p}\colon\sum_{y\colon A}y\eq Ax}\;
        \sum_{u\colon P(y)}p_*(f(y, u))\eq{Q(x)}v.
    \]
    By Lemma~\ref{lem:Sigma-identity-is-contractible}, the based path space $\sum_{y\colon A}y\eq Ax$ is contractible with center of contraction $\dep{x, \fun{refl}_x}$. Consequently, the entire $\Sigma$-type is equivalent to evaluating the inner sum at this center:
    \begin{align*}
        \fun{fib}_{\fun{total}(f)}(\dep{x,v})
            &\simeq\sum_{u\colon P(x)}
                (\fun{refl}_x)_*(f(x,u))\eq{Q(x)}v\\
            &\equiv\sum_{u\colon P(x)}f(x,u)\eq{Q(x)}v\\
            &\equiv\fun{fib}_{f(x)}(v).
    \end{align*}
    %
    
\end{proof}

\begin{cor}
    Let\/ $f$ be a fiberwise transformation. Then\/ $f$ is a fiberwise equivalence if, and only if, $\fun{total}(f)$ is an equivalence.
\end{cor}

\begin{proof}
    This is a direct consequence of the preceding theorem and Theorem~\ref{thm:isContr->ishae}.
    %
    
\end{proof}

\section{The Object Classifier}

In set theory, $\set{0,1}$ is the \textsl{subobject classifier}, representing the set of truth values. Given a base set $A$ and a subset $S \subseteq A$, this subset is perfectly described by its characteristic function $\chi_S \colon A \to \set{0,1}$. The function $\chi_S$ acts as a classifier: it points to $1$ if an element belongs to $S$, and $0$ otherwise. The subset $S$ is mathematically recovered as the preimage of $1$ under the function~$\chi_S$.

In Homotopy Type Theory, the focus expands from classifying subsets to classifying arbitrary type families. Instead of a subset inclusion $S \hookrightarrow A$, we work with a total space $\Sigma_AB$ equipped with the projection map $\pi_1\colon\Sigma_AB\to A$.

Here, the universe $\univ$ generalizes $\set{0,1}$, becoming the \textsl{object classifier}. The type family $B \colon A \to \univ$ acts precisely like the characteristic function $\chi_S$. Rather than returning a simple truth value, evaluating $B$ at a point $a \colon A$ returns the entire fiber over that point. Just as $\chi_S$ completely determines the subset~$S$, the classifying function $B$ completely determines the structure of the fibration defined by the projection map $\pi_1$.

\begin{prop}\label{prop:pi_1-fibers}
    Let\/ $A\colon\univ$ be a type and\/ $B\colon A\to\univ$ a type family over $A$. Then, the fiber of\/ $\pi_1\colon\Sigma_AB\to A$ over\/ $a\colon A$ is equivalent to\/ $B(a)$:
    \[
        \fun{fib}_{\pi_1}(a) \simeq B(a).
    \]
\end{prop}

\begin{proof}
    \begin{align*}
        \fun{fib}_{\pi_1}(a)
            &\equiv\sum_{u\colon\Sigma_AB}\pi_1(u)\eq Aa\\
            &\simeq\sum_{x\colon A}\;\sum_{u\colon B(x)}
                x\eq Aa
                &&\text{; Prop.~\ref{prop:sigma-associativity}}\\
            &\simeq\sum_{x\colon A}\;\sum_{p\colon x\eq Aa}
                B(x)
                &&\text{; Prop.~\ref{prop:sigma-commutativity}}\\
            &\simeq\sum_{\zeta\colon\sum_{x\colon A}
                x\eq Aa}\!\!\!B(\pi_1(\zeta))
                &&\text{; Prop.~\ref{prop:sigma-associativity}}\\
            &\simeq B(\pi_1(\dep{a,\fun{refl}_a}))
                &&\text{; Lem.~\ref{lem:Sigma-identity-is-contractible} \& Thm.~\ref{thm:contractible-types}~d)}\\
            &\equiv B(a).
    \end{align*}\qedhere
\end{proof}


\begin{lem}\label{lem:fib-reconstruction}
    For any function\/ $f\colon A\to B$, we have
    \[
        A \simeq \sum_{b\colon B}\fun{fib}_f(b).
    \]
\end{lem}

\begin{proof}
    The map that induces the equivalence is defined by
    \[
        \gamma\colon A\to\sum_{b\colon B}\fun{fib}_f(b),
        \quad a\mapsto\dep{f(a),\dep{a,\fun{refl}_{f(a)}}}.
    \]
    Its quasi-inverse is given by
    \[
        \pi\colon
            \Bigl(
                \sum_{b\colon B}\fun{fib}_f(b)
            \Bigr)
            \to A,
            \quad\dep{b,\dep{x, p}}\mapsto x.
    \]
    Indeed, the compositions compute to
    \begin{align*}
        a\xmapsto{\gamma}
            \dep{
                f(a),
                \dep{a,\fun{refl}_{f(a)}}
            }
            \xmapsto{\pi}a
    \intertext{and}
        \dep{b,\dep{x,p}}
            \xmapsto{\pi}x
            \xmapsto{\gamma}
                \dep{f(x),\dep{x,\fun{refl}_{f(x)}}}.
    \end{align*}
    Since $p\colon f(x)\eq{B}b$, showing that 
    \[
        \dep{f(x),\dep{x,\fun{refl}_{f(x)}}}
            \eq{\Sigma_B\fun{fib}_f}\dep{b,\dep{x,p}}
    \]
    amounts to proving that
    \[
        \fun{transport}^{\fun{fib}_f}
            (
                p,\dep{x,\fun{refl}_{f(x)}}
            )
            \eq{\fun{fib}_f(b)}\dep{x,p}.
    \]
    By applying based path induction on $p$, the goal reduces to the case $b\equiv f(x)$ and $p\equiv\fun{refl}_{f(x)}$, which yields the definitional equality
    \[
        \fun{transport}^{\fun{fib}_f}
            (
                \fun{refl}_{f(x)},\dep{x,\fun{refl}_{f(x)}}
            )
            \equiv\dep{x,\fun{refl}_{f(x)}}.
    \]
\end{proof}

\begin{thm}\fe\label{thm:HoTT-Grothendieck-construction}
    For any type\/ $B$, there is an equivalence
    \[
        \chi\colon
            \Bigl(
                \sum_{A\colon\univ}A\to B
            \Bigr)
            \simeq(B\to\univ).
    \]
\end{thm}

\begin{proof}
    We define the forward function by
    \begin{align*}
        \chi\colon
            \Bigl(
                \sum_{A\colon\univ}A\to B
            \Bigr)
            &\to(B\to\univ)\\
        \dep{A,f}&\mapsto\fun{fib}_f,
    \intertext{where $\fun{fib}_f\defeq\lambda b.\,\fun{fib}_f(b)$. The backward function is given by}
        \chi^{-1}\colon(B\to\univ)&\to
            \Bigl(
                \sum_{A\colon\univ}A\to B
            \Bigr)\\
        g&\mapsto\Bigdep{\sum_{b\colon B}g(b),\pi_1}.
    \end{align*}
    To verify that these maps are mutually inverse, we compute both compositions. Given $g\colon B\to\univ$, we have
    \begin{align*}
        (\chi\circ\chi^{-1})(g)
            &\equiv\chi(\chi^{-1}(g))\\
            &\equiv\chi
                \Bigl(
                    \Bigdep{\sum_{b\colon B}g(b),\pi_1}
                \Bigr)\\
            &\equiv\fun{fib}_{\pi_1}.
    \end{align*}
    By Proposition~\ref{prop:pi_1-fibers}, for each $b\colon B$ there is an equivalence $e_b\colon\fun{fib}_{\pi_1}(b)\simeq g(b)$. Therefore,
    \[
        \fun{ua}(e_b)\colon\fun{fib}_{\pi_1}(b)\eq{\univ}g(b).
    \]
    By function extensionality, this family of paths induces an equality of functions
    \[
        \fun{funext}(\lambda b.\,\fun{ua}(e_b))\colon
            \fun{fib}_{\pi_1}\eq{B\to\univ}g,
    \]
    which yields the required path $(\chi\circ\chi^{-1})(g)\eq{B\to\univ}g$.

    On the other hand, given $A\colon\univ$ and $f\colon A\to B$, we have
    \begin{align*}
        \chi^{-1}\circ\chi(\dep{A,f})
            &\equiv\chi^{-1}(\fun{fib}_f)\\
            &\equiv
                \Bigdep{
                    \sum_{b\colon B}\fun{fib}_f(b),\pi_1
                }
    \end{align*}
    To complete the proof, we must show that
    \[
        \Bigdep{
            \sum_{b\colon B}\fun{fib}_f(b),\pi_1
        }
        \eq{\sum_{A\colon\univ}(A\to B)}\dep{A,f}.
    \]
    For the first component, we take $p\defeq\fun{ua}(e^{-1})$, where $e\colon A\simeq\sum_{b\colon B}\fun{fib}_f(b)$ is the equivalence established in Lemma~\ref{lem:fib-reconstruction}. For the second component, we must show that
    \[
        \fun{transport}^{\lambda X.\,(X\to B)}(p,\pi_1)\eq{A\to B}f.
    \]
    As in the proof of Lemma~\ref{lem:fib-reconstruction}, let $\gamma$ and $\pi$ be the quasi-inverses inducing the equivalence $e$. Proposition~\ref{prop:transport-ua-precomp} implies that
    \begin{align*}
        \fun{transport}^{\lambda X.\,(X\to B)}(p,\pi_1)
            &\sim\pi_1\circ\gamma.
    \intertext{Evaluating the \rhs at an arbitrary $a\colon A$, we get}
        \pi_1(\gamma(a))
            &\equiv\pi_1\bigl(\dep{f(a),\dep{a,\fun{refl}_{f(a)}}}\bigr)\\
            &\equiv f(a).
    \end{align*}
    Now function extensionality gives the desired equality.
\end{proof}

\begin{rem}
    Given a type $B\colon\univ$, the total space
    \[
        E\defeq\sum_{A\colon\univ}A\to B
    \]
    is the type-theoretic equivalent of the slice category $\cat{Set}_B$.
    
    To understand this correspondence at the level of morphisms, let us consider how mappings between objects are defined in both contexts. As established in \S\,\ref{slice-category}, a morphism in $\cat{Set}_B$ between two objects $f\colon A\to B$ and $f'\colon A'\to B$ is a function $h\colon A\to A'$ such that the corresponding triangle commutes, i.e., $f'\circ h=f$. In Homotopy Type Theory, the type of maps over $B$ from $\dep{A,f}$ to $\dep{A',f'}$ is given precisely by
    \[
        \sum_{h\colon A\to A'}f'\circ h\sim f.
    \]
    \needspace{3\baselineskip}
    Furthermore, the characterization of equality for $\Sigma$-types dictates that a path between $\dep{A,f}$ and $\dep{A',f'}$ consists of:
    \begin{itemize}
        \item a path $A\eq{\univ}A'$, which the univalence axiom reduces to an equivalence $h\colon A\simeq A'$, and
        
        \item a homotopy $\fun{transport}^{\lambda X.\,(X\to B)}(\fun{ua}(h),f)\sim f'$, which Proposition~\ref{prop:transport-ua-precomp} renders equivalent to a homotopy $f\circ h^{-1}\sim f'$ or $f\sim f'\circ h$.
    \end{itemize}
    Thus, the type-theoretic structure naturally captures the objects, the general morphisms (via the dependent pair type), and the isomorphisms (via the identity type) of the categorical slice construction.

    As discussed in \S\,\ref{types-as-categories}~\nameref{types-as-categories}, we can view types as categories where the terms are the objects and the identity paths are the morphisms. Functors in this context are functions between types, which act on paths via the $\fun{ap}$ operator.

    However, when we let the universe $\univ$ play the role of $\cat{Set}$, our correspondence requires a categorical shift in perspective. In $\cat{Set}$, the morphisms are arbitrary functions, not just bijections. Therefore, when viewing $\univ$ as a category, we treat types as objects, functions as morphisms, and equivalences as isomorphisms. Consequently, a type family $P\colon B\to\univ$ acts as a functor by mapping:
    \begin{itemize}
        \item objects: a term $b\colon B$ is mapped to the type $P(b)\colon\univ$.

        \item isomorphisms: a path $p\colon b\eq{B}b'$ (which is an isomorphism in the groupoid $B$) is mapped to an equivalence in $\univ$. Specifically, $P$ transforms $p$ into the equivalence $\fun{idtoeqv}(\fun{ap}_P(p))\colon P(b)\simeq P(b')$. By Proposition~\ref{prop:id-transport-and-ap}, the underlying function of this equivalence is exactly the transport map $\fun{transport}^P(p,\dummy)\colon P(b)\to P(b')$.
    \end{itemize}

    In category theory, a natural transformation between two functors consists of two separate pieces of data:
    \begin{enumerate}[a)]
        \item a family of morphisms representing the object-level components, and
        
        \item a proof that these components make the associated naturality squares commute for every morphism in the base category.
    \end{enumerate}
    Given families $P, Q\colon B\to\univ$, a dependent function $\eta\colon\prod_{b\colon B}P(b)\to Q(b)$ provides exactly the object-level data required by part (a). For every $b\colon B$, it gives a function $\eta(b)\colon P(b)\to Q(b)$.
    
    The profound difference in type theory is that part (b) is an intrinsic consequence of path induction. To verify this naturality, we must show that for any path $p\colon b\eq{}b'$ and any element $x\colon P(b)$, the evaluation of $\eta(b')$ after transporting $x$ equals the transport of the evaluation of $\eta(b)$ at $x$. This is expressed by the propositional equality
    \[
        \eta(b')(\fun{transport}^P(p,x)) \eq{} \fun{transport}^Q(p,\eta(b)(x)).
    \]
    We can construct a path witnessing this equality by induction on $p$. Assuming $b'\equiv b$ and $p\equiv\fun{refl}_b$, the transport operations reduce definitionally to the identity function. The equation then simplifies to
    \[
        \eta(b)(x) \eq{} \eta(b)(x),
    \]
    which is trivially inhabited by $\fun{refl}_{\eta(b)(x)}$. By path induction, this commutativity holds for all paths $p$. 
    
    Therefore, simply providing the dependent function $\eta$ is entirely sufficient. The rules of identity types guarantee that any such function is natural with respect to paths.

    Under these considerations it is now clear that Theorem~\ref{thm:HoTT-Grothendieck-construction} is the type-theoretic version of the Grothendieck Construction~Theorem~\ref{thm:grothendieck-construction}.
\end{rem}

\begin{ntn}
    Let\/ $U,V,Q\colon\univ$ be types, and consider functions\/ $g_U\colon U\to Q$ and\/ $g_V\colon V\to Q$. The canonical pullback type\/ $U\times_Q V$ is defined as
    \[
        U\times_Q V
            \defeq
                \sum_{u\colon U}\;
                \sum_{v\colon V}g_U(u)\eq{Q}g_V(v),
    \]
    where\/ $\pi_U\colon(U\times_Q V)\to U$ is the first projection\/ $\pi_1$, and\/ $\pi_V\colon(U\times_Q V)\to V$ is the composition\/ $\pi_1\circ\pi_2$.
\end{ntn}

\begin{lem}\fe\label{lem:pullback}
    With the preceding notation, the canonical construction\/ $U\times_Q V$, equipped with the projections\/ $\pi_U$ and\/ $\pi_V$, is the pullback of\/ $g_U$ and\/ $g_V$.
\end{lem}

\begin{proof}
    By Exercise~\ref{exr:pullback}, we have to show that, for any $X\colon\univ$, there is an equivalence
    \[
        \bigl(X\to(U\times_Q V)\bigr)
        \simeq
            \biggl(
                \sum_{(u,v)\colon X\to(U\times V)}
                g_U\circ u\eq{X\to Q}g_V\circ v
            \biggr).
    \]
    By \nref{lpar:axiom-of-choice} applied twice,
    \begin{align*}
        \bigl(X\to(U\times_Q V)\bigr)
            &\equiv\prod_{x\colon X}\;
                \sum_{u\colon U}\;
                \sum_{v\colon V}g_U(u)\eq{Q}g_V(v)\\
            &\simeq\sum_{u\colon X\to U}\;
                \prod_{x\colon X}\;
                \sum_{v\colon V}g_U(u(x))\eq{Q}g_V(v)\\
            &\simeq\sum_{u\colon X\to U}\;
                \sum_{v\colon X\to V}\;
                \prod_{x\colon X}g_U(u(x))\eq{Q}g_V(v(x)).
    \end{align*}
    Then, by function extensionality,
    \[
        \rhs
            \simeq\sum_{u\colon X\to U}\;
                \sum_{v\colon X\to V}
                    g_U\circ u\eq{X\to Q}g_V\circ v.
    \]
    Grouping $u$ and $v$ into a single map of type $X\to(U\times V)$, we obtain
    \begin{align*}
        \rhs
            &\simeq\sum_{(u,v)\colon X\to(U\times V)}
                g_U\circ u\eq{X\to Q}g_V\circ v,
    \end{align*}
    as required.
\end{proof}

\begin{thm}\fe
    Let\/ $f\colon A\to B$ be a function. Then the diagram
    \[
        \begin{tikzcd}
            A
                    \arrow[r,"\vartheta_f"]
                    \arrow[d,"f"']
                &\puniv
                    \arrow[d,"\pi_1"]\\
            B
                    \arrow[r,"\fun{fib}_f"']
                &\univ
        \end{tikzcd}
    \]
    is a pullback square, where
    \[
        \vartheta_f\defeq\lambda a.\,
            \dep{
                \fun{fib}_f(f(a)),
                \dep{
                    a,
                    \fun{refl}_{f(a)}
                }
            }.
    \]
\end{thm}

\begin{proof}
    First observe that Lemma~\ref{lem:Sigma-identity-is-contractible} implies that, for any\/ $b\colon B$, the type
    \[
        C_b\defeq\sum_{X\colon\univ}\fun{fib}_f(b)\eq{\univ}X
    \]
    is contractible with center\/ $c\defeq\dep{\fun{fib}_f(b),\fun{refl}_{\fun{fib}_f(b)}}$. By associativity of\/ $\Sigma$-types, we have
    \[
        \sum_{X\colon\univ}\;
        \sum_{p\colon\fun{fib}_f(b)=X}X
        \simeq\sum_{\omega\colon C_b}\pi_1(\omega).
    \]
    Therefore, by part b) of Theorem~\ref{thm:contractible-types}, evaluating the family\/ $\pi_1$ at the center of contraction yields the equivalence
    \begin{align*}
        \sum_{\omega\colon C_b}\pi_1(\omega)
            \simeq\pi_1(c)\equiv\fun{fib}_f(b).
    \end{align*}
    It follows that
    \[\tag{$*$}
        \fun{fib}_f(b)\simeq
            \sum_{X\colon\univ}\;
            \sum_{p\colon\fun{fib}_f(b)=X}X.
    \]
    On the other hand, since the type family
    \begin{align*}
        P\colon(\fun{fib}_f(b)\eq{\univ}X)&\to\univ\\
        p&\mapsto X   
    \end{align*}
    is constant, we can swap its components and obtain the equivalence
    \begin{align*}\tag{$**$}
        \Bigl(\sum_{p\colon\fun{fib}_f(b)=X}X\Bigr)
        &\xto{\simeq}\sum_{x\colon X}\fun{fib}_f(b)\eq{\univ}X\\
        \dep{p,x}&\mapsto\dep{x,p}.
    \end{align*}
    Then,
    \begin{align*}
        A
            &\simeq\sum_{b\colon B}\fun{fib}_f(b)
                &&\text{; Lem.~\ref{lem:fib-reconstruction}}\\
        &\simeq
            \sum_{b\colon B}\;
            \sum_{X\colon\univ}\;
            \sum_{p\colon\fun{fib}_f(b)\eq{\univ}X}X
            &&\text{; ($*$)}\\
        &\simeq
            \sum_{b\colon B}\;
            \sum_{X\colon\univ}\;
            \sum_{x\colon X}\fun{fib}_f(b)\eq{\univ}X
            &&\text{; ($**$)}\\
        &\simeq
            \sum_{b\colon B}\;
            \sum_{Y\colon\puniv}\fun{fib}_f(b)\eq{\univ}\pi_1(Y)
            &&\text{; Def.~of\/ }\puniv\\
        &\equiv B\times_{\univ}\puniv,
            &&\text{; Lem.~\ref{lem:pullback}}
    \end{align*}
    Let\/ $e\colon A\xto{\simeq} B\times_\univ\puniv$ be the function inducing the equivalence above. Following the composition step by step, we obtain
    \begin{align*}
        a&\mapsto\dep{
            \explain[B]{f(a)},
            \explain[\fun{fib}_f(f(a))]
            {\dep{a,\fun{refl}_{f(a)}}}
            }\\
        &\mapsto
            \dep{
                \explain[B]{f(a)},
                \dep{
                    \explain[\univ]{\fun{fib}_f(f(a))},
                    \dep{
                        \explain[\fun{fib}_f(f(a))\eq{\univ}X]{
                            \fun{refl}_{\fun{fib}_f(f(a))}
                        },
                        \explain[X]{
                            \dep{
                                a,
                                \fun{refl}_{f(a)}
                            }
                        }
                    }
                }
            }\\
        &\mapsto
            \dep{
                \explain[B]{f(a)},
                \dep{
                    \explain[\univ]{\fun{fib}_f(f(a))},
                    \dep{
                        \explain[X]{\dep{a,\fun{refl}_{f(a)}}}, 
                        \explain[\fun{fib}_f(f(a))\eq{\univ}X]{
                            \fun{refl}_{\fun{fib}_f(f(a))}
                        }
                    }
                }
            }\\
        &\mapsto
            \dep{
                \explain[B]{f(a)},
                \dep{
                    \explain[\puniv]{
                        \dep{
                            \fun{fib}_f(f(a)),
                            \dep{a,\fun{refl}_{f(a)}}
                        }
                    },
                    \explain[\fun{fib}_f(f(a))\eq{\univ}\pi_1(Y)]{
                        \fun{refl}_{\fun{fib}_f(f(a))}
                    }
                }
            }.
    \end{align*}
    Therefore, we get homotopies\/ $f\sim\pi_1\circ e$ and\/ $\vartheta_f\sim\pi_1\circ\pi_2\circ e$.
\end{proof}

\begin{rem}
    The pullback square of the universal fibration $\pi_1\colon\puniv\to\univ$ establishes a direct link between the geometric universal property of limits and the logical equivalence of the Grothendieck construction. Suppose two functions $f,g\colon A\to B$ have propositionally equal fibers, meaning $\fun{fib}_f\eq{B\to\univ}\fun{fib}_g$. They therefore share the same classifying map $\chi\colon B\to\univ$ defined by their common fibers.

    Because both $f$ and $g$ can be recovered as pullbacks of the universal bundle $\pi_1\colon\puniv\to\univ$ along $\chi$, they both exhibit the domain $A$ as the limit of the exact same diagram. By the universal property of pullbacks, limits are unique up to equivalence. This guarantees the existence of a canonical automorphism $\alpha\colon A\simeq A$ over $B$, yielding the homotopy $f\sim g\circ\alpha$.

    This geometric uniqueness perfectly mirrors the algebraic equality within the Grothendieck construction. Under the equivalence
    \[
        \Bigl(\sum_{X\colon\univ}X\to B\Bigr)\simeq(B\to\univ),
    \]
    mapping both functions to the same family implies the equality $\dep{A,f}\eq{}\dep{A,g}$ as points in the $\Sigma$-type. Equality in this total space consists of an equivalence $\alpha\colon A\simeq A$ in the first coordinate and a path $f\eq{A\to B}g\circ\alpha^{-1}$ in the second.
    
    Thus, the universal property of the pullback acts as the local geometric mechanism that computes the global structural equivalence between ordinary functions mapping into a base type $B$ and dependent type families indexed by~$B$.
\end{rem}

\section{Univalence Implies Function Extensionality}\label{UA-implies-FE}

In this section we prove that $\UA\Rightarrow\WE\Rightarrow\FE$.

\begin{rem}
    The following lemma is proved without using the function extensionality axiom.
\end{rem}

\begin{lem}\label{lem:base-equivalence-implies-bundle-equivalence}
    Assuming the univalence axiom, for any types\/ $A,B,X\colon\univ$ and any equivalence\/ $e\colon A\simeq B$, post-composition with\/ $e$ induces an equivalence
    \[
        (X\to A)\simeq(X\to B).
    \]
\end{lem}

\begin{proof}
    By the univalence axiom, the map $\fun{idtoeqv}\colon(A\eq{\univ}B)\to(A\simeq B)$ induces an equivalence with quasi-inverse $\fun{ua}$. Thus, to prove the statement for an arbitrary equivalence $e\colon A\simeq B$, it suffices to assume that $B\equiv A$ and that $e\equiv\fun{idtoeqv}(\fun{refl}_A)$ is the identity equivalence on~$A$.

    Under this assumption, the underlying function of $e$ is $\id_A\colon A\to A$. We must provide an equivalence $(X\to A)\simeq(X\to A)$ whose underlying map is post-composition with $\id_A$.

    We choose the identity equivalence on $X\to A$. Its underlying map is the identity function $\id_{X\to A}$, and for any $h\colon X\to A$, we have the judgmental equalities
    \[
        \id_{X\to A}(h)\equiv h\equiv\id_A\circ h.
    \]
    Thus, the underlying map of the identity equivalence is judgmentally equal to post-composition with the underlying map of $e$, completing the proof.
    
\end{proof}

\begin{prop}\label{prop:projection-equivalence}
    Let\/ $P\colon A\to\univ$ be a family of contractible types. Then the projection\/ $\pi_1\colon\Sigma_AP\to A$ is an equivalence. If\/ $\univ$ is univalent, then post-composition with\/ $\pi_1$ gives an equivalence
    \[
        (A\to\Sigma_AP)\simeq(A\to A).
    \]
\end{prop}

\begin{proof}
    By Proposition~\ref{prop:pi_1-fibers}, given $x\colon A$, we have an equivalence
    \[
        \fun{fib}_{\pi_1}(x)\simeq P(x).
    \]
    In particular, $\fun{fib}_{\pi_1}(x)$ is contractible itself by Lemma~\ref{lem:contractible-equivalences}. Thus, Theorem~\ref{thm:isContr->ishae} implies that $\pi_1$ induces an equivalence. The second conclusion is a direct consequence of Lemma~\ref{lem:base-equivalence-implies-bundle-equivalence}.
    %
    
\end{proof}

\begin{thm}\label{thm:univalence-implies-weak-funext}
    The univalence axiom implies the weak function extensionality principle.
\end{thm}

\begin{proof}
    Suppose that $\univ$ is univalent, and let $P\colon A\to\univ$ be a family of contractible types. We must show that the dependent product $\prod_{x\colon A}P(x)$ is contractible.
    
    Let
    \[
        \alpha\colon(A\to\Sigma_AP)\simeq(A\to A)
    \]
    be the equivalence of Proposition~\ref{prop:projection-equivalence}. We will show that the product is a retract of $\fun{fib}_\alpha(\id_A)$.

    Define the maps
    \begin{align*}
        \sigma&\colon\Bigl(\prod_{x\colon A}P(x)\Bigr)
        \to
        \fun{fib}_\alpha(\id_A)\\
        \sigma(f)&\defeq\dep{
            \lambda x.\,\dep{x,f(x)},
            \fun{refl}_{\id_A}
        },
    \intertext{and}
        \rho&\colon\fun{fib}_\alpha(\id_A)
        \to
        \prod_{x\colon A}P(x)\\
        \rho(\dep{g,p})&\defeq
            \lambda x.\,\fun{happly}(p, x)_*(\pi_2(g(x))).
    \end{align*}
    Evaluating their composition yields
    \begin{align*}
        \rho(\sigma(f))
            &\equiv\rho(\dep{
                \lambda x.\,\dep{x,f(x)},
                \fun{refl}_{\id_A}
                })\\
            &\equiv\lambda x.\,
                \fun{happly}(\fun{refl}_{\id_A},x)_*(f(x))\\
            &\equiv\lambda x.\,
                (\fun{refl}_{\id_A(x)})_*(f(x))
                &&\text{; Eq.~\eqref{eq:happly-refl}}\\
            &\equiv\lambda x.\,f(x)\\
            &\equiv f,
    \end{align*}
    as desired. Since the fiber is contractible by Remark~\ref{rem:equivalence-fib-is-contractible}, the result follows from Proposition~\ref{prop:contractible-retracts}.
    %
    
\end{proof}

\begin{thm}\label{thm:weak-funext-implies-funext}
    The weak function extensionality principle implies the function extensionality axiom.
\end{thm}

\begin{proof}
    Let
    \[
        \Gamma\defeq\prod_{x\colon A}P(x).
    \]
    Fix $f\colon\Gamma$ and define
    \begin{align*}
        \gamma\colon\Bigl(
            \sum_{g\colon\Gamma}f\eq{\Gamma}g
        \Bigr)
        &\to
        \sum_{g\colon\Gamma}f\sim g\\
        \dep{g,p}
        &\mapsto\dep{g,\fun{happly}(p)}.
    \end{align*}
    By Lemma~\ref{lem:Sigma-identity-is-contractible}, the domain type of this map is contractible. For the codomain type we can apply \nref{lpar:axiom-of-choice} and obtain an equivalence
    \begin{align*}
        \sum_{g\colon\Gamma}f\sim g
            &\equiv\sum_{g\colon\Gamma}\;
                \prod_{x\colon A}f(x)\eq{P(x)}g(x)\\
            &\simeq\prod_{x\colon A}\;
                \sum_{u\colon P(x)}f(x)\eq{P(x)}u.
    \end{align*}
    Note that the inner $\Sigma$-type is contractible by the same lemma as before. From weak function extensionality, it follows that the outer $\Pi$-type is contractible as well. Hence the codomain type of $\gamma$ is contractible. Using Lemmas~\ref{lem:contractible-equivalences} and~\ref{lem:mere-proposition-equivalence}, we see that $\gamma$ induces an equivalence.

    Consider the fiberwise transformation
    \[
        h\colon\prod_{g\colon\Gamma}\bigl((f\eq{\Gamma}g)\to(f\sim g)\bigr)
    \]
    defined by $h(g)\defeq\fun{happly}(f,g)$. According to Definition~\ref{defn:total-map},
    \[
        \gamma\equiv\fun{total}(h).
    \]
    By Theorem~\ref{thm:total-fib-equivalence}, for $\eta\colon f\sim g$, we have
    \[
        \fun{fib}_\gamma(\dep{g,\eta})
        \simeq
        \fun{fib}_{h(g)}(\eta).
    \]
    Since the fibers of $\gamma$ are contractible by Theorem~\ref{thm:isContr-ishae-equivalence}, we deduce that the fibers of $h(g)$ are contractible. Hence, by the same theorem, $h(g)$ induces an equivalence.
    %
    
\end{proof}

\begin{exr}\fe
    Consider the type of ``two-sided adjoint equivalence data'' for\/ $f\colon A\to B$,
    \[
        \sum_{g\colon B\to A}\;
        \sum_{\eta\colon f\circ g\sim\id_B}\;
        \sum_{\vartheta\colon g\circ f\sim\id_A}
            (f\lwct\vartheta\sim\eta\rwct f)
            \times
            (g\lwct\eta\sim\vartheta\rwct g).
    \]
    By\/ {\upshape Proposition~\ref{prop:coherence-swap}}, we know that if\/ $f$ is an equivalence, then this type is inhabited. Give a characterization of this type analogous to\/ {\upshape Proposition~\ref{prop:qinv-id-homotopy-equivalence}}.

    Can you give an example showing that this type is not generally a mere proposition? (This will be easier after\/ {\upshape Chapter~6}.)
\end{exr}

\begin{solution}
    Recall that
    \begin{align*}
        \fun{rcoh}_f\colon\fun{rinv}(f)&\to\univ\\
        \dep{g,\eta}&\mapsto
            \sum_{\vartheta\colon g\circ f\sim\id_A}
                f\lwct\vartheta\sim\eta\rwct f.
    \end{align*}
    Let $G$ denote the target type. By $\Sigma$-associativity, we can expand $G$ as
    \begin{align*}
        G
            &\equiv
                \sum_{\dep{g,\eta}\colon\fun{rinv}(f)}\;
                \sum_{\vartheta\colon g\circ f\sim\id_A}
                    (f\lwct\vartheta\sim\eta\rwct f)
                    \times
                    (g\lwct\eta\sim\vartheta\rwct g).
    \end{align*}
    Consider the type of right-half-adjoint equivalence data,
    \begin{align*}
        G'\defeq
            \sum_{\dep{g,\eta}\colon\fun{rinv}(f)}\;
            \fun{rcoh}_f(\dep{g,\eta}).
    \end{align*}
    There is an equivalence
    \begin{align*}
        G&\simeq
        \sum_{
            \dep{
                \dep{g,\eta},
                \dep{\vartheta,\tau}
            }\colon G'
        }
        g\lwct\eta\sim\vartheta\rwct g,
    \end{align*}
    induced by the canonical mapping
    \[
        \bigdep{
            \dep{g,\eta},
            \dep{\vartheta,(\tau,\sigma)}
        }
        \mapsto
        \bigdep{
            \dep{
                \dep{g,\eta},
                \dep{\vartheta,\tau}},
            \sigma
        }.
    \]
    As we saw in the proof of Theorem~\ref{thm:ishae-is-mere-proposition}, $G'\simeq\fun{ishae}(f)$. In particular, if $f$ is an equivalence, then $G'$ is contractible. By evaluating the sum at the center of contraction $\dep{\dep{g,\eta}, \dep{\vartheta,\tau}}$ of $G'$, we obtain the equivalence
    \[
        G\simeq (g\lwct\eta\sim\vartheta\rwct g).
    \]
    Thus, $G$ is equivalent to the space of choices for this secondary coherence condition—a ``loose end,'' as described in Remark~\ref{rem:why-half}.

    Regarding the second part of the exercise, a counterexample can be provided by any type whose identity type is not always a set. Since we have not yet seen such a type, we postpone a complete solution to a later chapter.
    %
    
\end{solution}

\begin{exr}\fe
    Show that for any types\/ $A, B\colon\univ$, the following type is equivalent to\/ $A\simeq B$.
    \[
        \sum_{R\colon A\to B\to\univ}
        \Bigl(
            \prod_{a\colon A}
            \fun{isContr}\Bigl(\sum_{b\colon B}R(a,b)\Bigr)
        \Bigr)
        \times
        \Bigl(
            \prod_{b\colon B}
            \fun{isContr}\Bigl(\sum_{a\colon A}R(a,b)\Bigr)
        \Bigr).
    \]
    Can you extract from this a definition of a type satisfying the three desiderata of\/ $\fun{isequiv}(f)$? Given a type family $\fun E\colon(A\to B)\to\univ$, these are:
    \begin{enumerate}[i),font=\upshape]
        \item For each\/ $f\colon A\to B$ there is a function\/ $\fun{qinv}(f)\to\fun{E}(f)$.
        
        \item For each\/ $f\colon A\to B$ there is a function\/ $\fun{E}(f)\to\fun{qinv}(f)$.
        
        \item $\fun{E}(f)$ is a mere proposition.
    \end{enumerate}
\end{exr}

\begin{solution}
    With abbreviated notation we can rewrite the target type as
    \[
        \sum_{R\colon A\to B\to\univ}
            \Bigl(
                \prod_{a\colon A}
                    \fun{isContr}(\Sigma_BR(a,\dummy))
            \Bigr)
            \times
            \Bigl(
                \prod_{b\colon B}
                    \fun{isContr}(\Sigma_AR(\dummy,b))
            \Bigr).
    \]
    Expanding the definition of contractibility, the first part of the Cartesian product is given by
    \begin{align*}
        \prod_{a\colon A}\fun{isContr}(\Sigma_BR(a,\dummy))
            &\equiv\prod_{a\colon A}\;
                \sum_{u\colon\Sigma_BR(a,\dummy)}\;
                \prod_{v\colon\Sigma_BR(a,\dummy)}u\eq{}v\\
            &\simeq\prod_{a\colon A}\;
                \sum_{b\colon B}\;
                \sum_{r\colon R(a,b)}\;
                \prod_{v\colon\Sigma_BR(a,\dummy)}
                    \dep{b,r}\eq{}v
                &&\text{; $\Sigma$-assoc.}.
    \end{align*}
    Applying \nref{lpar:axiom-of-choice}, the \rhs satisfies
    \[\tag{$1$}
        \rhs
            \simeq\sum_{f\colon A\to B}\;
                \prod_{a\colon A}\;
                \sum_{r\colon R(a,f(a))}\;
                \prod_{v\colon\Sigma_BR(a,\dummy)}
                    \dep{f(a),r}\eq{}v,
    \]
    which allows us to extract, via the first projection of the outer $\Sigma$-type, a function $f\colon A\to B$ and a proof that the type
    \[
        \prod_{a\colon A}R(a,f(a))
    \]
    is inhabited.\footnote{Given $a\colon A$, since $P(a)\defeq\Sigma_BR(a,\dummy)$ is contractible we might have defined $f(a)$ as its center of contraction. Similarly for the center $g(b)$ of $Q(b)\defeq\Sigma_AR(\dummy,b)$.}
    
    Repeating this same reasoning for the second part of the product, we extract a function $g\colon B\to A$ and proof that the type 
    \[\tag{$2$}
        \prod_{b\colon B}R(g(b),b)
    \]
    is inhabited.
    
    Fix $b\colon B$. From $(2)$ we obtain a term $s\colon R(g(b),b)$. the evaluation of~($1$) at $a\defeq g(b)$ produces a term $r\colon R(g(b),f(g(b)))$ and a map
    \[
        \rho\colon\prod_{v\colon\Sigma_BR(g(b),\dummy)}
            \dep{f(g(b)),r}\eq{} v.
    \]
    Because $s\colon R(g(b),b)$, we can form the pair $v\defeq\dep{b,s}\colon\Sigma_BR(g(b),\dummy)$. Evaluating $\rho$ at this $v$, we arrive at the equality
    \[
        \dep{f(g(b)),r}\eq{}\dep{b,s}.
    \]
    In particular, $f(g(b))\eq Bb$. Since $b\colon B$ was arbitrarily chosen, this constructs the right homotopy $f\circ g\sim\id_B$. The left homotopy $g\circ f\sim\id_A$ follows symmetrically.

    Conversely, if $g\colon B\to A$ induces an equivalence, we have a relation
    \[
        R\colon A\to B\to\univ,\quad
        R(a,b)\defeq g(b)\eq{A}a.
    \]
    Since the fibers of $g$ are contractible, we obtain a witness for the first part of the product:
    \[
        \prod_{a\colon A}\fun{isContr}
            \Bigl(
                \sum_{b\colon B}g(b)\eq{A}a
            \Bigr).
    \]
    For the second part, observe that for any fixed $b\colon B$, the sum
    \[
        \sum_{a\colon A}g(b)\eq{A}a
    \]
    is also contractible by Lemma~\ref{lem:Sigma-identity-is-contractible} applied to $g(b)\colon A$. Therefore, the second part of the Cartesian product is inhabited for all $b\colon B$. This shows that the entire target type has a witness.

    \bigskip

    For the final question, fix\/ $f\colon A\to B$, and consider the relation
    \[
        R(a,b)\defeq f(a)\eq Bb.
    \]
    By definition, the target type computes to
    \[
        \fun{E}(f)
            \defeq
                \prod_{a\colon A}\;
                \fun{isContr}
                    \Bigl(
                        \sum_{b\colon B}f(a)\eq Bb
                    \Bigr)
                \times
                \prod_{b\colon B}
                    \fun{isContr}
                        \Bigl(
                            \sum_{a\colon A}f(a)\eq Bb
                        \Bigr).
    \]
    By Lemma~\ref{lem:Sigma-identity-is-contractible}, the first part of this Cartesian product is inhabited. Thus, by Corollary~\ref{cor:A-contractible-implies-isContr(A)-contractible} and part a) of Theorem~\ref{thm:contractible-types}, it is contractible. The second part of the Cartesian product is definitionally equal to\/ $\fun{isContrMap}(f)$. Therefore, we obtain an equivalence
    \[
        \fun{E}(f)\simeq\fun{isContrMap}(f),
    \]
    which is equivalent to\/ $\fun{ishae}(f)$ by Theorem~\ref{thm:isContr->ishae} and its preceding comment.
    %
    
\end{solution}

\begin{exr}{\upshape[The unstable octahedral axiom]}
    Let\/ $f\colon A\to B$ and\/ $g\colon B\to C$ be functions, and fix an element\/ $b\colon B$.
    \begin{enumerate}[a),font=\upshape]
        \item Construct a natural map\/ $\fun{fib}_{g\circ f}(g(b))\to\fun{fib}_g(g(b))$ and prove that its fiber over the point\/ $\dep{b,\fun{refl}_{g(b)}}$ is equivalent to\/ $\fun{fib}_f(b)$.
        \item Show that for any\/ $c\colon C$, there is an equivalence
        \[
            \fun{fib}_{g\circ f}(c)\simeq\sum_{w\colon\fun{fib}_g(c)}\fun{fib}_f(\pi_1(w)).
        \]
    \end{enumerate}
\end{exr}

\begin{solution}
    \begin{enumerate}[a),font=\upshape]
        \item Let $\dep{x,p}\colon\fun{fib}_{g\circ f}(g(b))$. By definition, $p\colon g(f(x))\eq{C}g(b)$. It follows that $\dep{f(x),p}\colon\fun{fib}_g(g(b))$, and we can define
        \begin{align*}
            \phi\colon\fun{fib}_{g\circ f}(g(b))
                &\to\fun{fib}_g(g(b))\\
            \dep{x,p}&\mapsto\dep{f(x),p}.
        \end{align*}
        In particular,
        \begin{align*}
            \fun{fib}_\phi(\dep{b,\fun{refl}_{g(b)}})
            &\equiv
                \sum_{w\colon\fun{fib}_{g\circ f}(g(b))}
                    \dep{f(\pi_1(w)),\pi_2(w)}
                    \eq{\fun{fib}_g(g(b))}
                    \dep{b,\fun{refl}_{g(b)}}\\
            &\simeq
                \sum_{x\colon A}\;
                \sum_{p\colon g(f(x))\eq{C}g(b)}
                    \dep{f(x),p}
                    \eq{\fun{fib}_g(g(b))}
                    \dep{b,\fun{refl}_{g(b)}}\\
            &\simeq
                \sum_{x\colon A}\;
                \sum_{p\colon g(f(x))\eq{C}g(b)}\;
                \sum_{q\colon f(x)\eq{B}b}
                    \!\!\!q_*(p)
                    \eq{g(b)\eq{C}g(b)}
                    \fun{refl}_{g(b)}                    
        \end{align*}
        where
        \begin{align*}
            q_*(p)&\equiv
                \fun{transport}^{y\mapsto g(y)\eq{C}g(b)}
                    (q,p)\\
            &\eq{g(b)\eq{C}g(b)}
                \fun{transport}^{z\mapsto z\eq{C}g(b)}
                    (\fun{ap}_g(q),p)
                &&\text{; Lem.~\ref{lem:post-composition-transport}}\\
            &\eq{g(b)\eq{C}g(b)}
                \fun{ap}_g(q)^{-1}\ct p
                &&\text{; Prop.~\ref{prop:transport-decomposition-for-paths}.}
        \end{align*}
        Hence,
        \begin{align*}
            \fun{fib}_\phi(\dep{b,\fun{refl}_{g(b)}})
            &\simeq
                \sum_{x\colon A}\;
                \sum_{p\colon g(f(x))\eq{C}g(b)}\;
                \sum_{q\colon f(x)\eq{B}b}
                    p
                    \eq{g(f(x))\eq{C}g(b)}
                    \fun{ap}_g(q)\\
            &\simeq
                \sum_{x\colon A}\;
                \sum_{p\colon g(f(x))\eq{C}g(b)}\;
                \sum_{q\colon f(x)\eq{B}b}
                    \fun{ap}_g(q)
                    \eq{g(f(x))\eq{C}g(b)}
                    p\\
            &\simeq
                \sum_{x\colon A}\;
                \sum_{q\colon f(x)\eq{B}b}\;
                \explain[\text{Contractible by Lem.~\ref{lem:Sigma-identity-is-contractible}}]{
                \sum_{p\colon g(f(x))\eq{C}g(b)}
                    \fun{ap}_g(q)
                    \eq{g(f(x))\eq{C}g(b)}
                    p}\\
            &\simeq
                \sum_{x\colon A}\;
                \sum_{q\colon f(x)\eq{B}b}
                    \fun1\\
            &\simeq\sum_{x\colon A}f(x)\eq{B}b\\
            &\equiv\fun{fib}_f(b).
        \end{align*}

        \textbf{Note.} The sequence
        \[
            \fun{fib}_f(b)\to\fun{fib}_{g\circ f}(g(b))\xrightarrow{\phi}\fun{fib}_g(g(b))
        \]
        shows that $\fun{fib}_f(b)$ behaves as the ``kernel'' of $\phi$. It is what in algebraic topology is called a \textsl{fiber sequence}, the geometric analogue of an algebraic exact sequence.


        \item Given $c\colon C$, we have
        \begin{align*}
            \sum_{w\colon\fun{fib}_g(c)}
                \fun{fib}_f(\pi_1(w))
                &\simeq
                    \sum_{y\colon B}\;
                    \sum_{q\colon g(y)\eq{C}c}\;
                    \sum_{x\colon A}f(x)\eq{B}y\\
                &\simeq
                    \sum_{x\colon A}\;
                    \sum_{y\colon B}\;
                    \sum_{q\colon g(y)\eq{C}c}
                    f(x)\eq{B}y\\
                &\simeq
                    \sum_{x\colon A}\;
                    \sum_{y\colon B}\;
                    (f(x)\eq{B}y)\times(g(y)\eq{C}c).
        \end{align*}
        Since
        \[
            \sum_{y\colon B}f(x)\eq{B}y
        \]
        is contractible with center $\dep{f(x),\fun{refl}_{f(x)}}$ by Lemma~\ref{lem:Sigma-identity-is-contractible}, we can apply Lemma~\ref{lem:contractible-sigma-product} and replace the inner $\Sigma$-type with
        \[
            g(f(x))\eq{C}c,
        \]
        which yields $\fun{fib}_{g\circ f}(c)$, as desired.

        \textbf{Note.} The equivalence
        \[
            \fun{fib}_{g\circ f}(c)\simeq\sum_{w\colon\fun{fib}_g(c)}\fun{fib}_f(\pi_1(w))
        \]
        geometrically describes how the fibers of $g\circ f$ can be recovered from the fibers of~$f$ and~$g$.

    \end{enumerate}
\end{solution}

\begin{exr}
    Prove that equivalences satisfy the $2$-out-of-$6$ property: given\/ $f\colon A\to B$ and\/ $g\colon B\to C$ and\/ $h\colon C\to D$, if\/ $g\circ f$ and\/ $h\circ g$ are equivalences, so are\/ $f$, $g$, $h$, and\/ $h\circ g\circ f$. Use this to give a higher-level proof of\/ {\upshape Theorem~\ref{thm:f-equiv-implies-apf-equiv}}.
\end{exr}

\begin{solution}\ 

    \textsc{First part.}
    Suppose that $\dep{u,(\eta,\vartheta)}$ is a quasi-inverse of $g\circ f$ and $\dep{v,(\rho,\theta)}$ is a quasi-inverse of $h\circ g$. Then
    \begin{align*}
        \eta\colon g\circ f\circ u\sim\id_C
        \quad\text{and}\quad
        \theta\colon v\circ h\circ g\sim\id_A,
    \end{align*}
    which show that
    \[
        (\dep{f\circ u,\eta},\dep{v\circ h,\theta})
        \colon\fun{isequiv}(g).
    \]
    Hence, $g$ has a quasi-inverse $g^{-1}$, and so $f\sim g^{-1}\circ(g\circ f)$ and $h\sim(h\circ g)\circ g^{-1}$ have quasi-inverses as well.

    \medskip

    \textsc{Second part.} We give two proofs. The first one following the approach suggested by the exercise. The second, derived from the previous exercise. In both cases we suppose that\/ $\dep{g,(\eta,\vartheta)}$ is a quasi-inverse of\/ $f$.

    \textit{First proof.}
    Given $x,y\colon A$ and $p\colon x\eq{A}y$, the naturality of $\vartheta\colon g\circ f\sim\id_A$ yields
    \[
        \vartheta_x\ct p\eq{}\fun{ap}_{g\circ f}(p)\ct\vartheta_y.
    \]
    By right-multiplying by $\vartheta_y^{-1}$, we obtain
    \begin{align*}
        \fun{ap}_g(\fun{ap}_f(p))
            &\eq{}\fun{ap}_{g\circ f}(p)\\
            &\eq{}\vartheta_x\ct p\ct\vartheta_y^{-1}.
    \end{align*}
    Thus, the function $\fun{ap}_g\circ\fun{ap}_f$ is given by pre- and post-composition with fixed paths, which is an equivalence with quasi-inverse
    \[
        q\mapsto\vartheta_x^{-1}\ct q\ct\vartheta_y.
    \]
    By a symmetric argument using the right homotopy $\eta\colon f\circ g\sim\id_B$, the composition $\fun{ap}_f\circ\fun{ap}_g$ is also an equivalence.

    The conclusion follows by applying the $2$-out-of-$6$ property to the following sequence of maps:
    \[
    \begin{tikzcd}
        f(x)\eq{B}f(y)
                \arrow[r,"\fun{ap}_g"]
            &g(f(x))\eq{A}g(f(y))
                \arrow[d,"\fun{ap}_f"]\\
        x\eq{A}y
                \arrow[u,"\fun{ap}_f"]
            &f(g(f(x))\eq{B}f(g(f(y)))).
    \end{tikzcd}
    \]

    \textit{Second proof.}
    As established in part~a) of the previous exercise, we have
    \begin{align*}
        \fun{fib}_\phi(\dep{b,\fun{refl}_{g(b)}})
        &\simeq
            \sum_{x\colon A}\;
            \sum_{p\colon g(f(x))\eq{A}g(b)}\;
            \sum_{q\colon f(x)\eq{B}b}
                p
                \eq{g(f(x))\eq{A}g(b)}
                \fun{ap}_g(q).
    \end{align*}
    By path inversion, the innermost\/ $\Sigma$-type is equivalent to the fiber\/ $\fun{fib}_{\fun{ap}_g}(p)$. Hence, we obtain
    \begin{align*}
        \fun{fib}_\phi(\dep{b,\fun{refl}_{g(b)}})
            &\simeq
                \sum_{x\colon A}\;
                \sum_{p\colon g(f(x))\eq{A}g(b)}
                    \fun{fib}_{\fun{ap}_g}(p)\\
            &\simeq
                \sum_{w\colon\fun{fib}_{g\circ f}(g(b))}
                    \fun{fib}_{\fun{ap}_g}(\pi_2(w))
                &&\text{; $\Sigma$-assoc.}
    \end{align*}
    As $f$ is an equivalence, $\fun{fib}_f(b)$ is contractible. Since $\fun{fib}_\phi(\dep{b,\fun{refl}_{g(b)}})\simeq\fun{fib}_f(b)$, it follows that the $\Sigma$-type above is also contractible. Because $g\circ f$ is an equivalence, the base space $\fun{fib}_{g\circ f}(g(b))$ is contractible as well. Hence, the projection
    \[
        \pi_1\colon
            \Bigl(
            \sum_{w\colon\fun{fib}_{g\circ f}(g(b))}
                \fun{fib}_{\fun{ap}_g}(\pi_2(w))
            \Bigr)
            \to
            \fun{fib}_{g\circ f}(g(b))
    \]
    is a map between contractible spaces, which implies it is an equivalence. Consequently, it has contractible fibers. By Proposition~\ref{prop:pi_1-fibers}, the fiber over $w$ is equivalent to $\fun{fib}_{\fun{ap}_g}(\pi_2(w))$. 

    Let $w\defeq\dep{x,p}$. Then the fiber $\fun{fib}_{\fun{ap}_g}(p)$ is contractible for all $x\colon A$ and $p\colon g(f(x))\eq{A}g(b)$. 
    
    To show that this holds for an arbitrary $y\colon B$ and $q\colon g(y)\eq{A}g(b)$, consider $x\defeq g(y)$. Then, $\eta(y)\colon f(x)\eq{B}y$. We can define the property of having contractible fibers as a type family over $B$:
    \[
        P(z)\defeq\prod_{p\colon g(z)\eq{A}g(b)}\fun{isContr}(\fun{fib}_{\fun{ap}_g}(p)).
    \]
    Since we established that $P(f(x))$ is inhabited, we can transport this witness along the path $\eta(y)$ to obtain a witness for $P(y)$. Because $y$ was arbitrarily chosen, the fibers of $\fun{ap}_g$ are contractible everywhere. Hence, $\fun{ap}_g$ is an equivalence. By symmetry, $\fun{ap}_f$ is also an equivalence.    
    %
    
\end{solution}

\begin{exr}
    For\/ $A,B\colon\univ$, define
    \[
        \fun{idtoqinv}_{A,B}\colon(A\eq\univ B)
            \to
            \sum_{f\colon A\to B}\fun{qinv}(f)
    \]
    by path induction in the obvious way. Let \textsl{$\fun{qinv}$-univalence} denote the modified form of the univalence axiom which asserts that for all\/ $A,B\colon\univ$ the function\/ $\fun{idtoqinv}_{A,B}$ has a quasi-inverse.
    \begin{enumerate}[a),font=\upshape]
        \item Show that $\fun{qinv}$-univalence can be used instead of univalence in the proof of function extensionality in\/ {\upshape\S\,\ref{UA-implies-FE}}.
        
        \item Show that $\fun{qinv}$-univalence can be used instead of univalence in the proof of\/ {\upshape Theorem~\ref{thm:qinv-is-not-mere-proposition}}.
        
        \item Show that $\fun{qinv}$-univalence is inconsistent (i.e.\ allows construction of an inhabitant of\/ $\fun0$).

        {\upshape Hint: Consider the fiber $\fun{fib}_{\fun{idtoqinv}_{A,A}}(\dep{\id_A,(\fun{refl},\fun{refl})})$.}
    \end{enumerate}
    Thus, the use of a ``good'' version of\/ $\fun{isequiv}$ is essential in the statement of univalence.
\end{exr}

\begin{solution}
    Let us begin by observing that $\fun{idtoqinv}$ is defined by path induction because, in the case $B\equiv A$ and $p\equiv\fun{refl}_A$, the identity map $\id_A\colon A\to A$ has the quasi-inverse $\dep{\id_A,(\fun{refl},\fun{refl})}$.

    \begin{enumerate}[a)]
        \item We need to analyze the following results:
        \begin{itemize}
            \item Lemma~\ref{lem:base-equivalence-implies-bundle-equivalence}. To derive this lemma from $\fun{qinv}$-univalence, it suffices to show that $e\colon A\simeq B$ implies $A\eq\univ B$, because the rest of the proof proceeds exactly as it does with standard univalence. From $e$, we can extract its underlying function $f\colon A\to B$ along with a quasi-inverse $\dep{g,(\eta,\vartheta)}\colon\fun{qinv}(f)$. This gives us the pair
            \[
                \dep{f,\dep{g,(\eta,\vartheta)}}
                \colon\sum_{f\colon A\to B}\fun{qinv}(f).
            \]
            Applying the inverse of $\fun{idtoqinv}_{A,B}$ to this pair produces the desired path $A\eq\univ B$.
            
            \item Proposition~\ref{prop:projection-equivalence}. This proposition holds under $\fun{qinv}$-univalence because it relies on univalence exclusively through Lemma~\ref{lem:base-equivalence-implies-bundle-equivalence}.
            
            \item Theorem~\ref{thm:univalence-implies-weak-funext}. The proof of this theorem invokes univalence only via Proposition~\ref{prop:projection-equivalence}.
            
            \item Theorem~\ref{thm:weak-funext-implies-funext}. The proof of this theorem does not rely on univalence at all.
        \end{itemize}

        \item We need to show that condition~a) of Lemma~\ref{lem:total-space-with-mere-proposition-fibers} follows from $\fun{qinv}$-univalence, i.e., that $\fun2\eq\univ\fun2$ is a set, provided $\fun{qinv}$-univalence.

        Hence, it suffices to prove that the type
        \[
            \sum_{f\colon\fun2\to\fun2}\fun{qinv}(f)
        \]
        is a set.
        
        Since $\fun2$ is a set by Theorem~\ref{thm:boolean-type-is-set}, the function type $\fun2\to\fun2$ is a set by part~f) of Theorem~\ref{thm:sets}. Note that we can use this part of the theorem because it relies on function extensionality, which we derived from $\fun{qinv}$-univalence in the previous part of this exercise that it follows from $\fun{qinv}$-univalence.
        
        In addition, $\fun{qinv}(f)$ is a set by Proposition~\ref{prop:qinv-is-a-set}, because that proposition is based on part~a) of Theorem~\ref{thm:mere-propositions}, which also requires function extensionality. Therefore, the result follows from part~e) of Theorem~\ref{thm:sets}.
            
        \item Consider the propositionally commutative triangle
        \[
            \begin{tikzcd}[row sep=large]
                A\eq\univ B
                        \arrow[r,"{\fun{idtoqinv}_{A,B}}"]
                        \arrow[rd,"\fun{idtoeqv}_{A,B}"']
                    &[4em]\sum_{f\colon A\to B}\fun{qinv}(f)
                        \arrow[d,"\phi"]
                    &[-2.5em]\scriptstyle\dep{f,\dep{g,(\eta,\vartheta)}}
                        \arrow[d,mapsto]\\
                    &\sum_{f\colon A\to B}\fun{isequiv}(f)
                    &\scriptstyle\dep{f,(\dep{g,\eta},\dep{g,\vartheta})}
            \end{tikzcd}
        \]
        
        
        As suggested, let $e\defeq\dep{\id_A,(\fun{refl},\fun{refl})}$, and consider
        \[
            \fun{fib}_{\fun{idtoqinv}_{A,A}}(e)
                \equiv \sum_{p\colon A\eq{\univ}A}
                    \fun{idtoqinv}_{A,A}(p)\eq{\Sigma_{A\to A}\fun{qinv}}e.
        \]
        By Theorem~\ref{thm:equivalence-fibers-are-contractible}, this type is contractible, with center of contraction $\dep{\fun{refl}_A,\fun{refl}_e}$. Let $\dep{p,\epsilon}$ be an arbitrary point in the fiber. Then,
        \[
            p\colon A\eq{\univ}A
            \quad\text{and}\quad
            \epsilon\colon\fun{idtoqinv}_{A,A}(p)\eq{}e,
        \]
        and we have a path
        \[
            \dep{p,\epsilon}
            \eq{\fun{fib}_{\fun{idtoqinv}_{A,A}}(e)}
            \dep{\fun{refl}_A,\fun{refl}_e}.
        \]
        Consequently, there exists a path $\alpha\colon p\eq{}\fun{refl}_A$ such that
        \[
            \fun{transport}^P(\alpha,\epsilon) \eq{} \fun{refl}_e,
        \]
        where the type family $P\colon(A\eq{\univ}A)\to\univ$ is defined by
        \[
            P(q)\defeq\fun{idtoqinv}_{A,A}(q) \eq{\Sigma_{A\to A}\fun{qinv}} e.
        \]
        Let $Q\colon\Sigma_{A\to A}\fun{qinv}\to\univ$ be defined as
        \[
            Q(\zeta)\defeq\zeta\eq{\Sigma_{A\to A}\fun{qinv}}e.
        \]
        Then $P\equiv Q\circ\fun{idtoqinv}_{A,A}$, and by Lemma~\ref{lem:post-composition-transport} and Proposition~\ref{prop:transport-decomposition-for-paths}, we have
        \[
            \fun{transport}^P(\alpha,\epsilon)
            \eq{} \fun{transport}^Q(\fun{ap}_{\fun{idtoqinv}_{A,A}}(\alpha),\epsilon)
            \eq{} \fun{ap}_{\fun{idtoqinv}_{A,A}}(\alpha)^{-1}\ct\epsilon.
        \]
        Hence, since the transported path equals $\fun{refl}_e$, we conclude that
        \[
            \epsilon\eq{}\fun{ap}_{\fun{idtoqinv}_{A,A}}(\alpha).
        \]
        It follows that given any $p\colon A\eq\univ A$, there is a path $\alpha\colon p\eq{}\fun{refl}_e$, such that the dependent pair $\dep{p,\fun{ap}_{\fun{idtoqinv}_{A,A}}(\alpha)}$ is a term in the fiber of $\fun{idtoqinv}_{A,A}$ at $e$.
    \end{enumerate}

\end{solution}
