\chapter{Sets \texorpdfstring{\&}{and} Logic}

\section{Sets}

\begin{defn}
    A type $A$ is a \textsl{set} if
    \[
        \fun{isSet}(A)\defeq\prod_{x,y\colon A}\;\prod_{p,q\colon x\eq Ay}
            p\eq{x\eq Ay}q
    \]
    is inhabited.
\end{defn}


\begin{thm}\fe\label{thm:sets}
    The following closure properties of sets hold:
    \begin{enumerate}[a),font=\upshape]
        \item The empty product\/ $\fun1$ is a set.
        \item The empty coproduct\/ $\fun0$ is a set.
        \item The type\/ $\N$ of natural numbers is a set.
        \item If\/ $A$ and\/ $B$ are sets, then\/ $A\times B$ is a set.
        \item If\/ $A\colon\univ$ is a set and\/ $P\colon A\to\univ$ is a type family such that\/ $P(a)$ is a set for all\/ $a\colon A$, then\/ $\Sigma_AP$ is a set.
        \item If\/ $A$ is any type and\/ $B\colon A\to\univ$ is a type family such that each\/ $B(x)$ is a set, then\/ $\prod_{x\colon A}B(x)$ is a set.\footnote{Function extensionality is used here.}
    \end{enumerate}
\end{thm}

\begin{proof}${}$
    \begin{enumerate}[a),font=\upshape]
        \item This is a direct consequence of Theorem~\ref{thm:1-equivalence}: for any paths $p,q\colon x\eq{\fun1}y$, we have $\fun{encode}(p)\eq{\fun1}\fun{encode}(q)$ because all elements of $\fun1$ are propositionally equal to $\star$. Since $\fun{encode}$ is an equivalence, it is an embedding, which implies that $p\eq{x\eq{\fun1}y}q$.

        \item A witness for $\fun{isSet}(\fun0)$ is given by:
        \[
            \lambda x.\,
                \fun{rec}_{\fun0}
                    \Bigl(
                        \prod_{y\colon\fun0}\;
                        \prod_{p,q\colon x\eq{\fun0}y}
                            p\eq{x\eq{\fun0}y}q
                    \Bigr)\colon\fun{isSet}(\fun0).
        \]

        \item To construct a proof, it suffices to define a witness of the motive
        \[
            E(x)\defeq
                \prod_{y\colon\N}\;
                \prod_{c,d\colon\fun{Code}(x,y)}c\eq{\fun{Code}(x,y)}d.
        \]
        This requires establishing a base case and an inductive step.
        
        In the base case $x\equiv0$, the motive becomes
        \[
            E(0)\defeq
                \prod_{y\colon\N}\;\prod_{c,d\colon\fun{Code}(0,y)}
                    c\eq{\fun{Code}(0,y)}d.
        \]
        We proceed by induction on $y$ using the motive 
        \[
            E_0(y)\defeq\prod_{c,d\colon\fun{Code}(0,y)}c\eq{\fun{Code}(0,y)}d.
        \]
        
        \begin{enumerate}[-]
            \item If $y\equiv0$, then $\fun{Code}(0,0)\equiv\fun1$. Thus, given $c,d\colon\fun1$, we have $c\eq{\fun1}d$, as both equal $\star$.
            
            \item If $y\equiv\fun{succ}(m)$, then $\fun{Code}(0,\fun{succ}(m))\equiv\fun0$. Hence, given an inhabitant $c\colon\fun0$, we have
            \[
                \fun{rec}_{\fun0}
                    \Bigl(
                        \prod_{d\colon\fun0}c\eq{\fun0}d
                    \Bigr)(c)
                        \colon \prod_{d\colon\fun0}c\eq{\fun0}d.
            \]
        \end{enumerate}
        
        For the inductive step, we assume the hypothesis $h\colon E(n)$ for a given $n\colon\N$. We must construct a term of type
        \[
            E(\fun{succ}(n))\equiv
            \prod_{y\colon\N}\;\prod_{c,d\colon\fun{Code}(\fun{succ}(n),y)}
                c\eq{\fun{Code}(\fun{succ}(n),y)}d.
        \]
        We proceed by induction on $y$ with the motive
        \[
            E_{\fun{succ}(n)}(y)
                \defeq\prod_{c,d\colon\fun{Code}(\fun{succ}(n),y)}
                    c\eq{\fun{Code}(\fun{succ}(n),y)}d.
        \]
        
        \begin{enumerate}[-]
            \item If $y\equiv0$, then $\fun{Code}(\fun{succ}(n),0)\equiv\fun0$. As before, applying the induction principle of the empty type to $c\colon\fun0$ trivially yields the required path.
            
            \item If $y\equiv\fun{succ}(m)$, then $\fun{Code}(\fun{succ}(n),\fun{succ}(m))\equiv\fun{Code}(n,m)$. Then, $h(m)(c,d)$ yields the required path of type $c\eq{\fun{Code}(n,m)}d$.
        \end{enumerate}

        \item Let $p,q\colon x\eq{A\times B}y$. As shown in \nameref{equality-in-cartesian-products}, we have path codifications
        \[
            \fun{pair}^\times(p)
                \equiv(\fun{ap}_{\pi_1}(p),\fun{ap}_{\pi_2}(p))
            \quad\text{and}\quad
            \fun{pair}^\times(q)
                \equiv(\fun{ap}_{\pi_1}(q),\fun{ap}_{\pi_2}(q)).
        \]
        Set
        \[
            p_1\defeq\fun{ap}_{\pi_1}(p),\quad
            p_2\defeq\fun{ap}_{\pi_2}(p),\quad
            q_1\defeq\fun{ap}_{\pi_1}(q),\quad
            q_2\defeq\fun{ap}_{\pi_2}(q)
        \]
        and
        \[
            x_1\defeq\pi_1(x),\quad x_2\defeq\pi_2(x),\quad
            y_1\defeq\pi_1(y),\quad y_2\defeq\pi_2(y).
        \]
        Since the hypothesis implies that $A$ and $B$ are sets, we obtain
        \[
            p_1\eq{A}q_1
            \quad\text{and}\quad
            p_2\eq{B}q_2,
        \]
        by using the functions $\fun{in}_b\defeq a\mapsto(a,b)$ and $\fun{in}_a\defeq b\mapsto(a,b)$ and applying the action on paths, we obtain
        \begin{align*}
            (p_1,p_2)
                &\eq{(x_1\eq{A}y_1)\times(x_2\eq{B}y_2)}
                    (p_1,q_2)\\
                &\eq{(x_1\eq{A}y_1)\times(x_2\eq{B}y_2)}
                    (q_1,q_2).
        \end{align*}
        Thus, $\fun{pair}^\times(p)\eq{(x_1\eq{A}y_1)\times(x_2\eq{B}y_2)}\fun{pair}^\times(q)$, and the result follows by decoding with $\fun{pair}^=$.
        
        \item Let $\omega,\omega'\colon\Sigma_AP$. The codification of a path $p\colon\omega\eq{\Sigma_AP}\omega'$ is defined as
        \[
            \fun{encode}(p)\equiv
            \dep{\fun{ap}_{\pi_1}(p),\fun{apd}_{\pi_2}(p)}.
        \]
        If $q$ is another path with the same ends,
        \[
            \fun{encode}(q)\equiv
            \dep{\fun{ap}_{\pi_1}(q),\fun{apd}_{\pi_2}(q)}.
        \]
        Introduce
        \[
            x\defeq\pi_1(\omega),\quad y\defeq\pi_2(\omega),\quad
            x'\defeq\pi_1(\omega'),\quad y'\defeq\pi_2(\omega').
        \]
        Given that
        \[
            \fun{ap}_{\pi_1}(p),\,\fun{ap}_{\pi_1}(q)\colon x\eq Ax',
        \]
        the hypothesis on $A$ implies the existence of a witness
        \[
            \epsilon\colon\fun{ap}_{\pi_1}(p)
                \eq{x\eq Ax'}
                \fun{ap}_{\pi_1}(q).
        \]
        Set
        \[
            p_1\defeq\fun{ap}_{\pi_1}(p),\quad p_2\defeq\fun{apd}_{\pi_2}(p),
            \quad
            q_1\defeq\fun{ap}_{\pi_1}(q),\quad q_2\defeq\fun{apd}_{\pi_2}(q).
        \]
        Then,
        \[
            p_2\colon(p_1)_*(y)\eq{P(x')}y'
            \quad\text{and}\quad
            q_2\colon(q_1)_*(y)\eq{P(x')}y'.
        \]
        If we define
        \begin{align*}
            R&\colon(x\eq Ax')\to\univ\\
            R(r)&\defeq r_*(y)\eq{P(x')}y',
        \end{align*}
        then, $p_2\colon R(p_1)$ and $q_2\colon R(q_1)$.
        
        Since $\epsilon\colon p_1\eq{x\eq Ax'}q_1$, it follows that  $\epsilon_*\colon R(p_1)\to R(q_1)$ satisfies
        \[
            \epsilon_*(p_2)
                \colon(q_1)_*(y)\eq{P(x')}y'.
        \]
        Thus, the hypothesis on $P(x')$ implies the existence of a witness
        \[
            \rho\colon\epsilon_*(p_2)\eq{R(q_1)}q_2.
        \]
        Therefore, the pair of paths $(\epsilon,\rho)$ constructs a path
        \[
            \dep{p_1,p_2}
            \eq{\Sigma_{x\eq Ax'}R}
            \dep{q_1,q_2},
        \]
        which definitionally means that $\fun{encode}(p)\eq{}\fun{encode}(q)$. Since $\fun{encode}$ is an equivalence, we conclude that $\Sigma_AP$ is a set.

        \item Set $\Gamma\defeq\prod_{x\colon A}B(x)$ and consider two paths $p,q\colon f\eq{\Gamma}g$. Then,
        \[
            \fun{happly}(p),\,\fun{happly}(q)
                \colon\prod_{x\colon A}f(x)\eq{B(x)}g(x).
        \]
        Hence, for any $x\colon A$, we have
        \[
            \fun{happly}(p)(x),\,\fun{happly}(q)(x)\colon f(x)\eq{B(x)}g(x).
        \]
        Since $B(x)$ is a set, any two paths in it are equal, so we obtain a path
        \[
            \eta(x)\colon\fun{happly}(p)(x)
                \eq{f(x)\eq{B(x)}g(x)}\fun{happly}(q)(x)
        \]
        that defines a homotopy $\eta\colon\fun{happly}(p)\sim\fun{happly}(q)$. Thus, by function extensionality, we obtain a path
        \[
            \fun{funext}(\eta)\colon\fun{happly}(p)\eq{f\sim g}\fun{happly}(q).
        \]
        Applying the action on paths of $\fun{funext}$ yields
        \[
            \fun{ap}_{\fun{funext}}(\fun{funext}(\eta))\colon
                \fun{funext}(\fun{happly}(p))
                \eq{f\eq{\Gamma}g}
                \fun{funext}(\fun{happly}(q)).
        \]
        Since $\fun{funext}$ is the quasi-inverse of $\fun{happly}$, this directly decodes to a path $p\eq{f\eq{\Gamma}g}q$, proving that $\Gamma$ is a set.
    \end{enumerate}
\end{proof}

\begin{defn}\label{defn:1-type}
    A type $A$ is a \textsl{$1$-type} if for all $x,y\colon A$, $p,q\colon x\eq Ay$, and $r,s\colon p\eq{x\eq Ay}q$, we have $r\eq{p\eq{x\eq Ay}q}s$.
\end{defn}

\begin{rem}
    Equivalently, the definition means that $A$ is a $1$-type when the identity type $x\eq Ay$ is a set for all $x,y\colon A$.
\end{rem}

\begin{lem}\label{lem:set-implies-id-types-are-sets}
    Every set is a\/ $1$-type.
\end{lem}

\begin{proof}
    Let $A$ be a set. We have to prove that, given $x,y\colon A$ and $p,q\colon x\eq Ay$, any two paths $r,s\colon p\eq{x\eq Ay}q$ are equal.
    
    By definition, there is a dependent function
    \begin{align*}
        f&\colon\prod_{x,y\colon A}\;\prod_{p,q\colon x\eq Ay}
            p\eq{x\eq Ay}q.\\
    \intertext{Fix $x,y\colon A$ and $p\colon x\eq Ay$ and define}
        g&\colon\prod_{q\colon x\eq Ay}p\eq{x\eq Ay}q\\
        g(q)&\defeq f(x,y,p,q).
    \end{align*}
    Suppose that $q\colon x\eq Ay$ and that $r\colon p\eq{x\eq Ay}q$. By Lemma~\ref{lem:dependent-map} applied to $x\eq Ay$, $g$, and $P(\zeta)\defeq p\eq{x\eq Ay}\zeta$, we obtain
    \[
        \fun{apd}_g(r)\colon r_*(g(p))\eq{p\eq{x\eq Ay}q}g(q),
    \]
    where
    \[
        r_*\defeq\fun{transport}^P(r)\equiv\fun{transport}^{\zeta\mapsto (p\eq{x\eq Ay}\zeta)}(r).
    \]
    By Proposition~\ref{prop:transport-decomposition-for-paths}, we have
    \[
        r_*(g(p))\eq{p\eq{x\eq Ay}q}g(p)\ct r,
    \]
    which implies that
    \[
        g(p)\ct r\eq{p\eq{x\eq Ay}q}g(q)
    \]
    is inhabited. Since this conclusion is also valid for $s$, we deduce that
    \[
        g(p)\ct r\eq{p\eq{x\eq Ay}q}g(q)
        \quad\text{and}\quad
        g(p)\ct s\eq{p\eq{x\eq Ay}q}g(q)
    \]
    are inhabited. It follows that
    \[
        g(p)\ct r\eq{p\eq{x\eq Ay}q}g(p)\ct s,
    \]
    is inhabited, which implies that $r\eq{p\eq{x\eq Ay}q}s$ is inhabited as well.
    
\end{proof}

\begin{xmpl}\label{xmpl:univ-isnt-a-set}
    The universe is not a set.
    
    To see this, consider the Boolean type $\fun2$. As shown in Exercise~\ref{exr:2-simeq-2-simeq-2}, we have an equivalence $(\fun2\simeq\fun2)\simeq\fun2$. Since the lemma in that exercise shows that $0_{\fun2}\ne1_{\fun2}$, we deduce that there are two distinct equivalences from $\fun2$ to $\fun2$ (namely, the identity and the swap function). By the univalence axiom, these two equivalences correspond to two distinct paths in the identity type $\fun2\eq{\univ}\fun2$, contradicting the definition of a set.
\end{xmpl}

\begin{lem}\label{lem:A+negA-to-negnegA-to-A}
    For any type\/ $A$, there is a map\/ $(A+\neg A)\to(\neg\neg A\to A)$.
\end{lem}

\begin{proof}
    To construct a map from $A+\neg A$ to $\neg\neg A\to A$, we first define the constant function
    \begin{align*}
        c\colon A&\to(\neg\neg A\to A)\\
        a&\mapsto(\beta\mapsto a).
    \end{align*}
    Second, fix $b\colon\neg A$. Using recursion, we obtain
    \begin{align*}
        \rho_b\colon\neg\neg A&\to A\\
        \beta&\mapsto\fun{rec}_{\fun0}(A)(\beta(b)).
    \end{align*}
    This yields a function
    \begin{align*}
        \rho\colon\neg A&\to(\neg\neg A\to A)\\
        b&\mapsto\rho_b.
    \end{align*}
    By recursion on $A+\neg A$, we arrive at the desired map:
    \begin{align*}
        \varphi_A&\colon(A+\neg A)\to(\neg\neg A\to A)\\
        \varphi_A(x)&\defeq\fun{rec}_{A+\neg A}(\neg\neg A\to A,c,\rho,x).
    \end{align*}
\end{proof}


\begin{thm}\fe
    It is not the case that for all\/ $A\colon\univ$ we have\/ $\neg\neg A\to A$.
\end{thm}

\begin{proof}
    The statement means that
    \begin{equation}\label{eq:notnotA-to-A}
        \Bigl(\prod_{A\colon\univ}\neg\neg A\to A\Bigr)\to\fun0
    \end{equation}
    is inhabited. Consider the type family
    \begin{align*}
        P&\colon\univ\to\univ\\
        P(A)&\defeq\neg\neg A\to A
    \end{align*}
    and the function
    \begin{align*}
        \fun{not}\colon\fun2&\to\fun2\\
        0_{\fun2}&\mapsto1_{\fun2}\\
        1_{\fun2}&\mapsto0_{\fun2},
    \end{align*}
    introduced in Exercise~\ref{exr:not}.
     
    Being an involution, $\fun{not}$ defines an equivalence $\fun2\simeq\fun2$. Hence, by the univalence axiom, we obtain $p\defeq\fun{ua}(\fun{not})\colon\fun2\eq{\univ}\fun2$.

    Given $f\colon\prod_{A\colon\univ}\neg\neg A\to A\equiv\prod_{A\colon\univ}P(A)$, we have $f(\fun2)\colon P(\fun2)\equiv\neg\neg\fun2\to\fun2$ and
    \begin{equation}\label{eq:f(2)}
        \fun{apd}_f(p)\colon\fun{transport}^P(p,f(\fun2))\eq{P(\fun2)}f(\fun2).
    \end{equation}
    By Proposition~\ref{prop:non-dependent-transport-composition} for $A(X)\defeq\neg\neg X$, $B\defeq\id_\univ$, and $g\equiv f(\fun2)$, we have $P\equiv A\to B$, and so
    \begin{align*}
        \fun{transport}^P(p,f(\fun2))
            &\eq{P(\fun2)}
                \fun{transport}^{\id_\univ}(p)
                    \circ f(\fun2)
                    \circ 
                    \fun{transport}^{X\mapsto\neg\neg X}(p^{-1}).
    \end{align*}
    Evaluating this equality at any $\zeta\colon\neg\neg\fun2$, we obtain
    \begin{align*}
        \fun{transport}^P\bigl(p,f(\fun2)\bigr)(\zeta)
            \eq{\fun2}
            \fun{transport}^{\id_\univ}\bigl(p,f(\fun2)(\xi)\bigr),
    \end{align*}
    where
    \[
        \xi\defeq\fun{transport}^{X\mapsto\neg\neg X}(p^{-1},\zeta).
    \]
    We claim that $\xi\eq{\neg\neg\fun2}\zeta$. To see this, note that for any $c\colon\neg\fun2$, both $\xi(c)$ and $\zeta(c)$ are elements of $\fun0$. Applying $\fun{rec}_{\fun0}$ we deduce that $\xi(c)\eq{\fun0}\zeta(c)$. By function extensionality, $\xi\eq{\neg\neg\fun2}\zeta$. 

    Applying $\fun{happly}$ to \eqref{eq:f(2)} and evaluating at $\zeta$, we also know that
    \[
        \fun{transport}^P\bigl(p,f(\fun2)\bigr)(\zeta)\eq{\fun2}f(\fun2)(\zeta).
    \]
    Combining these paths yields
    \[
        \fun{transport}^{\id_\univ}
            \bigl(p,f(\fun2)(\zeta)\bigr)\eq{\fun2}f(\fun2)(\zeta).
    \]
    By Theorem~\ref{thm:idtoeqv-def}, we deduce that
    \[
        \fun{not}\bigl(f(\fun2)(\zeta)\bigr)\eq{\fun2}f(\fun2)(\zeta).
    \]
    Take $u\colon\neg\fun2$. Then $u\colon\fun2\to\fun0$ and we can evaluate $u$ at $0_{\fun2}$, i.e., we can define
    \[
        e\colon\neg\neg\fun2,\quad e\defeq\lambda u.\,u(0_{\fun2}).
    \]
    Then $b\defeq f(\fun2)(e)$ is an element of $\fun2$. The equality above constructs a path
    \[
        \epsilon\colon\fun{not}(b)\eq{\fun2}b.
    \]
    We can define
    \begin{align}\label{eq:not-eq-id->0}
        &\psi\colon\prod_{x\colon\fun2}(\fun{not}(x)\eq{\fun2}x)\to\fun0\\
        &\psi(0_{\fun2})\defeq\fun{neq}\nonumber\\
        &\psi(1_{\fun2})\defeq\lambda p.\,\fun{neq}(p^{-1}),\nonumber
    \end{align}
    where $\fun{neq}\colon(1_{\fun2}\eq{\fun2}0_{\fun2})\to\fun0$ is the function defined in the lemma of Exercise~\ref{exr:2-simeq-2-simeq-2}.
    
    In consequence, the function
    \[
        f\mapsto\psi(b,\epsilon)
    \]
    is a witness of the type given in \eqref{eq:notnotA-to-A}.

\end{proof}

\begin{rem}\label{rem:notid}
    The function\/ $\psi$ of \eqref{eq:not-eq-id->0} allows us to define
    \begin{align*}
        \fun{notid}\colon\Bigl(\sum_{x\colon\fun2}\fun{not}(x)\eq{\fun2}x\Bigr)
            &\to\fun0\\
        \dep{x,p}&\mapsto\psi(\dep{x,p}).
    \end{align*}
\end{rem}

\begin{cor}\label{cor:A+negA-not-always-true}
    It is not the case that for all\/ $A\colon\univ$ we have\/ $A+\neg A$.
\end{cor}

\begin{proof}
    The statement means that
    \[
        \Bigl(\prod_{A\colon\univ}A+\neg A\Bigr)\to\fun0
    \]
    is inhabited.

    By Lemma~\ref{lem:A+negA-to-negnegA-to-A}, for any $A\colon\univ$ there is a map $\varphi_A\colon(A+\neg A)\to(\neg\neg A\to A)$. This family of functions leads to
    \begin{align*}
        \Phi\colon\Bigl(\prod_{A\colon\univ}A+\neg A\Bigr)
            &\to\prod_{A\colon\univ}\neg\neg A\to A\\
        g&\mapsto\lambda A.\,\varphi_A(g(A)).
    \end{align*}
    The conclusion follows by composing $\Phi$ with the function given in \eqref{eq:notnotA-to-A}.
\end{proof}

\section{Mere Propositions}

\begin{defn}
    A type $P$ is a \textsl{mere proposition} (a.k.a.~\textsl{subsingleton}, a.k.a.~\textsl{h-proposition}) if the type
    \[
        \fun{isProp}(P)\defeq\prod_{x,y\colon P}x\eq Py
    \]
    is inhabited.\footnote{We have adopted the function name given in \citep{hottbook}.}
\end{defn}

\begin{rem}\label{rem:ids-in-a-set-are-mere-propositions}
    If $A$ is a set, then $x\eq Ay$ is a mere proposition for all $x,y\colon A$.
\end{rem}

\begin{lem}\label{lem:mere-prop-and-1}
    Every inhabited mere proposition is equivalent to the unit type\/~$\fun1$.
\end{lem}

\begin{proof}
    Fix $\pi\colon\fun{isProp}(P)$ and $a\colon P$. Define
    \begin{align*}
        \phi\colon P&\to\fun1 & \psi\colon\fun1&\to P\\
        x&\mapsto\fun\star & \fun\star&\mapsto a.
    \end{align*}
    Then,
    \[
        \phi(\psi(\fun\star))\equiv\phi(a)\equiv\fun\star,
    \]
    which implies by induction on $\fun1$ that $\phi(\psi(z))\eq{\fun1}z$ for all\/ $z\colon\fun1$. Conversely, for any $x\colon P$,
    \[
        \psi(\phi(x))\equiv\psi(\fun\star)\equiv a \stackrel{\pi(a,x)}{\eq{P}} x,
    \]
    showing that $\lambda x.\,\pi(a,x)\colon\psi\circ\phi\sim\id_P$.
\end{proof}

\begin{lem}\label{lem:mere-proposition-equivalence}
    Let\/ $P$ and\/ $Q$ be mere propositions. Then any pair of functions\/ $P\to Q$ and\/ $Q\to P$ yields an equivalence\/ $P\simeq Q$.
\end{lem}

\begin{proof}
    Let $\phi\colon P\to Q$ and $\psi\colon Q\to P$, and fix two witnesses $\pi\colon\fun{isProp}(P)$ and $\sigma\colon\fun{isProp}(Q)$. 

    To show that $\phi$ is an equivalence with quasi-inverse $\psi$, we must provide homotopies $\eta\colon\phi\circ\psi\sim\id_Q$ and $\vartheta\colon\psi\circ\phi\sim\id_P$.

    Given any $y\colon Q$, by hypothesis, 
    \[
        \sigma(\phi(\psi(y)),y)\colon\phi(\psi(y))\eq Qy,
    \]
    and we can define the  homotopy~$\eta$.

    Similarly,
    \[
        \pi(\psi(\phi(x)),x)\colon\psi(\phi(x))\eq Px,
    \]
    which yields the homotopy $\vartheta$.
\end{proof}

\begin{rem}
    Lemma \ref{lem:mere-prop-and-1} could have been proved as a consequence of Lemma~\ref{lem:mere-proposition-equivalence}.
\end{rem}

\begin{lem}\label{lem:mere-propositions-are-sets}
    Every mere proposition is a set.
\end{lem}

\begin{proof}
    Let $A$ be a mere proposition. Fix $\pi\colon\fun{isProp}(A)$, two points $x,y\colon A$, and paths $p,q\colon x\eq{A}y$. 
    
    Consider the dependent function
    \[
        f\defeq\lambda z.\,\pi(x,z)\colon\prod_{z\colon A}P(z),
    \]
    where $P\colon A\to\univ$ is given by $P(z)\defeq x\eq{A}z$. We obtain
    \[
        \fun{apd}_f(p)\colon\fun{transport}^P\bigl(p,\pi(x,x)\bigr)\eq{x\eq{A}y}\pi(x,y).
    \]
    By Proposition~\ref{prop:transport-decomposition-for-paths}, we have
    \[
        \fun{transport}^P\bigl(p,\pi(x,x)\bigr)\eq{x\eq{A}y}\pi(x,x)\ct p.
    \]
    Therefore, by path concatenation, we have
    \[
        \pi(x,x)\ct p\eq{x\eq{A}y}\pi(x,y).
    \]
    By symmetry, we also obtain
    \[
        \pi(x,x)\ct q\eq{x\eq{A}y}\pi(x,y).
    \]
    Therefore,
    \begin{align*}
        p &\eq{x\eq{A}y} \pi(x,x)^{-1}\ct\pi(x,y) \\
          &\eq{x\eq{A}y} q.
    \end{align*}
\end{proof}

\begin{rem}\label{rem:ids-in-mere-propositions-are-singletons}
    Combining the lemma with Remark~\ref{rem:ids-in-a-set-are-mere-propositions}, we deduce that, if $P$ is a mere proposition, $x\eq Py$ is also a mere proposition for all $x,y\colon P$.
\end{rem}


\begin{thm}\label{thm:boolean-type-is-set}
    The boolean type\/ $\fun2$ is a set.
\end{thm}

\begin{proof}
    Let $x,y\colon\fun2$ and consider two paths $p,q\colon x\eq{\fun2}y$. 
    
    By Theorem~\ref{thm:bool-equality}, the map $\fun{encode}(x,y)$ is an equivalence, and therefore an embedding. Thus, to prove $p\eq{x\eq{\fun2}y}q$, it suffices to show that $\fun{Code}(x,y)$ is a mere proposition for any $x$ and $y$; that is, we need to construct an inhabitant of
    \[
        \prod_{x,y\colon\fun2}\fun{isProp}(\fun{Code}(x,y)).
    \]
    We proceed to construct this witness by double induction on $x$ and $y$:
    \begin{enumerate}[-]
        \item If $x\equiv0_{\fun2}$ and $y\equiv0_{\fun2}$, or $x\equiv1_{\fun2}$ and $y\equiv1_{\fun2}$, the type $\fun{Code}(x,y)$ is definitionally equal to $\fun1$, which is a mere proposition.
        
        \item If $x\equiv0_{\fun2}$ and $y\equiv1_{\fun2}$, or $x\equiv1_{\fun2}$ and $y\equiv0_{\fun2}$, the type $\fun{Code}(x,y)$ is definitionally equal to $\fun0$, which is trivially a mere proposition.
    \end{enumerate}
\end{proof}

\begin{prop}\fe\label{prop:isSet-and-isProp-are-mere-propositions}
    For any type\/ $A$, the types\/ $\fun{isProp}(A)$ and\/ $\fun{isSet}(A)$ are mere propositions.
\end{prop}

\begin{proof}
    Consider $f,g\colon\fun{isProp}(A)$. To show that $\fun{isProp}(A)$ is a mere proposition, we must show that $f\eq{\fun{isProp}(A)}g$. By function extensionality, it suffices to fix $x,y\colon A$ and show that $f(x,y)\eq{x\eq{A}y}g(x,y)$. The terms evaluate as
    \[
        f(x,y)\colon x\eq{A}y
        \quad\text{and}\quad
        g(x,y)\colon x\eq{A}y.
    \]
    Since $f$ is a witness that $A$ is a mere proposition, Lemma~\ref{lem:mere-propositions-are-sets} implies that $A$ is a set. Therefore, $x\eq{A}y$ is a mere proposition and so $f(x,y)\eq{x\eq{A}y}g(x,y)$, as desired.

    In the case where $f,g\colon\fun{isSet}(A)$, fix $x,y\colon A$ and $p,q\colon x\eq{A}y$. Then,
    \[
        f(x,y,p,q),\,g(x,y,p,q)\colon p\eq{x\eq{A}y}q.
    \]
    The existence of $f$ implies that $A$ is a set. So, by Lemma~\ref{lem:set-implies-id-types-are-sets}, $x\eq{A}y$ is a set. Thus, $p\eq{x\eq{A}y}q$ is a mere proposition. Therefore, $f(x,y,p,q)\eq{p\eq{x\eq{A}y}q}g(x,y,p,q)$. The proof concludes by function extensionality.
\end{proof}

\begin{defns}\label{defn:lem}${}$
    \begin{enumerate}[-]
        \item A type $A$ is \textsl{decidable} if $A+\neg A$ is inhabited.
        
        \item A family $C\colon A\to\univ$ is \textsl{decidable} if $\prod_{a\colon A}C(a)+\neg C(a)$ is inhabited.
        
        \item The \textsl{law of excluded middle} is the type
        \[
            \fun{Lem}\defeq\prod_{A\colon\univ}\fun{isProp}(A)\to(A+\neg A).
        \]
        In other words, the law of excluded middle states that all mere propositions are decidable.
    \end{enumerate}
\end{defns}

\begin{prop}\label{prop:decidability-of-equality-on-N}${}$
    \begin{enumerate}[a), font=\upshape]
        \item The types\/ $\fun0$ and\/ $\fun1$ are both decidable.

        \item If\/ $A$ and\/ $B$ are decidable, then\/ $A+B$ is decidable.

        \item Equality on\/ $\N$ is decidable; that is, given\/ $n,m\colon\N$, we have 
        \[
            (n\eq{\N} m)+\neg(n\eq{\N} m).
        \]

        \item Given $n,m\colon\N$, if\/ $\neg(n<m)$ and $\neg(m<n)$ then\/ $n\eq\N m$.

        %\item Given $n,m\colon\N$, we have $(n<\fun{succ}(m))\simeq(n\le m)$.

        \item Given $n,m\colon\N$, if\/ $n<\fun{succ}(m)$ then\/ $n\le m$.
    \end{enumerate}   
\end{prop}

\begin{proof}${}$
    \begin{enumerate}[a), font=\upshape]
        \item We have the terms $\fun{inr}(\id_{\fun0})\colon\fun0+\neg\fun0$ and $\fun{inl}(\ast)\colon\fun1+\neg\fun1$.
        
        \item The diagram below captures the cases needed to define a map
        \[
            \bigl((A+\neg A)\times(B+\neg B)\bigr)\to\bigl((A+B)+\neg(A+B)\bigr).
        \]
        \[
            \begin{tikzcd}[column sep=-0.8cm]
                    &\fun0\\
                A
                    \arrow[rd,"\fun{inl}"']
                    \arrow[ru,"u"]
                    &&B
                    \arrow[ld,"\fun{inr}"]
                    \arrow[lu,"v"']\\
                &A+B
                    \arrow[d,"\fun{inl}"']
                    \arrow[uu,"u+v"]\\
                &(A+B)+\neg(A+B)
            \end{tikzcd}
        \]
        More precisely, let $C\defeq(A+B)+\neg(A+B)$. By the universal property of the product, we can equivalently define a curried map
        \[
            f\colon(A+\neg A)\to(B+\neg B)\to C.
        \]
        To construct $f$, we apply the recursion principle for $A+\neg A$. 
        
        For the left branch, we take
        \[
            \phi\defeq\lambda a.\,\lambda y.\,\fun{inl}(\fun{inl}(a))\colon A\to((B+\neg B)\to C).
        \]
        
        For the right branch, we define
        \begin{align*}
            \psi&\colon\neg A\to((B+\neg B)\to C)\\
            \psi(u)&\defeq\fun{rec}_{B+\neg B}(C,\fun{inl}\circ\fun{inr},\lambda v.\,\fun{inr}(\fun{rec}_{A+B}(\fun0,u,v))).
        \end{align*}
        
        Combining these branches, we obtain
        \[
            f\defeq\fun{rec}_{A+\neg A}((B+\neg B)\to C,\phi,\psi).
        \]
        \item Using the encoding and decoding of equalities in $\N$ introduced in \S\,\ref{equality-in-N}, it is straightforward to construct a witness
        \[
            \epsilon\colon
                (\fun{Code}(n,m)\simeq\fun0)
                +(\fun{Code}(n,m)\simeq\fun1).
        \]
        Let $E$ be the coproduct above and $C\defeq\fun{Code}(n,m)+\neg\fun{Code}(n,m)$. To construct an inhabitant of $C$, it suffices to define a map from $E$ to $C$. We proceed by recursion on the coproduct $E$.
        \begin{enumerate}[-]
            \item For the left case, given an equivalence $\dep{f_0,(\eta_0,\vartheta_0)}\colon\fun{Code}(n,m)\simeq\fun0$, since $f_0\colon\fun{Code}(n,m)\to\fun0$, we can specify
            \[
                g_0(\dep{f_0,(\eta_0,\vartheta_0)})
                    \defeq\fun{inr}(f_0).
            \]
            \item For the right case, given an equivalence $\dep{f_1,(\eta_1,\vartheta_1)}\colon\fun{Code}(n,m)\simeq\fun1$, we specify
            \[
                g_1(\dep{f_1,(\eta_1,\vartheta_1)})
                    \defeq\fun{inl}(f_1^{-1}(\ast)).
            \]
        \end{enumerate}
        As a result, we obtain
        \[
            \fun{rec}_E(C,g_0,g_1,\epsilon)\colon C.
        \]

        \item Let $u\colon\neg(n<m)$ and $v\colon\neg(m<n)$ be two witnesses. By part~c) the type $(n\eq\N m)+\neg(n\eq\N m)$ is inhabited. Therefore, to prove that $n\eq\N m$ it suffices to define a map
        \[
            \bigl((n\eq\N m)+\neg(n\eq\N m)\bigr)\to(n\eq\N m).
        \]
        We proceed by recursion on the coproduct.
        \begin{enumerate}[-]
            \item For the left case, we take the identity $\id_{n\eq\N m}$.
            
            \item For the right case, consider $z\colon\neg(n\eq\N m)$. By Lemma~\ref{lem:order-in-N}, there is a term
            \[
                w\colon(n\le m)+(m\le n).
            \]
            We will use $z$ to define a map
            \[
                \zeta_z\colon ((n\le m)+(m\le n))\to(n\eq\N m),
            \]
            and $z\mapsto\zeta_z(w)$ will provide the required map 
            \[
                \neg(n\eq\N m)\to(n\eq\N m).
            \]
            We proceed by recursion on the coproduct domain:
            \begin{itemize}
                \item If $r\colon n\le m$, the pair $(r,z)$ inhabits $(n\le m)\times\neg(n\eq\N m)$, which is definitionally $n<m$. Applying $u$ to this pair yields a term in~$\fun0$. Hence, we can apply $\fun{rec}_{\fun0}(n\eq\N m)$ to obtain our goal.
                
                \item If $s\colon m\le n$, then $(s,z')$, with $z'\defeq\lambda q\colon(m\eq\N n).\,z(q^{-1})$, is a term of $m<n$. Thus, $v(s,z')\colon\fun0$ and
                \[
                    \fun{rec}_{\fun0}(n\eq\N m,v(s,z'))\colon(n\eq\N m).
                \]
            \end{itemize}
        \end{enumerate}

        \item By definition $(n<\fun{succ}(m))\equiv(n\le \fun{succ}(m))\times\neg(n\eq\N\fun{succ}(m))$. Take $(p,u)$ in the \rhs with 
        \[
            p\equiv\dep{k,q},
            \quad q\colon n+k\eq\N\fun{succ}(m),
            \quad\text{and }u\colon(n\eq\N\fun{succ}(m))\to\fun0.
        \]
        Since $(k\eq\N0)+\neg(k\eq\N0)$, we deduce from $u$ and $q$ that $\neg(k\eq\N0)$. Then, by Lemma~\ref{lem:order-in-N}, $k\eq\N\fun{succ}(\fun{pred}(k))$. It follows that
        \[
            \fun{succ}(n+\fun{pred}(k))\eq\N\fun{succ}(m).
        \]
        Hence, $n+\fun{pred}(k)\eq\N m$, which proves that $n\le m$.
    \end{enumerate}
\end{proof}

\begin{lem}\label{lem:total-space-with-mere-proposition-fibers}
    Let\/ $P\colon A\to\univ$ be a type family such that\/ $P(x)$ is a mere proposition for all\/ $x\colon A$. For any\/ $u,v\colon\Sigma_AP$, the first projection induces an equivalence between their identity types:
    \[
        (u\eq{\Sigma_AP}v)\simeq(\pi_1(u)\eq A\pi_1(v)).
    \]
\end{lem}

\begin{proof}
    Consider
    \begin{align*}
        f\colon(u\eq{\Sigma_AP}v)&\to(\pi_1(u)\eq A\pi_1(v))\\
        p&\mapsto\fun{ap}_{\pi_1}(p).
    \end{align*}
    To construct a backward map
    \[
        g\colon(\pi_1(u)\eq A\pi_1(v))\to(u\eq{\Sigma_AP}v),
    \]
    given a path $q\colon\pi_1(u)\eq A\pi_1(v)$, take a witness
    $\epsilon\colon\fun{isProp}(P(\pi_1(v)))$.
    Then,
    \[
        \gamma\defeq\epsilon(q_*(\pi_2(u)),\pi_2(v))
    \]
    inhabits $q_*(\pi_2(u))\eq{P(\pi_1(v))}\pi_2(v)$. Therefore, we can define the backward map as
    \[
        g(q)\defeq\fun{pair}^=(q,\gamma).
    \]
    To conclude that $f$ and $g$ are quasi-inverses, we must verify that their compositions are homotopic to the respective identity maps:
    \begin{enumerate}[-]
        \item For any $q\colon\pi_1(u)\eq A\pi_1(v)$, it is clear that $f(g(q))\eq{}q$.
        
        \item For any $p\colon u\eq{\Sigma_AP}v$, we can write $p\eq{} \fun{pair}^=(p_1,p_2)$ for some transport path $p_2\colon(p_1)_*(\pi_2(u)) \eq{P(\pi_1(v))} \pi_2(v)$, where $p_1\defeq\fun{ap}_{\pi_1}(p)$.
        
        Evaluating $g(f(p))$ yields $\fun{pair}^=(p_1,\gamma')$, where 
        $\gamma'\defeq\epsilon((p_1)_*(\pi_2(u)),\pi_2(v))$. Because $P(\pi_1(v))$ is a mere proposition, it is also a set, meaning any two parallel paths within it are propositionally equal. In particular, $\gamma'\eq{}p_2$. Applying the function $\fun{ap}$ to this equality yields 
        \[
            \fun{pair}^=(p_1,\gamma')\eq{}\fun{pair}^=(p_1,p_2),
        \]
        which implies $g(f(p))\eq{}p$.
    \end{enumerate}
\end{proof}

\begin{ntn}

    \begin{equation}\label{eq:set-notation}

    \end{equation}

\begin{thm}\fe\label{thm:mere-propositions}
    Mere propositions are closed under the following constructions:\footnote{Function extensionality is used in parts a) and b).}
    \begin{enumerate}[a),font=\upshape]
        \item \textsc{\upshape Dependent Functions:} For any arbitrary type\/ $A$ and any family of mere propositions\/ $P\colon A\to\univ$, the dependent product\/ $\prod_{x\colon A}P(x)$ is a mere proposition.
        
        \item \textsc{\upshape Functions \& Negation:} If\/ $Q$ is a mere proposition, the type\/ $A\to Q$ is a mere proposition for any arbitrary type\/ $A$. In particular, the negation\/ $\neg A$ is always a mere proposition.
        
        \item \textsc{\upshape Products:} If\/ $P$ and\/ $Q$ are mere propositions, then their Cartesian product\/ $P\times Q$ is a mere proposition.
        
        \item \textsc{\upshape Dependent Pairs:} If\/ $A$ is a mere proposition and\/ $P\colon A\to\univ$ is a family of mere propositions, then the total space\/ $\Sigma_AP$ is a mere proposition.
        
        \item \textsc{\upshape Mutually Exclusive Coproducts:} If\/ $P$ and\/ $Q$ are mere propositions, and they are \textsl{mutually exclusive} (i.e., we have a function\/ $P\to\neg Q$), then the coproduct\/ $P+Q$ is a mere proposition.
        
        \item \textsc{\upshape Equivalences \& Identity:} If\/ $P$ and\/ $Q$ are mere propositions, the type of equivalences\/ $P\simeq Q$ and their propositional equality\/ $P\eq{\fun{Prop}_\univ}Q$ are mere propositions.

    \end{enumerate}
\end{thm}

\needspace{2\baselineskip}
\begin{proof}${}$
    \begin{enumerate}[a)]
        \item Let $f,g\colon\prod_{x\colon A}P(x)$. Given $x\colon A$, since $P(x)$ is a mere proposition, we have $f(x)\eq{P(x)}g(x)$. Thus, the conclusion follows by function extensionality.

        \item This is a direct consequence of part a) for the particular case where $P(x)\equiv Q$ for all $x\colon A$. Since $\fun0$ is clearly a mere proposition (witnessed by $\fun{ind}_{\fun0}(\prod_{x\colon\fun0}\,\prod_{y\colon\fun0}x\eq{\fun0}y)$), we deduce that $\neg A\equiv A\to\fun0$ is a mere proposition.

        \item Let $(a,b),(c,d)\colon P\times Q$. Then $a\eq Pc$ and $b\eq Qd$ and so $(a,b)\eq{P\times Q}(c,b)\eq{P\times Q}(c,d)$. The conclusion follows because, for all $x\colon P\times Q$, we have $x\eq{P\times Q}(\pi_1(x),\pi_2(x))$.

        
        \item This is a direct consequence of Lemma~\ref{lem:total-space-with-mere-proposition-fibers}.

        \item Fix $f\colon P\to\neg Q$, and let $\epsilon\colon\fun{isProp}(P)$ and $\delta\colon\fun{isProp}(Q)$. To construct a dependent function of type
        \[
            \fun{isProp}(P+Q)\equiv\prod_{u,v\colon P+Q}u\eq{P+Q}v,
        \]
        we proceed by double induction on $u,v\colon P+Q$.

        For the cross-cases, given $x\colon P$ and $y\colon Q$, define:
        \begin{align*}
            g&\colon(P\times Q)\to\fun0,
            &(x,y)&\mapsto f(x)(y),\\[2mm]
            \fun{rec}&\colon\!\!\prod_{x\colon P}\;\prod_{y\colon Q}
                \fun0\to(\fun{inl}(x)\eq{P+Q}\fun{inr}(y)),
            &(x,y)&\mapsto\fun{rec}_{\fun0}(\fun{inl}(x)\eq{P+Q}\fun{inr}(y)),\\[2mm]
            \zeta&\colon\!\!\prod_{x\colon P}\;\prod_{y\colon Q}
                \fun{inl}(x)\eq{P+Q}\fun{inr}(y),
            &(x,y)&\mapsto\fun{rec}(x,y)(g(x,y)).
        \end{align*}
    
        Take $x,x'\colon P$. We have $\epsilon(x,x')\colon x\eq{P}x'$ and so
        \[
            \epsilon_x\colon\prod_{x'\colon P}
                \fun{inl}(x)\eq{P+Q}\fun{inl}(x'),
                \quad\epsilon_x(x')\defeq\fun{ap}_{\fun{inl}}(\epsilon(x,x')).
        \]
        Let the target family for the inner induction be
        \[
            C_x(v)\defeq\fun{inl}(x)\eq{P+Q}v.
        \]
        Then
        \[
            \epsilon_x\colon\prod_{x'\colon P}C_x(\fun{inl}(x')).
        \]
        Moreover,
        \begin{align*}
            \zeta(x,\dummy)\colon&\prod_{y\colon Q}
                \fun{inl}(x)\eq{P+Q}\fun{inr}(y)\\
                \equiv&\prod_{y\colon Q}C_x(\fun{inr}(y)).
        \end{align*}
        Thus, when $u\equiv\fun{inl}(x)$, induction on $v\colon P+Q$ yields
        \[
            c_{\fun{inl}}(x)\defeq\fun{ind}_{P+Q}
            \bigl(
                C_x,
                \epsilon_x,
                \zeta(x,\dummy)
            \bigr)\colon\prod_{v\colon P+Q} \fun{inl}(x)\eq{P+Q}v.
        \]
        Similarly, given $y\colon Q$, define the family
        \[
            D_y(v)\defeq\fun{inr}(y)\eq{P+Q}v.
        \]
        Then,
        \begin{align*}
            \delta_y\colon\prod_{y'\colon Q}D_y(\fun{inr}(y')),
            \quad\delta_y(y')\defeq\fun{ap}_{\fun{inr}}(\delta(y,y')).
        \end{align*}
        and
        \begin{align*}
            \zeta(\dummy,y)^{-1}\colon&\prod_{x\colon P}
                \fun{inr}(y)\eq{P+Q}\fun{inl}(x)\\
                \equiv&\prod_{x\colon P}D_y(\fun{inl}(x)).
        \end{align*}
        Thus, when $u\equiv\fun{inr}(y)$, induction on $v\colon P+Q$ yields
        \[
            c_{\fun{inr}}(y)\defeq\fun{ind}_{P+Q}
            \bigl(
                D_y,
                \zeta(\dummy,y)^{-1},
                \delta_y
            \bigr)\colon\prod_{v\colon P+Q} \fun{inr}(y)\eq{P+Q}v.
        \]
        Finally, define the family for the outer induction as
        \[
            E(u)\defeq\prod_{v\colon P+Q}u\eq{P+Q}v.
        \]
        By induction on $u\colon P+Q$, we obtain
        \[
            \fun{ind}_{P+Q}
            \bigl(
                E,
                c_{\fun{inl}},
                c_{\fun{inr}}
            \bigr)\colon\prod_{u,v\colon P+Q}u\eq{P+Q}v,
        \]
        which is a witness of $\fun{isProp}(P+Q)$.

        \item By definition,
        \[
            P\simeq Q\equiv\sum_{f\colon P\to Q}\fun{isequiv}(f).
        \]
        By part b), the base type $P\to Q$ is a mere proposition. By part d), to prove that the total space $P\simeq Q$ is a mere proposition, it suffices to show that the fiber $\fun{isequiv}(f)$ is a mere proposition for all $f\colon P\to Q$.
        
        Fix $f\colon P\to Q$. Recall that $\fun{isequiv}(f)$ is the product of two $\Sigma$-types:
        \[
            \Bigl(\sum_{h\colon Q\to P}f\circ h\sim\id_Q\Bigr)
            \times
            \Bigl(\sum_{k\colon Q\to P}k\circ f\sim\id_P\Bigr).
        \]
        By part c), we need only show that each factor is a mere proposition. Consider the first factor. By part b), the base type $Q\to P$ is a mere proposition. Hence, any two elements in this factor have equal first components. Furthermore, the fiber family is given by
        \[
            (f\circ h\sim\id_Q)\equiv\prod_{y\colon Q}f(h(y))\eq{Q}\id_Q(y).
        \]
        Because $Q$ is a mere proposition, Remark~\ref{rem:ids-in-mere-propositions-are-singletons} ensures that the identity type $f(h(y))\eq{Q}\id_Q(y)$ is a mere proposition. Part a) then guarantees that the homotopy is a mere proposition. Since the fibers are mere propositions and the base components are necessarily equal, Lemma~\ref{lem:total-space-with-mere-proposition-fibers} implies that any two elements in this first factor are equal.
        
        A symmetric argument applies to the second factor. Hence, $\fun{isequiv}(f)$ is a mere proposition, which concludes the first claim.

        \bigskip

        For the second claim, let $P\equiv\dep{P_0,f}$ and $Q\equiv\dep{Q_0,g}$ be two types in $\fun{Prop}_\univ\equiv\sum_{A\colon\univ}\fun{isProp}(A)$. By the characterization of paths in $\Sigma$-types, we have an equivalence
        \[
            (P\eq{\fun{Prop}_\univ}Q)\simeq\sum_{h\colon P_0\eq{\univ}Q_0}h_*(f)\eq{\fun{isProp}(Q_0)}g.
        \]
        By the Univalence Axiom, $(P_0\eq{\univ}Q_0)\simeq(P_0\simeq Q_0)$. Since $P_0$ and $Q_0$ are mere propositions, the first claim of this part implies that $P_0\simeq Q_0$ is a mere proposition. Thus, its equivalent type $P_0\eq{\univ}Q_0$ is also a mere proposition.
        
        Furthermore, by Proposition~\ref{prop:isSet-and-isProp-are-mere-propositions}, $\fun{isProp}(Q_0)$ is a mere proposition, which implies by Remark~\ref{rem:ids-in-mere-propositions-are-singletons} that the identity type $h_*(f)\eq{\fun{isProp}(Q_0)}g$ is a mere proposition for any $h$.
        
        Since both the base type and the fibers of this $\Sigma$-type are mere propositions, part d) dictates that the total space is a mere proposition. Because the property of being a mere proposition is preserved under equivalence, we conclude that the identity type $P\eq{\fun{Prop}_\univ}Q$ is a mere proposition.\qedhere
        

    \end{enumerate}
    
\end{proof}

\begin{rem}
    The double induction on $u,v\colon P+Q$ in the proof of part~(e) is geometrically analogous to integrating over the two-dimensional space
    \[
        (P+Q)\times(P+Q).
    \]
    The induction principle naturally partitions this space into four distinct \textit{quadrants}:
    \[
        P\times P,\quad P\times Q,\quad Q\times P,\quad \text{and}\quad Q\times Q.
    \]
    The proof effectively \textit{integrated} over the cross-quadrants $P\times Q$ and $Q\times P$ using $\zeta$ and $\zeta^{-1}$, and over the squares $P\times P$ and $Q\times Q$ using the internal mere proposition witnesses $\epsilon$ and $\delta$.
\end{rem}

\begin{cor}\fe
    Let\/ $f,g\colon A\to P$ be two functions. If\/ $P$ is a mere proposition, then\/ $f\sim g$ is a mere proposition as well.
\end{cor}

\begin{proof}
    This is actually contained in the proof of the last part of the theorem. Take two inhabitants
    \[
        \eta,\vartheta\colon f\sim g\equiv\prod_{x\colon A}f(x)\eq Pg(x).
    \]
    By Remark~\ref{rem:ids-in-mere-propositions-are-singletons}, we deduce that $\eta(x)\eq{f(x)\eq Pg(x)}\vartheta(x)$ for all\/ $x\colon A$. Then, function extensionality yields $\eta\eq{f\sim g}\vartheta$. The conclusion follows.
\end{proof}

    Let\/ $A$ and\/ $B$ be sets, and let\/ $f\colon A\to B$ be a function. Then the type\/ $\fun{isequiv}(f)$ is a set. If\/ $A$ and\/ $B$ are mere propositions, then $\fun{isequiv}(f)$ is a mere proposition.
\end{cor}

\begin{proof}
    To prove that $\fun{isequiv}(f)$ is a set, we have to prove that the type
    \[
        \Bigl(
            \sum_{g\colon B\to A}f\circ g\sim\id_B
        \Bigr)
        \times
        \Bigl(
            \sum_{h\colon B\to A}h\circ f\sim\id_A
        \Bigr)
    \]
    is a set.

    By the symmetry between $A$ and $B$, and the fact that the Cartesian product of sets is a set given in part~d) of Theorem~\ref{thm:sets}, it suffices to show that the first $\Sigma$-type is a set.

    By part~f) of the same theorem, the type $B\to A$ is a set. Therefore, to reach the conclusion, it suffices to show that we can apply part~e), i.e., that $f\circ g\sim\id_B$ is a set for every $g\colon B\to A$.

    To see this, take $\eta_1,\eta_2\colon f\circ g\sim\id_B$. Given $y\colon B$, we have
    \[
        \eta_1(y),\eta_2(y)\colon f(g(y))\eq By.
    \]
    Since $B$ is a set, we obtain $\eta_1(y)\eq{}\eta_2(y)$ in $f(g(y))\eq By$. Therefore, function extensionality produces $\eta_1\eq{}\eta_2$ in $f\circ g\sim\id_B$. Thus, $f\circ g\sim\id_B$ is a mere proposition, hence a set. The conclusion follows.

    In the case where $A$ and $B$ are mere propositions, since they are also sets, the fact that $f\circ g\sim\id_B$ is a mere proposition allows us to invoke part~d) of Theorem~\ref{thm:mere-propositions}, because $B\to A$ is a mere proposition by part~b) of the same theorem. By symmetry, the second $\Sigma$-type is also a mere proposition, and since the product of mere propositions is a mere proposition (part~c), the type $\fun{isequiv}(f)$ is a mere proposition.
\end{proof}

\begin{cor}\label{cor:set-simeq-set-is-set}
    Let\/ $A$ and\/ $B$ be sets. Then the type\/ $A\simeq B$ is a set.
\end{cor}

\begin{proof}
    By definition, the type of equivalences is given by the $\Sigma$-type:
    \[
        A\simeq B\defeq\sum_{f\colon A\to B}\fun{isequiv}(f).
    \]
    To show that a $\Sigma$-type is a set, by part~e) of Theorem~\ref{thm:sets}, it suffices to show that $A\to B$ is a set and that each $\fun{isequiv}(f)$ is a set.
    
    The first requirement is covered by part~f) of said theorem. The second by Corollary~\ref{cor:isequiv(f)-is-set}.
\end{proof}

\begin{prop}\label{prop:qinv-is-a-set}
    Let\/ $A$ and\/ $B$ be sets, and let\/ $f\colon A\to B$ be a function. Then the type\/ $\fun{qinv}(f)$ is a set.
\end{prop}

\begin{proof}
    By definition,
    \[
        \fun{qinv}(f) \equiv \sum_{g\colon B\to A} (f\circ g \sim \id_B) \times (g\circ f \sim \id_A).
    \]
    By part e) of Theorem~\ref{thm:sets}, to show that $\fun{qinv}(f)$ is a set, it suffices to show that the base type $B\to A$ is a set, and that for each $g\colon B\to A$, the fiber $(f\circ g \sim \id_B) \times (g\circ f \sim \id_A)$ is a set.

    The former type is a set by part f) of the same theorem .
    
    For the latter type, since the Cartesian product of sets is a set (same theorem part d), we need to show that both factors are set. Consider the first one:
    \[
        f\circ g\sim\id_B
            \equiv\prod_{y\colon B}f(g(y))\eq By.
    \]
    Because $B$ is a set, given $y\colon B$, the identity type $f(g(y)) \eq{B} y$ is a mere proposition by Remark~\ref{rem:ids-in-a-set-are-mere-propositions}. By part a) of Theorem~\ref{thm:mere-propositions}, a dependent product of mere propositions is a mere proposition, hence, a set. Thus, $f\circ g \sim \id_B$ is a set.

    By symmetry, the second factor
    \[
        g\circ f \sim \id_A
    \]
    is a set as well.
\end{proof}


\section{Propositional Truncation}\label{propositional-truncation}
We can define a new type former that converts any type $A\colon\univ$ into $\lVert A\rVert\colon\univ$, whose canonical terms are of the form $\lvert a\rvert$ for $a\colon A$. The distinguishing characteristic of $\lVert A\rVert$ is that it is a mere proposition; that is, for any $x,y\colon\lVert A\rVert$, we have $x\eq{\lVert A\rVert}y$. More precisely, there exists a dependent function
\[
    \fun{squash}\colon\prod_{x,y\colon\lVert A\rVert}x\eq{\lVert A\rVert}y.
\]
The induction principle, which can be applied to mere proposition families, is given by:
\begin{equation}\label{eq:ind-truncation}
    \fun{ind}_{\lVert A\rVert}\colon
        \prod_{C\colon\fun{Pred}(\lVert A\rVert)}
        \Bigl(\prod_{a\colon A} C(\lvert a\rvert)\Bigr)
        \to
        \prod_{z\colon\lVert A\rVert} C(z),
\end{equation}
satisfying $\fun{ind}_{\lVert A\rVert}(C,f,\lvert a\rvert)\defeq f(a)$ for $a\colon A$.

In the case of non-dependent mere proposition families, we have the recursion principle
\begin{align}\label{eq:rec-truncation}
    \fun{rec}_{\lVert A\rVert}&\colon
        \prod_{B\colon\fun{Prop}_\univ}(A\to B)\to(\lVert A\rVert\to B)\\
    \fun{rec}_{\lVert A\rVert}(B,f)&\defeq\fun{ind}_{\lVert A\rVert}
        (a\mapsto B,f).\nonumber
\end{align}

\begin{rem}
    One might be tempted to simplify the defining requirement $x\eq{\lVert A\rVert}y$ for all $x,y\colon\lVert A\rVert$, to merely $\lvert a\rvert\eq{\lVert A\rVert}\lvert b\rvert$ for all $a,b\colon A$, and then attempt to use induction on $\lVert A\rVert$ to deduce the former. However, to apply the induction principle, we must first show that the target family $C(x,y)\defeq x\eq{\lVert A\rVert}y$ is a predicate. This requires already knowing that the identity type $x\eq{\lVert A\rVert}y$ is a mere proposition, which is equivalent to knowing that $\lVert A\rVert$ is a set. Because we do not yet know this, we are trapped in a circular reasoning.

    Imposing paths only on canonical elements fails to define a set. For instance, consider the type $\fun1+\fun2$. This type has exactly three discrete elements, say $x,y,z$, derived from injecting $\fun\star$ on the left, and $0_{\fun2}$ and $1_{\fun2}$ on the right. If we explicitly adjoin paths $p_{xy}\colon x\eq{}y$, $p_{yz}\colon y\eq{}z$, and $p_{xz}\colon x\eq{}z$, we obtain a hollow triangle. In this space, there are two distinct ways to travel from $x$ to $z$: the path $p_{xz}$, and the concatenation $p_{xy}\ct p_{yz}$. Because we have not explicitly stated that $p_{xz}\eq{}p_{xy}\ct p_{yz}$, these two paths are not equal. Consequently, the identity type $x\eq{}z$ is not a mere proposition, and the resulting type is not a set.
\end{rem}

\begin{lem}\label{lem:truncation-functor}
    Truncation is a functorial operation. That is,
    \begin{enumerate}[a),font=\upshape]
        \item Every map $f\colon A\to B$ produces a map $\lVert f\rVert\colon\lVert A\rVert\to\lVert B\rVert$.
        \item For any type $A$, we have $\lVert\id_A\rVert\sim\id_{\lVert A\rVert}$.
        \item For any $f\colon A\to B$ and $g\colon B\to C$, we have $\lVert g\circ f\rVert\sim\lVert g\rVert\circ\lVert f\rVert$.
    \end{enumerate}
    Consequently, if $A\simeq B$, then $\lVert A\rVert\simeq\lVert B\rVert$.
\end{lem}

\begin{proof}
    Given $f\colon A\to B$, we define the map $\lVert f\rVert$ using the recursion principle $\fun{rec}_{\lVert A\rVert}$. Since the target $\lVert B\rVert$ is a mere proposition, it suffices to define its action on the canonical constructor:
    \[
        \lVert f\rVert\defeq\fun{rec}_{\lVert A\rVert}(\lVert B\rVert,a\mapsto\lvert f(a)\rvert).
    \]
    
    For b), let $x\colon\lVert A\rVert$. We must show that $\lVert\id_A\rVert(x)\eq{\lVert A\rVert}x$. By part~d) of Theorem~\ref{thm:mere-propositions} applied to the constant family $P(x)\defeq A$, the identity type of a mere proposition is itself a mere proposition. Therefore, we can proceed by induction on $x$. It suffices to consider the case $x\equiv\lvert a\rvert$ for some $a\colon A$. In this case, $\lVert\id_A\rVert(\lvert a\rvert)$ reduces definitionally to $\lvert\id_A(a)\rvert\equiv\lvert a\rvert$, so the equality holds by reflexivity.

    For c), by induction on $x\colon\lVert A\rVert$, we only need to check the base case $x\equiv\lvert a\rvert$. Since both $\lVert g\circ f\rVert(\lvert a\rvert)$ and $\lVert g\rVert(\lVert f\rVert(\lvert a\rvert))$ compute definitionally to $\lvert g(f(a))\rvert$, the required path is simply reflexivity.

    Finally, suppose that $f\colon A\to B$ induces an equivalence and let $f^{-1}\colon B\to A$ be a quasi-inverse. By a), we obtain $\lVert f\rVert\colon\lVert A\rVert\to\lVert B\rVert$ and $\lVert f^{-1}\rVert\colon\lVert B\rVert\to\lVert A\rVert$. Since the target spaces are mere propositions, invoking Theorem~\ref{thm:mere-propositions} as before, we deduce that both compositions $\lVert f^{-1}\rVert\circ\lVert f\rVert$ and $\lVert f\rVert\circ\lVert f^{-1}\rVert$ are trivially homotopic to $\id_{\lVert A\rVert}$ and $\id_{\lVert B\rVert}$, respectively. Thus, $\lVert f\rVert$ is an equivalence.
    %
    
\end{proof}

\begin{lem}\label{lem:mere-propositions-are-truncated}
    If $A$ is a mere proposition, then $A\eq{\univ}\lVert A\rVert$.
\end{lem}

\begin{proof}
    Since $A$ is a mere proposition, recursion yields a function
    \[
        \fun{rec}_{\lVert A\rVert}(A,\id_A)\colon\lVert A\rVert\to A.
    \]
    Conversely, the canonical constructor gives:
    \[
        \lvert{-}\rvert\colon A\to\lVert A\rVert,\quad a\mapsto\lvert a\rvert.
    \]
    The conclusion follows from Lemma~\ref{lem:mere-proposition-equivalence} and the univalence axiom.
\end{proof}

\begin{lem}\label{lem:trucated-eq-to-set-is-set}
    Let\/ $S$ be a set, and let\/ $A\colon\univ$ be a type. If\/ $\lVert A\eq\univ S\rVert$ is inhabited, then\/ $A$ is a set.
\end{lem}

\begin{proof}
    We must construct an inhabitant of the type $\fun{isSet}(A)$. 
    
    By recursion for propositional truncation, to define a function $\lVert X\rVert\to P$, where $P$ is a mere proposition, it suffices to construct a map from $X$ to $P$. By Proposition~\ref{prop:isSet-and-isProp-are-mere-propositions}, we know that $\fun{isSet}(C)$ is a mere proposition for every $C\colon\univ$. Hence, we can apply recursion to the case $X\defeq A\eq\univ S$ and $P\defeq\fun{isSet}(A)$.
    
    To provide the function $f\colon(A\eq\univ S)\to\fun{isSet}(A)$, fix a witness $\epsilon\colon\fun{isSet}(S)$ and define:
    \[
        f(q)\defeq\fun{transport}^{\fun{isSet}}(q^{-1},\epsilon).
    \]
    Thus, if $h\colon\fun{isProp}(\fun{isSet}(A))$, we obtain the function
    \[
        \fun{rec}_{\lVert A\eq\univ S\rVert}\bigl(\fun{isSet}(A),h,f\bigr)
            \colon\lVert A\eq\univ S\rVert\to\fun{isSet}(A).
    \]
    The conclusion follows.
\end{proof}

\begin{ntn}\label{ntn:logic}
    Truncation allows us to interpret traditional logical notation via the following type-theoretic operations. Although we will not adopt this notation extensively ---reserving it primarily for contexts strictly focused on sets and propositions--- it establishes a precise dictionary between classical logic and type theory. Here, $P$ and\/ $Q$ denote mere propositions (or mere proposition families), and\/ $A$ is an arbitrary type. 
    
    For operations that inherently preserve mere propositions, no truncation is necessary:
    \begin{align*}
        \top\defeq\fun1&\text{\upshape\ \& }\bot\defeq\fun0
            &\neg P&\defeq P\to\fun0\\[1.9mm]
        P\land Q&\defeq P\times Q\qquad
            &\forall(x\colon A).\,P(x)&\defeq\prod_{x\colon A}P(x)\\[1.9mm]
        P\Rightarrow Q&\defeq P\to Q
            &P\Leftrightarrow Q&\defeq P\eq\univ Q
    \end{align*}
    For operations that can construct spaces with multiple distinct points, we must apply propositional truncation to strictly yield a mere proposition:
    \begin{align*}
        P\lor Q&\defeq\lVert P+Q\rVert\\
        \exists(x\colon A).\,P(x)&\defeq\Bigl\lVert\sum_{x\colon A}P(x)\Bigr\rVert
    \end{align*}
\end{ntn}

\section{Contractibility}

\begin{defn}
    A type $A$ is \textsl{contractible} (a.k.a.~a \textsl{singleton}) if there is an element $a\colon A$, called the \textsl{center of contraction}, such that every element in $A$ is equal to $a$. We define the type of such witnesses as
    \[
        \fun{isContr}(A)\defeq\sum_{a\colon A}\;\prod_{x\colon A}a\eq{A}x.
    \]
\end{defn}

\begin{lem}\label{lem:contractible-equivalences}
    For a type\/ $A$, the following are logically equivalent.
    \begin{enumerate}[a), font=\upshape]
        \item $A$ is contractible.
        \item $A$ is a mere proposition, and there is a point\/ $a\colon A$.
        \item $A$ is equivalent to\/ $\fun1$.
    \end{enumerate}
\end{lem}

\begin{proof}${}$
    \begin{enumerate}[a)]
        \item $\Rightarrow$~b) The hypothesis provides a witness $h\colon\fun{isContr}(A)$. Composition with first projection yields a witness $\pi_1(h))\colon A$. Hence, $A$ is inhabited.

        \item $\Rightarrow$~c) See Lemma~\ref{lem:mere-prop-and-1}.

        \item $\Rightarrow$~a) The canonical element $\fun\star\colon\fun1$ is a center of contraction as shown by the $\fun{uniq}_{\fun1}$ function [cf.~\nref{lpar:Pi-types-over-product}].
    \end{enumerate}
    
\end{proof}

\begin{lem}\label{lem:transport-prod-id}
    Let\/ $A$ be a type, $a,b\colon A$, and define the type family
    \[
        P\colon A\to\univ,\quad P(z)\defeq\prod_{x\colon A}z\eq{A}x.
    \]
    Given\/ $p\colon P(a)$ and\/ $\gamma\colon a\eq{A}b$, we have for any\/ $x\colon A$
    \[
        \fun{transport}^P(\gamma,p)(x)\eq{b\eq{A}x}\gamma^{-1}\ct p(x).
    \]
\end{lem}

\begin{proof}
    The transport on the \lhs is an element of $P(b)$. Its evaluation at $x$ yields a path in $b\eq{A}x$. On the other hand, the \rhs concatenates a path from $b$ to $a$ with a path from $a$ to $x$, yielding a path from $b$ to $x$. Thus, both sides inhabit the type $b\eq{A}x$.

    By based path induction on $\gamma\colon a\eq{A}b$ it suffices to assume that $b\equiv a$ and $\gamma\equiv\fun{refl}_a$. Under this assumption, the \lhs reduces trivially:
    \[
        \fun{transport}^P(\fun{refl}_a,p)(x)\equiv p(x).
    \]
    For the \rhs, the inverse of reflexivity is definitionally reflexivity, and since reflexivity acts trivially on the left, the equality with $p(x)$ follows.
\end{proof}

\begin{thm}\fe\label{thm:isContr-is-a-mere-proposition}
    For any type\/ $A$, the type\/ $\fun{isContr}(A)$ is a mere proposition.
\end{thm}

\begin{proof}
    Let $c,d\colon\fun{isContr}(A)$. Set
    \[
        a\defeq\pi_1(c),\quad p\defeq\pi_2(c)
        \quad\text{and}\quad
        b\defeq\pi_1(d),\quad q\defeq\pi_2(d).
    \]
    Then,
    \[
        p\colon\prod_{x\colon A}a\eq{A}x
        \quad\text{and}\quad
        q\colon\prod_{x\colon A}b\eq{A}x.
    \]
    Evaluating $p$ at $b$ yields a path $p(b)\colon a\eq{A}b$, which provides the equality of the first components. 

    Let
    \[
        P\colon A\to\univ,
        \quad P(z)\defeq\prod_{x\colon A}z\eq{A}x.
    \]
    To prove $c\eq{\fun{isContr}(A)}d$ we need to show that
    \[
        \fun{transport}^P(p(b),p)\eq{P(b)}q.
    \]
    By Lemma~\ref{lem:transport-prod-id} the transport in the \lhs satisfies
    \[
        \fun{transport}^P(p(b),p)(x)\eq{b\eq Ax}p(b)^{-1}\ct p(x),
    \]
    for $x\colon A$.
        
    Thus, by function extensionality, the goal reduces to showing
    \[
        p(b)^{-1}\ct p(x)\eq{b\eq{A}x}q(x).
    \]
    Since $c$ witnesses that $A$ is contractible, $A$ is a mere proposition. By Remark~\ref{rem:ids-in-mere-propositions-are-singletons}, $b\eq{A}x$ is a mere proposition. Since both $p(b)^{-1}\ct p(x)$ and $q(x)$ are elements of $b\eq{A}x$, they are necessarily equal. This completes the proof.
\end{proof}

\begin{cor}\label{cor:A-contractible-implies-isContr(A)-contractible}
    If\/ $A$ is contractible, $\fun{isContr}(A)$ is also contractible.
\end{cor}

\begin{proof}
    By the theorem, $\fun{isContr}(A)$ is a mere proposition. It is also inhabited because $A$ is contractible.  The conclusion follows from part~b) of Lemma~\ref{lem:contractible-equivalences}.
\end{proof}

\begin{rem}\label{rem:choice-free-center-construction}
    Let $A\colon\univ$ be a type and $P\colon A\to\univ$ a type family. Claiming that \textit{the type $P(x)$ is contractible for every $x\colon A$} means that we are given a dependent function
    \[
        h\colon\prod_{x\colon A}\fun{isContr}(P(x)),
    \]
    i.e.,
    \[
        h\colon\prod_{x\colon A}\;
            \sum_{c\colon P(x)}\;\prod_{z\colon P(x)}c\eq{P(x)}z.
    \]
    In particular, given $x\colon A$, the term $c_x\defeq\pi_1(h(x))$ is a center of contraction for $P(x)$. In other words, picking the center of contraction for every contractible fiber of $P$ is a constructive, choice-free operation.
\end{rem}

\begin{thm}\fe\label{thm:contractible-types}
    Contractible types are closed under the following type-theoretic constructions:\footnote{Function extensionality is needed in parts a), b), f), and g).}
    \begin{enumerate}[a),font=\upshape]
        \item \textsc{\upshape Dependent Functions:} For any arbitrary type\/ $A$ and any family of contractible types\/ $P\colon A\to\univ$, the dependent product\/ $\prod_{x\colon A}P(x)$ is contractible. 
        
        \item \textsc{\upshape Functions:} If\/ $Q$ is contractible, the type\/ $A\to Q$ is contractible for any arbitrary type\/ $A$.
        
        \item \textsc{\upshape Products:} If\/ $P$ and\/ $Q$ are contractible, then their Cartesian product\/ $P\times Q$ is contractible.
        
        \item \textsc{\upshape Dependent Pairs:} If\/ $A$ is contractible and\/ $P\colon A\to\univ$ is a family of contractible types, then the total space\/ $\Sigma_AP$ is contractible.
        
        \item \textsc{\upshape Identity Types:} If\/ $A$ is contractible, then for any\/ $x,y\colon A$, the identity type\/ $x\eq A y$ is contractible.
        
        \item \textsc{\upshape Equivalences \& Identity:} If\/ $P$ and\/ $Q$ are contractible, the type of equivalences\/ $P\simeq Q$ and their propositional equality\/ $P\eq{\univ}Q$ are contractible.

        \item \textsc{\upshape Homotopies:} If $f,g\colon\prod_{x\colon A}B(x)$, where $B(x)$ is contractible for $x\colon A$, then the type $f\sim g$ is contractible.

    \end{enumerate}
\end{thm}

\begin{proof}
    Since Theorem~\ref{thm:isContr-is-a-mere-proposition} establishes that every contractible type is a mere proposition, Theorem~\ref{thm:mere-propositions} guarantees that the constructed types are mere propositions. Therefore, by Lemma~\ref{lem:contractible-equivalences}, it suffices to show that each of them is inhabited. In cases a) to f), constructing a witness from the centers of contraction of the constituent contractible types is straightforward.

    To prove $g)$, observe that by part e) the identity type $f(x)\eq{B(x)}g(x)$ is contractible, which implies that $f\sim g$ is contractible by part~a).
\end{proof}

\begin{defn}\label{defn:type-retract}
    A \textsl{retraction} (a.k.a.~\textsl{split surjection}) is a function $r\colon A\to B$ such that there exists a function $s\colon B\to A$, called its \textsl{section}, and a homotopy $\eta\colon r\circ s\sim\id_B$. When a retraction exists, we say that $B$ is a \textsl{retract} of~$A$.
\end{defn}

\begin{lem}\label{lem:contractible-retracts}
    Every retract of a contractible type is contractible.
\end{lem}

\begin{proof}
    Let\/ $A$ be contractible and\/ $r\colon A\to B$ a retraction. If\/ $a\colon A$ denotes its center of contraction, we claim that\/ $b\defeq r(a)$ is a center of contraction for\/ $B$. This means that we must construct a witness of
    \[
        \prod_{y\colon B}b\eq By.
    \]
    Let\/ $s\colon B\to A$ be a section of\/ $r$, and\/ $\eta\colon r\circ s\sim\id_B$ the corresponding homotopy.

    The hypothesis implies that for every\/ $y\colon B$, the identity type\/ $a\eq As(y)$ is inhabited. Applying\/ $\fun{ap}_r$ yields a witness of\/ $b\eq Br(s(y))$, whose concatenation with\/ $\eta(y)$ constructs a witness of\/ $b\eq By$. The conclusion follows.
    %
    
\end{proof}

\begin{lem}\label{lem:Sigma-identity-is-contractible}
    Let\/ $A$ be a type. Given\/ $a\colon A$, the type\/ $\sum_{x\colon A}a\eq Ax$ is contractible with center of contraction\/ $\dep{a,\fun{refl}_a}$.
\end{lem}

\begin{proof}
    Let $\Sigma$ denote this $\Sigma$-type. Given $\dep{x,p}\colon\Sigma$, we have $p\colon a\eq Ax$. Since
    \[
        \fun{transport}^{\zeta\mapsto(a\eq A\zeta)}(p,\fun{refl}_a)
            \eq{}\fun{refl}_a\ct p\equiv p,
    \]
    by part a) of Proposition~\ref{prop:transport-decomposition-for-paths}, the identity type $\dep{a,\fun{refl}_a}\eq\Sigma\dep{x,p}$ is inhabited.
\end{proof}

\begin{rem}\label{rem:contractible-center-to-center-path}
    Suppose that $A\colon\univ$ is a contractible type. If $c$ is a center of contraction for $A$, then for every $x\colon A$ we have a path $p_x\colon c\eq Ax$. Even though we cannot assume that $p_c\equiv\fun{refl}_c$, we do have
    \[
        p_c\eq{c\eq Ac}\fun{refl}_c,
    \]
    because every contractible type is a mere proposition, and every mere proposition is a set.
\end{rem}

\begin{thm}\fe\label{thm:contractible-data-ignored}
    The principle that contractible data can be freely ignored up to equivalence manifests in the following equivalences. Let\/ $A$ and\/ $B$ be types, and let\/ $P\colon A\to\univ$ be a type family.\footnote{Function extensionality is needed in parts c), d), f), and g).}
    \begin{enumerate}[a), font=\upshape]
        \item \textsc{\upshape Dependent Pairs (Right):} If for every\/ $x\colon A$, the type\/ $P(x)$ is contractible, then\/ $\sum_{x\colon A}P(x)\simeq A$.
        
        \item \textsc{\upshape Dependent Pairs (Left):} If\/ $A$ is contractible with center\/ $c\colon A$, then\/ $\sum_{x\colon A}P(x)\simeq P(c)$.
        
        \item \textsc{\upshape Dependent Functions (Left):} If\/ $A$ is contractible with center\/ $c\colon A$, then\/ $\prod_{x\colon A}P(x)\simeq P(c)$.
        
        \item \textsc{\upshape Dependent Functions (Right):} If for every\/ $x\colon A$, the type\/ $P(x)$ is contractible, then\/ $\prod_{x\colon A}P(x)$ is contractible, hence\/ $\prod_{x\colon A}P(x)\simeq\fun1$.
        
        \item \textsc{\upshape Cartesian Products:} If\/ $B$ is contractible, then\/ $A\times B\simeq A$. By symmetry, if\/ $A$ is contractible, then\/ $A\times B\simeq B$.
        
        \item \textsc{\upshape Functions (Domain):} If\/ $A$ is contractible, then\/ $(A\to B)\simeq B$.
        
        \item \textsc{\upshape Functions (Codomain):} If\/ $B$ is contractible, then\/ $A\to B$ is contractible, hence\/ $(A\to B)\simeq\fun1$.
    \end{enumerate}
\end{thm}

\begin{proof}${}$
    \begin{enumerate}[a)]
        \item Consider the function
        \begin{align*}
            \iota\colon A&\to\sum_{x\colon A}P(x)\\
            x&\mapsto\dep{x,c_x},
        \end{align*}
        where $c_x$ is the center of contraction for $P(x)$ [cf.~Remark~\ref{rem:choice-free-center-construction}]. Then $\pi_1\circ\iota\equiv\id_A$, and
        \[
            \iota(\pi_1(\dep{x,u}))\equiv\dep{x,c_x}.
        \]
        By equation \eqref{eq:Sigma-trivial-equality} of Remark~\ref{rem:section-type}, to verify that $\dep{x,c_x}\eq{\Sigma_AP}\dep{x,u}$, we must construct a witness of $c_x\eq{P(x)}u$. But this is trivial because the identity type $c_x\eq{P(x)}u$ is inhabited by hypothesis.

        \item Let
        \[
            h\colon\fun{isContr}(A)\equiv\sum_{c\colon A}\;\prod_{x\colon A}c\eq Ax.
        \]
        Then $c\defeq\pi_1(h)$ is a center for $A$, and given $x\colon A$, $p_x\defeq\pi_2(h)(x)$ satisfies $p_x\colon c\eq Ax$. Thus, we can introduce
        \begin{align*}
            f\colon\sum_{x\colon A}P(x)&\to P(c)\\
            \dep{x,u}&\mapsto\fun{transport}^P(p_x^{-1},u),
        \intertext{which is well-defined because $p_x^{-1}\colon x\eq Ac$. On the other hand, we have}
            \iota\colon P(c)&\to\sum_{x\colon A}P(x)\\
            u&\mapsto\dep{c,u}.
        \end{align*}
        Then,
        \begin{align*}
            f(\iota(u))&\equiv f(\dep{c,u})\\
            &\equiv\fun{transport}^P(p_c^{-1},u)\\
            &\eq{P(c)}\fun{transport}^P(\fun{refl}_c^{-1},u)\\
            &\equiv u,
        \end{align*}
        where the equality follows from Remark~\ref{rem:contractible-center-to-center-path} and $\fun{ap}_{\fun{transport}^P(\dummy,u)}$.
    
        Moreover,
        \[
            \iota(f(\dep{x,u}))
            \equiv
            \dep{c,\fun{transport}^P(p_x^{-1},u)}.
        \]
        Since $p_x\colon c\eq Ax$, to prove that $\iota\circ f(\dep{x,u})\eq{\Sigma_AP}(\dep{x,u})$, we need to verify that
        \[
            \fun{transport}^P\bigl(p_x,\fun{transport}^P(p_x^{-1},u)\bigr)\eq{P(x)}u,
        \]
        which is witnessed by $\fun{transportinv}_r(p_x,u)$ [cf.~Lemma~\ref{lem:transportinv}].

        \item Let $c\colon A$ be a center of contraction. Evaluation at $c$ gives us a map
        \begin{align*}
            \ev c\colon\prod_{x\colon A}P(x)&\to P(c)\\
            f&\mapsto f(c).
        \intertext{For the backward function, define}
            \gamma\colon P(c)&\to\prod_{x\colon A}P(x)\\
            u&\mapsto\lambda x.\,\fun{transport}^P(p_x,u),
        \end{align*}
        where $p_x\colon c\eq Ax$ is constructed as in part~b). These functions satisfy:
        \begin{align*}
            \ev c(\gamma(u))
                &\equiv\fun{transport}^P(p_c,u)\\
                &\eq{P(c)}\fun{transport}^P(\fun{refl}_c,u)\\
                &\equiv u,
        \intertext{where the equality follows from Remark~\ref{rem:contractible-center-to-center-path} and $\fun{ap}_{\fun{transport}^P(\dummy,u)}$. In addition, for $x\colon A$,}
            \gamma(f(c))(x)
                &\equiv\fun{transport}^P(p_x,f(c))\\
                &\eq{P(x)}f(x)
                    &&\text{; by }\fun{apd}_f(p_x),
        \end{align*}
        where the last equality follows from Lemma~\ref{lem:dependent-map}. By function extensionality, $\gamma(f(c))\eq{}f$ in $\displaystyle\prod_{x\colon A}P(x)$.
        
        \item For $x\colon A$, let $(f_x,\eta_x,\vartheta_x)\colon P(x)\simeq\fun1$ be an equivalence with quasi-inverse $f_x^{-1}$. Define
        \begin{align*}
            \gamma\colon\prod_{x\colon A}P(x)&\to\fun1\\
            u&\mapsto\fun\star,
        \intertext{and}
            \epsilon\colon\fun1&\to\prod_{x\colon A}P(x)\\
            z&\mapsto\lambda x.\,f_x^{-1}(z).
        \end{align*}
        Then,
        \[
            \gamma(\epsilon(\fun\star))\equiv\fun\star.
        \]
        Since $P(x)$ is a mere proposition, there is a path $p_x\colon f_x^{-1}(\fun\star)\eq{P(x)}u(x)$. Hence,
        \begin{align*}
            \epsilon(\gamma(u))(x)
                &\equiv f_x^{-1}(\fun\star)\\
                &\eq{P(x)}u(x)
                    &&\text{; by }p_x.
        \end{align*}
        Thus, by function extensionality,
        \[
            \epsilon(\gamma(u))\eq{\prod_{x\colon A}P(x)}u.
        \]

        \item Suppose that $B$ is contractible. Define
        \begin{align*}
            \iota\colon A&\to A\times B\\
            a&\mapsto(a,c),
        \end{align*}
        where $c$ is a center of contraction for $B$. Then, $\iota$ is a quasi-inverse of $\pi_1$, because $\pi_1(\iota(a))\equiv\pi_1(a,c)\equiv a$, and
        \begin{align*}
            \iota(\pi_1(a,b))&\equiv(a,c)\\
                &\eq{A\times B}(a,b)
                    &&\text{; by }\fun{ap}_{y\mapsto(a,y)}(p_b),
        \end{align*}
        where $p_b\colon c\eq Bb$.

        \item This is a particular case of part~c) for the constant family $B$.

        \item This is a particular case of part~d) for the constant family $B$.
    \end{enumerate}
\end{proof}

\begin{lem}
    A type\/ $A$ is a mere proposition if, and only if, for all\/ $x,y\colon A$, the type\/ $x\eq Ay$ is contractible.
\end{lem}

\begin{proof}
    If $A$ is a mere proposition, by Lemma~\ref{lem:mere-propositions-are-sets}, it is also a set. Therefore, by definition, $x\eq Ay$ is a mere proposition. Since $x$ (as well as $y$) is a witness of $A$, it follows from Lemma~\ref{lem:contractible-equivalences} that $A$ is contractible. Hence, it has a center, say $c$, and we can take $p\colon c\eq Ax$ and $q\colon c\eq Ay$ to construct $p^{-1}\ct q\colon x\eq Ay$.

    Conversely, if $x\eq Ay$ is contractible for all $x,y\colon A$, then $A$ is a mere proposition by definition.
\end{proof}

\begin{defns}\label{defn:fiber-over-y}
    Let $f\colon A\to B$ be a function.
    \begin{enumerate}[-]
        \item The \textsl{fiber of $f$} is the type family
        \begin{align*}
            \fun{fib}_f&\colon B\to\univ\\
            \fun{fib}_f(y)&\defeq\sum_{x\colon A}f(x)\eq By.
        \end{align*}
            
        \item The function $f$ is \textsl{contractible} if, for every $y\colon B$, the fiber $\fun{fib}_f(y)$ of $f$ over $y$ is contractible. In other words, the type
        \begin{equation}\label{eq:isContr-map}
            \fun{isContrMap}(f)
                \defeq\prod_{y\colon B}\fun{isContr}(\fun{fib}_f(y))
        \end{equation}
        is inhabited.
    \end{enumerate}
\end{defns}

\begin{lem}
    For any function\/ $f\colon A\to B$, the type\/ $\fun{isContrMap}(f)$ is a mere proposition.
\end{lem}

\begin{proof}
    By definition, the type of contractible maps is the dependent type given in \eqref{eq:isContr-map}.

    By Theorem~\ref{thm:isContr-is-a-mere-proposition}, the type $\fun{isContr}(\fun{fib}_f(y))$ is a mere proposition for every $y\colon B$. Then, the conclusion follows from part d) of Theorem~\ref{thm:mere-propositions}.
    
\end{proof}


\section{The Axiom of Choice}

\begin{rem}
    A consequence of part~d) of Theorem~\ref{thm:mere-propositions} applies to the universes of sets and mere propositions, $\fun{Set}_\univ$ and $\fun{Prop}_\univ$. By definition, elements of these universes are dependent pairs consisting of a base type and a witness to its particular characteristic. 
    
    Because the types $\fun{isSet}(A)$ and $\fun{isProp}(P)$ are themselves mere propositions, part~d) guarantees that the second coordinates of these pairs add no new information. In the language of type theory, being a set or a mere proposition is a \textit{property} of a type, rather than an additional \textit{structure} placed upon it. 

    Consequently, the first projections from $\fun{Set}_\univ$ and $\fun{Prop}_\univ$ into $\univ$ are embeddings. This rigorously justifies a standard, ubiquitous abuse of notation: it is perfectly valid to systematically identify pairs $A\colon\fun{Set}_\univ$ and $P\colon\fun{Prop}_\univ$ with their first coordinates, safely treating them as plain types, e.g., writing $x\colon A$ instead of $x\colon\pi_1(A)$.

    In particular, a type family $A\colon X\to\univ$ such that $\fun{isSet}(A(x))$ is inhabited for every $x\colon X$, will often be denoted as a family $A\colon X\to\fun{Set}_\univ$. Similarly, the notation $P\colon X\to\fun{Prop}_\univ$ refers to a mere proposition family, i.e., a family such that $P(x)$ is a mere proposition for all $x\colon X$.
\end{rem}

\separator

As we have seen in \nref{lpar:axiom-of-choice}, the ``naive'' extraction of a choice function from a dependent pair is a direct consequence of the rules for $\Pi$- and $\Sigma$-types, requiring no axioms. However, this relies on having the explicit data of the elements. 

To formulate the classical Axiom of Choice, we must express the proposition that a choice function exists even when the specific witnesses have been truncated into mere propositions. 

\begin{defn}\label{defn:tac}
    Let $X$ be a set, $A\colon X\to\fun{Set}_\univ$ be a family of sets, and $P\colon\prod_{x\colon X} A(x)\to\fun{Prop}_\univ$ be a family of mere propositions. The \textsl{axiom of choice} is the following inhabitancy assertion:
    \begin{equation}\label{eq:tac}
        \tac\colon
            \Bigl(
                \prod_{x\colon X}\;\Bigl\lVert\sum_{a\colon A(x)}P(x,a)\Bigr\rVert
            \Bigr)
            \to
            \Bigl\lVert
                \sum_{f\colon\prod_{x\colon X}A(x)}\;\prod_{x\colon X}P(x,f(x))
            \Bigr\rVert.
    \end{equation}
\end{defn}

\begin{lem}\fe\label{lem:traditional-tac}
    The axiom of choice is equivalent to the statement that for any set\/ $X$ and any family of sets\/ $C\colon X\to\fun{Set}_\univ$, the function type
    \begin{equation}\label{eq:traditional-tac}
        \Bigl(\prod_{x\colon X}\lVert C(x)\rVert\Bigr)
        \to
        \Bigl\lVert\prod_{x\colon X}C(x)\Bigr\rVert
    \end{equation}
    is inhabited.
\end{lem}

\begin{proof}
    Let $X,A,P$ be as in Definition~\ref{defn:tac}. By Theorem~\ref{thm:ac-generalized}, the \rhs of \eqref{eq:tac} is equivalent to
    \begin{equation}\label{eq:tac-codomain-equivalent}
        \Bigl\lVert
            \prod_{x\colon X}\;\sum_{y\colon A(x)}P(x,y)
        \Bigr\rVert.
    \end{equation}
    Let
    \[
        C(x)\defeq\sum_{y\colon A(x)}P(x,y).
    \]
    By Theorem~\ref{thm:sets}~e), $C(x)$ is a set for $x\colon X$. Hence, if \eqref{eq:traditional-tac} holds, applying it to $C$ yields a map to \eqref{eq:tac-codomain-equivalent}. Since the equivalence of Theorem~\ref{thm:ac-generalized} lifts through propositional truncation, composing these maps yields a witness for \eqref{eq:tac}.

    To show that \eqref{eq:tac} yields \eqref{eq:traditional-tac}, take $A(x)\defeq C(x)$ and $P(x,y)\defeq\fun1$. Then
    \[
        \sum_{a\colon A(x)}P(x,a)
            \equiv\Bigl(\sum_{a\colon C(x)}\fun1\Bigr)
            \simeq C(x)
    \]
    where the equivalence is given by $\pi_1$ and $c\mapsto\dep{c,\fun\star}$. 

    On the other hand,
    \[
        \prod_{x\colon X}P(x,f(x))
            \equiv\Bigl(\prod_{x\colon X}\fun1\Bigr)
            \equiv(X\to\fun1)
            \simeq\fun1,
    \]
    where the equivalence follows from part b) of Theorem~\ref{thm:mere-propositions} and Lemma~\ref{lem:mere-prop-and-1} because $X\to\fun1$ is inhabited by $x\mapsto\fun\star$.
    
    Consequently, the total space inside the truncation on the \rhs of \eqref{eq:tac} reduces to
    \[
        \Bigl(
            \sum_{f\colon\prod_{x\colon X}C(x)}\;\prod_{x\colon X}P(x,f(x))
        \Bigr)
        \simeq
        \Bigl(
            \sum_{f\colon\prod_{x\colon X}C(x)}\fun1
        \Bigr)
        \simeq
        \Bigl(
            \prod_{x\colon X}C(x)
        \Bigr).
    \]
    Because equivalences are preserved under truncation [cf.~Lemma~\ref{lem:truncation-functor}], the map provided by \eqref{eq:tac} constructs an inhabitant of \eqref{eq:traditional-tac}.
\end{proof}


\begin{thm}\label{thm:generalized-total-space}
    Let\/ $P\colon\univ\to\fun{Prop}_\univ$ be a mere proposition family. Then, for any\/ $u\equiv\dep{A,p}$ and\/ $v\equiv\dep{B,q}$ in\/ $\Sigma_\univ P$, there is an equivalence:
    \[
        (u\eq{\Sigma_\univ P}v)\simeq(A\simeq B).
    \]
\end{thm}

\begin{proof}
    By Lemma~\ref{lem:total-space-with-mere-proposition-fibers}, there is an equivalence:
    \[
        (u\eq{\Sigma_\univ P}v)\simeq(A\eq{\univ}B).
    \]
    The conclusion follows from the univalence axiom by composition of the former equivalence with:
    \[
        (A\eq{\univ}B)\simeq(A\simeq B).
    \]
\end{proof}


\begin{cor}\label{cor:general-tac-is-false}
    The following type is inhabited:
    \[
        \sum_{X\colon\univ}\;\sum_{C\colon X\to\fun{Set}_\univ}
        \biggl(
            \Bigl(\prod_{x\colon X}\lVert C(x)\rVert\Bigr)
            \to
            \Bigl\lVert\prod_{x\colon X}C(x)\Bigr\rVert
        \biggr)\to\fun0.
    \]
    In other words, there exists a type\/ $X$ and a family\/ $C\colon X\to\fun{Set}_\univ$ such that\/ {\upshape\eqref{eq:traditional-tac}} is false.
\end{cor}

\begin{proof}
    Define
    \[
        X\defeq\sum_{A\colon\univ}\lVert\fun2\eq{\univ}A\rVert.
    \]
    Since $\fun2$ is a set [cf.~Theorem~\ref{thm:boolean-type-is-set}], it follows by Lemma~\ref{lem:trucated-eq-to-set-is-set} that if $\dep{A,p}\colon X$, then $A$ must be a set as well.
    
    By Theorem~\ref{thm:generalized-total-space}, for any $u\defeq\dep{A,p}$ and $v\defeq\dep{B,q}$ in $X$, there is an equivalence:
    \begin{equation}\label{eq:pi-1-equivalence}
        (u\eq Xv)\simeq(A\simeq B).
    \end{equation}
    
    From \eqref{eq:pi-1-equivalence}, we infer using Corollary~\ref{cor:set-simeq-set-is-set} that $A\simeq B$ is a set. Hence, $u\eq Xv$ is a set as well.
    
    Let $x_0\defeq\dep{\fun2,\lvert\fun{refl}_{\fun2}\rvert}$. It follows that the family 
    \begin{align*}
        C\colon X&\to\univ\\
        x&\mapsto x_0\eq Xx, 
    \end{align*}
    has the property that $C(x)$ is a set for all $x\colon X$.
    
    In the particular case where $v\defeq u\defeq x_0$, \eqref{eq:pi-1-equivalence} yields
    \[
        (x_0\eq Xx_0)\simeq(\fun2\simeq\fun2).
    \]
    As Exercise~\ref{exr:2-simeq-2-simeq-2} shows, $(\fun2\simeq\fun2)\simeq\fun2$. Using the lemma from that exercise, we deduce that $x_0\eq Xx_0$ has two distinct inhabitants. Hence, $X$ is not a set. More precisely, there is a map $\fun{isSet}(X)\to\fun0$.

    Suppose
    \[
        f\colon
            \Bigl(\prod_{x\colon X}\lVert C(x)\rVert\Bigr)
            \to
            \Bigl\lVert\prod_{x\colon X}C(x)\Bigr\rVert.
    \]
    Given
    \[
        \zeta\colon\prod_{x\colon X}\lVert x_0\eq Xx\rVert,
    \]
    we obtain
    \[
        f(\zeta)\colon\Big\lVert\prod_{x\colon X}x_0\eq Xx\Bigr\rVert.
    \]
    Since $\fun{isContr}(X)$ is a mere proposition by Theorem~\ref{thm:isContr-is-a-mere-proposition}, using recursion on the \rhs, to construct a map
    \[
        \Big\lVert\prod_{x\colon X}x_0\eq Xx\Bigr\rVert\to\fun{isContr}(X),
    \]
    it suffices to provide a map
    \[
        \iota\colon\Bigl(\prod_{x\colon X}x_0\eq Xx\Bigr)\to\fun{isContr}(X),
    \]
    which we can do by setting $\iota(g)\defeq\dep{x_0,g}$.

    Applying the recursion principle of the propositional truncation, we obtain an element of $\fun{isContr}(X)$. Since $\fun{isContr}(X)$ is a mere proposition, and every mere proposition is a set [cf.~Proposition~\ref{prop:isSet-and-isProp-are-mere-propositions}], by composition we obtain a witness
    \[
        \epsilon\colon\Bigl\lVert\prod_{x\colon X}x_0\eq Xx\Bigr\rVert\to\fun0.
    \]
    Consequently, the map $\lambda f.\,f(\zeta)\circ\epsilon$ fulfills the requirement of the statement.
    %
    
\end{proof}

\begin{exr}\fe\label{exr:equivalent-to-set-is-set}
    Prove that if\/ $A\simeq B$ and\/ $A$ is a set, then so is\/ $B$.
\end{exr}

\begin{solution}
    Let $\dep{f,(\eta,\vartheta)}\colon A\simeq B$. Take $x,y\colon B$. Given $p,q\colon x\eq By$, we have
    \[
        \fun{ap}_{f^{-1}}(p),\fun{ap}_{f^{-1}}(q)
            \colon f^{-1}(x)\eq A f^{-1}(y).
    \]
    Since $A$ is a set, its identity types are mere propositions. Therefore,
    \[
        \fun{ap}_{f^{-1}}(p)
        \eq{f^{-1}(x)\eq A f^{-1}(y)}
        \fun{ap}_{f^{-1}}(q).
    \]
    Applying $\fun{ap}_f$ and using Proposition~\ref{prop:ap-properties}, we obtain
    \[
        \fun{ap}_{f\circ f^{-1}}(p)
        \eq{f(f^{-1}(x))\eq B f(f^{-1}(y))}
        \fun{ap}_{f\circ f^{-1}}(q).
    \]
    On the other hand, the naturality of the homotopy $\eta\colon f\circ f^{-1}\sim\id_B$ [cf.~\eqref{eq:nat-transformation}] yields
    \[
        \fun{ap}_{f\circ f^{-1}}(p)\ct\eta_y
            \eq{}\eta_x\ct\fun{ap}_{\id_B}(p)
            \eq{}\eta_x\ct p,
            \quad\text{in }f(f^{-1}(x))\eq B y,
    \]
    and similarly for $q$. By concatenating $\eta_y$ on the right it follows that
    \[
        \eta_x\ct p
        \eq{f(f^{-1}(x))\eq B y}
        \eta_x\ct q.
    \]
    Thus, concatenating with $\eta_x^{-1}$ on the left constructs the desired path
    \[
        p\eq{x\eq By}q.
    \]
\end{solution}

\begin{exr}
    Prove that if\/ $A$ and\/ $B$ are sets, then so is\/ $A+B$.
\end{exr}

\begin{solution}
    We have to construct a witness
    \[
        \sigma\colon\prod_{u,v\colon A+B}\;\prod_{p,q\colon u\eq{A+B}v}
            p\eq{u\eq{A+B}v}q.
    \]
    Let $\sigma_A\colon\fun{isSet}(A)$ and $\sigma_B\colon\fun{isSet}(B)$. By double induction on $u$ and $v$, we first consider the case where both are in the left component. With the homotopy
    \[
        \eta(p)\colon\fun{decode}(\fun{encode}(p))\eq{u\eq{A+B}v}p
    \]
    established by path induction, we define
    \[
        \sigma(\fun{inl}(x),\fun{inl}(y))(p,q)\defeq
            \eta(p)^{-1}\ct
            \fun{ap}_{\fun{decode}}
            \bigl(
                \sigma_A(x,y)(\fun{encode}(p),\fun{encode}(q))
            \bigr)
            \ct\eta(q),
    \]
    where $\fun{encode}$ and $\fun{decode}$ are the functions defined in \nref{equality-in-coproducts}. This is well-defined because, in this case,
    \[
        \fun{encode}(p),\fun{encode}(q)\colon x\eq Ay,
    \]
    and so
    \[
        \sigma_A(x,y)(\fun{encode}(p),\fun{encode}(q))
            \colon
                \fun{encode}(p)
                \eq{x\eq Ay}
                \fun{encode}(q).
    \]
    A similar construction yields
    \[
        \sigma(\fun{inr}(x),\fun{inr}(y))(p,q)\defeq
            \eta(p)^{-1}\ct
            \fun{ap}_{\fun{decode}}
            \bigl(
                \sigma_B(x,y)(\fun{encode}(p),\fun{encode}(q))
            \bigr)
            \ct\eta(q).
    \]
    In the cases $u\equiv\fun{inl}(x)$ and $v\equiv\fun{inr}(y)$, or $u\equiv\fun{inr}(x)$ and $v\equiv\fun{inl}(y)$, considering that $\fun{encode}(p),\fun{encode}(q)\colon\fun0$, we simply define
    \[
        \sigma(u,v)(p,q)
            \defeq\fun{rec}_{\fun0}(p\eq{u\eq{A+B}v}q)(\fun{encode}(p)).
    \]
\end{solution}

\begin{exr}\fe
    Show that\/ $A$ is a mere proposition if, and only if, $A\to A$ is contractible.
\end{exr}

\begin{solution}
    Suppose that $A$ is a mere proposition. By part b) of Theorem~\ref{thm:mere-propositions}, the type $A\to A$ is also a mere proposition. Since $\id_A\colon A\to A$, it is inhabited. Therefore, it is contractible by Lemma~\ref{lem:contractible-equivalences}.

    Conversely, assume that $A\to A$ is contractible. Define
    \begin{align*}
        \iota\colon A&\to(A\to A)\\
        x&\mapsto\lambda z.\,x.
    \end{align*}
    We have a witness
    \[
        h\colon\prod_{a\colon A}\;\prod_{f\colon A\to A}\iota(a)\eq{A\to A}f.
    \]
    Then the dependent function
    \begin{align*}
        k\colon\prod_{a\colon A}\;&\prod_{x\colon A}a\eq Ax\\
        a\mapsto&\lambda x.\,\fun{ap}_{\fun{ev}_a}(h(a,\iota(x)))
    \end{align*}
    is well-defined because $h(a,\iota(x))\colon\iota(a)\eq{A\to A}\iota(x)$ and so
    \[
        \fun{ap}_{\fun{ev}_a}(h(a,\iota(x)))\colon a\eq Ax.
    \]
\end{solution}

\begin{exr}\fe
    Show that\/ $\fun{isProp}(A)\simeq(A\to\fun{isContr}(A))$.
\end{exr}

\begin{solution}
    By Theorem~\ref{thm:isContr-is-a-mere-proposition}, $\fun{isContr}(A)$ is a mere proposition. By part b) of Theorem~\ref{thm:mere-propositions}, $A\to\fun{isContr}(A)$ is a mere proposition.

    Consider
    \begin{align*}
        \fun{isProp}(A)&\to(A\to\fun{isContr}(A))\\
        u&\mapsto\lambda a.\,\dep{a,\lambda x.\,u(a,x)},
    \end{align*}
    which is well-defined because $u(a,x)\colon a\eq Ax$ for all $a,x\colon A$.

    On the other hand, we have
    \begin{align*}
        (A\to\fun{isContr}(A))&\to\fun{isProp}(A)\\
        f&\mapsto\lambda x,y.\,\pi_2(f(x))(x)^{-1}\ct\pi_2(f(x))(y),
    \end{align*}
    also well-defined because
    \[
        f(x)\colon\sum_{a\colon A}\;\prod_{y\colon A}a\eq Ay
    \]
    and so, $\pi_2(f(x))\colon\prod_{y\colon A}\pi_1(f(x))\eq Ay$. Hence,
    \[
        \pi_2(f(x))(x)^{-1}\ct\pi_2(f(x))(y)\colon x\eq Ay.
    \]
    The result follows from Lemma~\ref{lem:mere-proposition-equivalence}.
\end{solution}

\begin{exr}\fe\label{exr:A+negA-is-mere-proposition}
    Show that if\/ $A$ is a mere proposition, then so is\/ $A+\neg A$. In particular, in the context of\/ {\upshape Notation~\ref{ntn:logic}}, there is no need to apply propositional truncation to define\/ $A\lor\neg A\defeq A+\neg A$.
\end{exr}

\begin{solution}
    To apply part~e) of Theorem~\ref{thm:mere-propositions}, we must first ensure that both components of the coproduct are mere propositions. We are given that\/ $A$ is a mere proposition. The second component is\/ $\neg A\equiv A\to\fun0$. To see that this is a mere proposition, let\/ $f,g\colon A\to\fun0$. For any\/ $x\colon A$, the terms\/ $f(x)$ and\/ $g(x)$ belong to\/ $\fun0$. Since\/ $\fun0$ is a mere proposition, we have a pointwise equality\/ $f(x)\eq{\fun0}g(x)$. By function extensionality, this pointwise equality yields a path\/ $f\eq{A\to\fun0}g$. Thus,\/ $\neg A$ is a mere proposition.

    Since both\/ $A$ and\/ $\neg A$ are mere propositions, it suffices to show that they are mutually exclusive. That is, we must provide a map
    \[
        A\to((A\to\fun0)\to\fun0).
    \]
    Given\/ $a\colon A$ and\/ $f\colon A\to\fun0$, we define such a map by
    \[
        \fun{ev}(a)(f)\defeq f(a).
    \]
\end{solution}

\begin{exr}\label{exr:when-isequiv-is-mere-proposition}
    Assume that some type\/ $\fun{isequiv}(f)$ satisfies that, for any two inhabitants\/ $e_1,e_2\colon\fun{isequiv}(f)$, we have\/ $e_1\eq{\fun{isequiv}(f)}e_2$. Show that the type\/ $\lVert\fun{qinv}(f)\rVert$ satisfies the same condition and is equivalent to\/ $\fun{isequiv}(f)$.
\end{exr}

\begin{solution}
    The condition on\/ $\fun{isequiv}(f)$ means that it is a mere proposition. By the universal property of propositional truncation, since\/ $\fun{isequiv}(f)$ is a mere proposition, the map\/ $\fun{qinv}(f)\to\fun{isequiv}(f)$ from Proposition~\ref{prop:qinv<->isequiv} induces a map\/ $\lVert\fun{qinv}(f)\rVert\to\fun{isequiv}(f)$.
    
    On the other hand, the map\/ $\fun{isequiv}(f)\to\fun{qinv}(f)$ from the same proposition, followed by the constructor\/ $\fun{qinv}(f)\to\lVert\fun{qinv}(f)\rVert$, $e\mapsto\lvert e\rvert$, is a function\/ $\fun{isequiv}(f)\to\lVert\fun{qinv}(f)\rVert$.

    By Lemma~\ref{lem:mere-proposition-equivalence}, having functions in both directions between mere propositions establishes an equivalence, so we deduce that\/ $\lVert\fun{qinv}(f)\rVert\simeq\fun{isequiv}(f)$. Moreover,\/ $\lVert\fun{qinv}(f)\rVert$ trivially satisfies the condition because propositional truncations are mere propositions by definition.
\end{solution}

\begin{exr}\label{exr:Lem-implies-2-props}
    Show that if\/ $\fun{Lem}$ {\upshape[Definition~\ref{defn:lem}]} holds, then
    \[
        \Bigl(\sum_{A\colon\univ}\fun{isProp}(A)\Bigr)\simeq\fun2.
    \]
\end{exr}

\begin{solution}
    \textsc{Idea:} The $\Sigma$-type classifies mere propositions into two groups: the empty ones and the contractible ones.
    
    Fix $A\colon\univ$. By hypothesis, there exists a function $\lambda\colon\fun{Lem}$ satisfying
    \[
        \lambda(A)\colon\fun{isProp}(A)\to(A+\neg A).
    \]
    Let $c_1\colon A\to\fun2$ and $c_0\colon\neg A\to\fun2$ be the constant maps taking values $1_{\fun2}$ and $0_{\fun2}$, respectively. These maps induce a function $c_A\colon A+\neg A\to\fun2$ via the coproduct recursor. Then, we can define
    \begin{align*}
        f\colon\Bigl(\sum_{A\colon\univ}\fun{isProp}(A)\Bigr)
            &\to\fun2\\
        \dep{A,\epsilon}&\mapsto c_A(\lambda(A)(\epsilon)).
    \end{align*}
    For the backward function, consider
    \begin{align*}
        \tau\colon\fun2&\to\sum_{A\colon\univ}\fun{isProp}(A)\\
        0_{\fun2}&\mapsto\dep{\fun0,\epsilon_0},\\
        1_{\fun2}&\mapsto\dep{\fun1,\epsilon_1},
    \end{align*}
    where $\epsilon_0\colon\fun{isProp}(\fun0)$ and $\epsilon_1\colon\fun{isProp}(\fun1)$ are the canonical dependent functions defined by induction on $\fun0$ and $\fun1$, respectively, as follows:
    \[
        \epsilon_0\defeq\fun{ind}_{\fun0}
            \Bigl(
                \lambda x.\,\prod_{y\colon\fun0}x\eq{\fun0}y
            \Bigr),\quad
        \epsilon_1\defeq\fun{ind}_{\fun1}
            \Bigl(
                \lambda x.\,\prod_{y\colon\fun1}x\eq{\fun1}y,
                \fun{ind}_{\fun1}
                    \Bigl(
                        \lambda y.\,\star\eq{\fun1}y,
                        \fun{refl}_{\star}
                    \Bigr)
            \Bigr).
    \]
    Let us now evaluate both compositions. For $f\circ\tau$, we claim that $(f\circ\tau)(0_{\fun2})\eq{}0_{\fun2}$ and $(f\circ\tau)(1_{\fun2})\eq{}1_{\fun2}$. Since $\lambda$ is an abstract function, the term $\lambda(\fun0)(\epsilon_0)$ does not compute definitionally. However, we can prove $c_{\fun0}(z)\eq{}0_{\fun2}$ for all $z\colon\fun0+\neg\fun0$ by induction on the coproduct. If $z\equiv\fun{inl}(x)$, the context is contradictory since $x\colon\fun0$. If $z\equiv\fun{inr}(y)$, then $c_{\fun0}(\fun{inr}(y))\equiv c_0(y)\equiv 0_{\fun2}$. Thus,
    \[
        (f\circ\tau)(0_{\fun2})
            \equiv f(\dep{\fun0,\epsilon_0})
            \equiv c_{\fun0}(\lambda(\fun0)(\epsilon_0))
            \eq{} 0_{\fun2}.
    \]
    By a symmetric argument on $\fun1+\neg\fun1$, any left injection maps to $1_{\fun2}$ definitionally, and the right injection is contradictory since $\neg\fun1$ is empty. Thus, $c_{\fun1}(z)\eq{}1_{\fun2}$ for all $z\colon\fun1+\neg\fun1$, which yields
    \[
        (f\circ\tau)(1_{\fun2})
            \equiv f(\dep{\fun1,\epsilon_1})
            \equiv c_{\fun1}(\lambda(\fun1)(\epsilon_1))
            \eq{} 1_{\fun2}.
    \]
    For the other composition, we need to prove that
    \[
        \tau(f(\dep{A,\epsilon}))
            \equiv\tau(c_A(\lambda(A)(\epsilon)))
            \eq{}\dep{A,\epsilon}.
    \]
    Therefore, it suffices to prove that $\tau(c_A(z))\eq{}\dep{A,\epsilon}$ for all $z\colon A+\neg A$. By induction on $A+\neg A$, we are reduced to the cases $z\equiv\fun{inl}(a)$ and $z\equiv\fun{inr}(\zeta)$, where $a\colon A$ and $\zeta\colon A\to\fun0$.

    In the first case, the goal reduces to showing that $\dep{\fun1,\epsilon_1}\eq{}\dep{A,\epsilon}$ is inhabited. Since $\epsilon\colon\fun{isProp}(A)$, $A$ is a mere proposition. By Lemma~\ref{lem:contractible-equivalences}, it is contractible because $a\colon A$. By the same lemma, it follows that $A\simeq\fun1$. Hence, the univalence axiom implies the existence of a path $p\colon\fun1\eq\univ A$. To prove the goal, we have to show that
    \[
        \fun{transport}^{\fun{isProp}}(p,\epsilon_1)
            \eq{\fun{isProp}(A)}\epsilon,
    \]
    which holds trivially because, by Proposition~\ref{prop:isSet-and-isProp-are-mere-propositions}, $\fun{isProp}(A)$ is a mere proposition.

    In the second case, the goal reduces to proving that $\dep{\fun0,\epsilon_0}\eq{}\dep{A,\epsilon}$.

    To see that $\fun0\eq\univ A$, we need the following lemma:

    \medskip

    \textbf{Lemma.} \textit{If $A\to\fun0$ is inhabited, then $A\eq\univ\fun0$.}
    
    \textsc{Proof:} By univalence, it suffices to find an equivalence $A\simeq\fun0$.
    
    Let $f\colon A\to\fun0$ be the function given by hypothesis. We claim that $g\defeq\fun{rec}_{\fun0}(A)$ is a quasi-inverse of $f$.

    We must construct a right homotopy $\eta\colon f\circ g\sim\id_{\fun0}$ and a left homotopy $\vartheta\colon g\circ f\sim\id_A$.

    By \nref{lpar:coproduct-induction}, for $\eta$ we have $\fun{ind}_{\fun0}(P)$, where $P\colon\fun0\to\univ$ is defined as $P(y)\defeq f(g(y))\eq{\fun0}y$. For $\vartheta$, fix $x\colon A$ and consider the type
    \[
        Q_x\defeq g(f(x))\eq{A}x.
    \]
    This defines the function $\fun{rec}_{\fun0}(Q_x)\colon\fun0\to Q_x$. Applying it to $f(x)$ yields a witness for $g(f(x))\eq{A}x$.

    Since we have maps in both directions and their corresponding homotopies, we conclude that $A\simeq\fun0$. The result follows from the univalence axiom. \hfill\lqed

    \medskip

    It remains to be shown that $\fun{transport}^{\fun{isProp}}(p,\epsilon_0)\eq{\fun{isProp}(A)}\epsilon$, where $p\colon\fun0\eq\univ A$. Again, this is trivial because $\fun{isProp}(A)$ is a mere proposition.
\end{solution}

\begin{exr}\label{exr:propositional-resizing}
    Show that if\/ $\univ_{i+1}$ satisfies\/ $\fun{Lem}$, then the canonical inclusion\/ $\fun{Prop}_{\univ_i}\to\fun{Prop}_{\univ_{i+1}}$ is an equivalence.
\end{exr}

\begin{solution}
    Let us recall that, by definition,
    \[
        \fun{Prop}_\univ\equiv\sum_{A\colon\univ}\fun{isProp}(A),
    \]
    and that the canonical inclusion is
    \begin{align*}
        \iota_i\colon\fun{Prop}_{\univ_i}
            &\to\fun{Prop}_{\univ_{i+1}}\\
        \dep{A,\epsilon}&\mapsto\dep{A,\epsilon},
    \end{align*}
    where $A$ is considered as a type in $\univ_{i+1}$ on the \rhs, by \textsc{\nameref{lpar:cumulative-property}}.
    
    By Exercise~\ref{exr:Lem-implies-2-props}, the hypothesis means that $\fun{Prop}_{\univ_{i+1}}\simeq\fun2$. In addition, the triangle
    \[
        \begin{tikzcd}
            \fun{Prop}_{\univ_i}
                    \arrow[r,"\iota_i"]
                    \arrow[rd,dashed,"\phi"]
                &\fun{Prop}_{\univ_{i+1}}
                    \arrow[d,"f_{i+1}"]\\
                &\fun2
                    \arrow[lu,"\tau_i",bend left]
                    \arrow[u,"\tau_{i+1}"',bend right=60]
        \end{tikzcd}
    \]
    where $\tau_i$, $\tau_{i+1}$, and $f_{i+1}$ are the maps constructed in that exercise, defines by composition a function $\phi\colon\fun{Prop}_{\univ_i}\to\fun2$.

    Moreover, using the notation of Exercise~\ref{exr:Lem-implies-2-props}, we have
    \begin{align*}
        (\phi\circ\tau_i)(0_{\fun2})
            &\equiv f_{i+1}(\iota_i(\dep{\fun0,\epsilon_0}))\\
            &\eq{} f_{i+1}(\tau_{i+1}(0_{\fun2}))\\
            &\equiv 0_{\fun2},
    \end{align*}
    and similarly for $1_{\fun2}$. Hence, by induction on $\fun2$, we deduce that $\phi\circ\tau_i\eq{\fun2\to\fun2}\id_{\fun2}$.

    For the other composition,
    \begin{align*}
        (\tau_i\circ\phi)(\dep{A,\epsilon})
            &\equiv
                \tau_i(f_{i+1}(\iota_i(\dep{A,\epsilon})))\\
            &\eq{}\tau_i(f_{i+1}(\dep{A,\epsilon}))\\
            &\equiv\tau_i(c_A(\lambda(A)(\epsilon))),
    \end{align*}
    where $\lambda$ is the witness of $\fun{Lem}\colon\univ_{i+1}$ given by hypothesis, and $c_A\colon(A+\neg A)\to\fun2$ is the map of the previous exercise.

    We claim that $\tau_i(c_A(z))\eq{}\dep{A,\epsilon}$ for all $A\colon\univ_i$ and $z\colon A+\neg A$. But this was proved in the previous exercise by induction on $z$, using the fact that $\fun{isProp}(A)$ is a mere proposition (and not that $\fun{Lem}\colon\univ_i$ was inhabited).

    Therefore, $\tau_i$ and $\phi$ are quasi-inverses, which implies that $\iota_i$ induces an equivalence.
    %
    
\end{solution}

\begin{exr}
    Prove that, in general, there is no map\/ $\lVert A\rVert\to A$ for an arbitrary type\/ $A\colon\univ$. (Note, however, that such maps do exist for specific types; for example, {\upshape Exercise~\ref{exr:when-isequiv-is-mere-proposition}} implies that\/ $\fun{qinv}(f)$ is one such type.)
\end{exr}

\begin{solution}
    Note that if $A$ has a witness $a\colon A$, then $\lVert A\rVert$ is an inhabited mere proposition, hence equivalent to $\fun1$. Thus, we can define a map $\lVert A\rVert\to A$ as the composition $\lVert A\rVert\to\fun1\to A$, where the second map is defined by mapping $\star\colon\fun1$ to $a$. What the exercise emphasizes is the nonexistence of a general function that would identify a witness for each inhabited type.

    To solve the exercise, we have to define a map
    \[
        \epsilon\colon
            \Bigl(
                \prod_{X\colon\univ}\lVert X\rVert\to X
            \Bigr)
            \to\fun0.
    \]
    Take $u$ in the domain. Then $u(\fun2)\colon\lVert\fun2\rVert\to\fun2$.

    Suppose that $p\colon A\eq\univ B$ is a path between two types $A$ and $B$. Then
    \[
        \fun{apd}_u(p)\colon
            \fun{transport}^P(p,u(A))\eq{\lVert B\rVert\to B}u(B),
    \]
    where $P\colon\univ\to\univ$ is defined as $P(X)\defeq\lVert X\rVert\to X$. Applying Proposition~\ref{prop:non-dependent-transport-composition} to the case $(A\to B)\equiv(\lVert-\rVert\to\id_\univ)$ and $f\equiv u(A)$, we obtain
    \[
        \fun{transport}^P(p,u(A))
        \eq{\lVert B\rVert\to B}
        \fun{transport}^{\id_\univ}(p)
            \circ u(A)
            \circ\fun{transport}^{\lVert-\rVert}(p^{-1}).
    \]
    By Theorem~\ref{thm:idtoeqv-def}, the first transport on the \rhs induces the equivalence $\fun{idtoeqv}(p)\colon A\simeq B$. The second transport is an equivalence between $\lVert B\rVert$ and $\lVert A\rVert$.

    In particular, if $p\equiv\fun{ua}(e)$ for some equivalence $e\colon A\simeq B$, then $\fun{idtoeqv}(p)\equiv e$. Applying $\fun{happly}$ to the path $\fun{apd}_u(p)$ yields, for any $y\colon\lVert B\rVert$, a pointwise path
    \[
        q_y\colon e\bigl(u(A)(\lVert e\rVert^{-1}(y))\bigr) \eq{B} u(B)(y).
    \]
    Specializing this to the case where $B\defeq A\defeq\fun2$, $e\equiv\fun{not}$, and $y\equiv\lvert 1_{\fun2}\rvert$, we obtain a path
    \[
        q_{\lvert 1_{\fun2}\rvert}\colon
            \fun{not}\bigl(u(\fun2)(\lVert\fun{not}\rVert^{-1}(\lvert 1_{\fun2}\rvert))\bigr)
            \eq{\fun2}
            u(\fun2)(\lvert 1_{\fun2}\rvert).
    \]
    Since $\lVert\fun2\rVert$ is a mere proposition, there is a path $s\colon\lVert\fun{not}\rVert^{-1}(\lvert 1_{\fun2}\rvert)\eq{\lVert\fun2\rVert}\lvert 1_{\fun2}\rvert$. Applying the function $\lambda z.\,\fun{not}(u(\fun2)(z))$ to $s$ yields a path
    \[
        s'\colon \fun{not}\bigl(u(\fun2)(\lVert\fun{not}\rVert^{-1}(\lvert 1_{\fun2}\rvert))\bigr)
        \eq{\fun2}
        \fun{not}(u(\fun2)(\lvert 1_{\fun2}\rvert)).
    \]
    Concatenating $(s')^{-1}$ with $q_{\lvert 1_{\fun2}\rvert}$, we obtain a path
    \[
        r\colon\fun{not}(u(\fun2)(\lvert 1_{\fun2}\rvert))\eq{\fun2}u(\fun2)(\lvert 1_{\fun2}\rvert).
    \]
    Thus, we can define
    \[
        \epsilon(u)\defeq\fun{notid}(u(\fun2)(\lvert 1_{\fun2}\rvert),r),
    \]
    where $\fun{notid}$ is the function defined in Remark~\ref{rem:notid}.
\end{solution}

\begin{exr}

    \[
        \bigl\lVert\lVert A\rVert\to A\bigr\rVert.
    \]
\end{exr}

\begin{solution}
    We have to construct a dependent function
    \[
        \epsilon\colon\prod_{A\colon\univ}
            \lVert P(A)\rVert,
    \]
    where $P(A)\defeq\lVert A\rVert\to A$.
    
    By hypothesis, we have a witness of $\fun{Lem}$, say
    \[
        \lambda\colon\prod_{A\colon\univ}\fun{isProp}(A)\to(A+\neg A).
    \]
    Fix $A\colon\univ$. We have
    \[
        \lambda(\lVert A\rVert)(\fun{squash})
            \colon\lVert A\rVert+\neg\lVert A\rVert.
    \]
    To define a map from $\lVert A\rVert+\neg\lVert A\rVert$ to $\lVert P(A)\rVert$, we have to specify two maps: $\alpha\colon\lVert A\rVert\to\lVert P(A)\rVert$ and $\beta\colon\neg\lVert A\rVert\to\lVert P(A)\rVert$.

    To define $\alpha$ we use recursion:
    \[
        \alpha\defeq
            \fun{rec}_{\lVert A\rVert}
                (
                    \lVert P(A)\rVert,
                    a\mapsto\lvert z\mapsto a\rvert
                ).
    \]
    For $\beta$, take $u\colon\lVert A\rVert\to\fun0$. Then,
    \[
        \beta(u)\defeq
            \lvert z\mapsto\fun{rec}_{\fun0}(A)(u(z))\rvert.
    \]
    Since $\alpha$ and $\beta$ are well defined, they induce a map
    \[
        \gamma(A)\colon(\lVert A\rVert+\neg\lVert A\rVert)\to\lVert P(A)\rVert.
    \]
    To conclude, we define
    \[
        \epsilon(A)\defeq\gamma(A)(\lambda(\lVert A\rVert)(\fun{squash})).
    \]
    \textbf{Note.} Saying that $\lVert P(A)\rVert$ is inhabited means that there is some witness of $P(A)$, i.e., some way to choose an element of $A$, provided that $A$ is inhabited. Since HoTT requires that every element be constructed, this claim is not generally true.
    %
    
\end{solution}

\begin{exr}\label{exr:lem-implies-neg-neg-equals-truncation}
    Show that assuming\/ $\fun{Lem}$, the double negation\/ $\neg\neg A$ has the same recursion principle as the propositional truncation\/ $\lVert A\rVert$ but with a propositional computation rule rather than a judgmental one. In other words, prove that assuming\/ $\fun{Lem}$, if\/ $B$ is a mere proposition and we have\/ $f\colon A\to B$, then there is an induced\/ $g\colon\neg\neg A\to B$ such that\/ $g(\lambda u.\,u(a))\eq Bf(a)$ for all\/ $a\colon A$. Deduce that (assuming\/ $\fun{Lem}$) we have\/ $\neg\neg A\simeq\lVert A\rVert$. Thus, under\/ $\fun{Lem}$, the propositional truncation can be defined rather than taken as a separate type former.
\end{exr}

\begin{solution}
    As we saw in Exercise~\ref{exr:neg-neng-neg}, given $a\colon A$, the evaluation at $a$, defined by $\ev a\defeq\lambda u\colon\neg A.\,u(a)$, is a term of $\neg\neg A$.

    As the following diagram shows, $f\colon A\to B$ induces a map $\neg f\colon\neg B\to\neg A$ defined by $v\mapsto v\circ f$.
    \[
        \begin{tikzcd}
            A
                \arrow[r,"f"]
                \arrow[rd,dashed]
            &B
                \arrow[d,"v"]\\
            &\fun0
        \end{tikzcd}
    \]
    Applying this once again yields a map
    \begin{align*}
       \neg\neg f\colon\neg\neg A&\to\neg\neg B\\
       \nu&\mapsto\nu\circ\neg f.
    \end{align*}
    Note moreover that
    \begin{align*}
       (\neg\neg f(\ev a))(v)
            &\equiv\ev a(\neg f(v))\\
            &\equiv\ev a(v\circ f)\\
            &\equiv (v\circ f)(a)\\
            &\equiv \ev{f(a)}(v),
    \end{align*}
    i.e., $\neg\neg f\circ\ev{}\equiv\ev{}\circ f$, which means that the following diagram commutes definitionally:
    \[
        \begin{tikzcd}[row sep=large]
            \neg\neg A
                \arrow[r,"\neg\neg f"]
            &\neg\neg B
                \arrow[d,dashed,bend left=49,"\theta",shift left]\\
            A
                \arrow[r,"f"]
                \arrow[u,"\ev{}"]
            &B
                \arrow[u,"\ev{}"']
        \end{tikzcd}
    \]
    Thus, to obtain $g\colon\neg\neg A\to B$, it suffices to construct $\theta\colon\neg\neg B\to B$.
    
    By Lemma~\ref{lem:A+negA-to-negnegA-to-A}, there is a map
    \[
        \varphi\colon(B+\neg B)\to(\neg\neg B\to B).
    \]
    Therefore, given witnesses $\ell\colon\fun{Lem}$ and $\pi\colon\fun{isProp}(B)$, we obtain a witness
    \[
        \ell(\pi)\colon B+\neg B.
    \]
    Applying $\varphi$, we arrive at the desired map $\theta$:
    \[
        \theta\defeq\varphi(\ell(\pi)).
    \]
    Hence, we can define $g\defeq\theta\circ\neg\neg f$.
    
    To verify that $g(\ev a)\eq Bf(a)$, i.e., that the square above commutes for $\theta$, recall that $B$ is a mere proposition and $\pi(g(\ev a),f(a))$ provides the required path.

    Note that since $\neg\neg X\equiv(\neg X\to\fun0)$, both $\neg\neg A$ and $\neg\neg B$ are mere propositions. In particular, $\ev{}$ and $\theta$ are quasi-inverses of each other by Lemma~\ref{lem:mere-proposition-equivalence}.

    In the case where $B\defeq\lVert A\rVert$ and $f$ is the canonical constructor $a\mapsto\lvert a\rvert$, the map $g\colon\neg\neg A\to\lVert A\rVert$ satisfies $g(\ev a)\eq{\lVert A\rVert}\lvert a\rvert$ for all $a\colon A$.

    Take an element $u\colon\neg A$. Since $u\colon A\to\fun0$ and $\fun0$ is a mere proposition, we obtain an element $\fun{rec}_{\lVert A\rVert}(\fun0,u)\colon\lVert A\rVert\to\fun0$. This defines a map
    \begin{align*}
        \rho\colon\neg A&\to\neg\lVert A\rVert\\
        u&\mapsto\fun{rec}_{\lVert A\rVert}(\fun0,u),
    \end{align*}
    which, in turn, yields a map $\neg\rho\colon\neg\neg\lVert A\rVert\to\neg\neg A$, as we saw above.

    Summarizing in a diagram, we have
    \[
        \begin{tikzcd}[row sep=large,column sep=huge]
            \neg\neg A
                    \arrow[rd,"g"]
                    \arrow[r,"\neg\neg\lvert-\rvert"]
                &\neg\neg\lVert A\rVert
                    \arrow[d,"\simeq",]
                    \arrow[l,"\neg\rho"',bend right=50]\\
            |[text=gray]|A
                    \arrow[r,"\lvert-\rvert"',gray]
                    \arrow[u,"\ev{}",gray]
                &\lVert A\rVert
                    %\arrow[u,"{\ev{}[\simeq]}"']
        \end{tikzcd}
    \]
    Now Lemma~\ref{lem:mere-proposition-equivalence} implies that the top horizontal functions are mutual quasi-inverses. In particular, $g$ induces the desired equivalence $\neg\neg A\simeq\lVert A\rVert$.
    %

\end{solution}

\begin{exr}\fe${}$
    \begin{enumerate}[a),font=\upshape]
        \item Show that for any\/ $A\colon\univ$, the type
        \[
            \prod_{P\colon\fun{Prop}_\univ}(A\to P)\to P
        \]
        has the same recursion principle as\/ $\lVert A\rVert$, at least relative to propositions in\/ $\univ$, with the same judgmental computation rule.
        \item The type considered in the previous part does not lie in the same universe\/ $\univ$, and its recursion principle only applies to mere propositions in\/ $\univ$. Show that if we assume propositional resizing, we can define a type that does lie in the same universe\/ $\univ$ and satisfies the same recursion principle as\/ $\lVert A\rVert$, albeit with only a propositional computation rule. Thus, we can also define the propositional truncation in this case.
    \end{enumerate}

\begin{solution}${}$
    \begin{enumerate}[a)]
        \item Let $\Pi$ denote the $\Pi$-type defined in the statement. We have to prove that $(\Pi\to Q)\simeq(A\to Q)$ for all mere propositions $Q\colon\univ$.

        Fix a mere proposition $Q$. Define the forward map as
        \begin{align*}
            \varphi\colon(\Pi\to Q)&\to(A\to Q)\\
            \alpha&\mapsto\lambda a.\,\alpha(\lambda P.\,\ev a^P),

        where $\ev a^P\colon(A\to P)\to P$ is the evaluation at $a$. This is well-defined because $(\lambda P.\,\ev a^P)\colon\Pi$.
        
        For the backward map, consider
        \begin{align*}
            \psi\colon(A\to Q)&\to(\Pi\to Q)\\
            f&\mapsto\lambda\pi.\,\pi(Q)(f).
        \end{align*}
        Since $\pi(Q)\colon(A\to Q)\to Q$, its value at $f$ inhabits $Q$.

        By part~b) of Theorem~\ref{thm:mere-propositions}, since $Q$ is a mere proposition, the function types $\Pi\to Q$ and $A\to Q$ are also mere propositions. Given that we have constructed maps in both directions, the equivalence follows directly from Lemma~\ref{lem:mere-proposition-equivalence}.
        
        It remains to be shown that the recursion principle of $\Pi$ evaluates judgmentally in the same way as that of the truncation.
        
        Before verifying this property, let us emphasize that, since $\Pi$ quantifies over all mere propositions of $\univ$, it cannot belong to $\univ$ and becomes an element of a universe above it. This differs from the type former $\lVert-\rVert$, which produces a type $\lVert A\rVert$ in the same universe as $A$.
        
        To check the recursion principle of $\Pi$, let us define
        \begin{align*}
            \ev{}\colon A&\to\Pi\\
            a&\mapsto\lambda P.\,\ev a^P.
        \end{align*}
        Given $f\colon A\to Q$, the function $\psi(f)$ makes the following diagram judgmentally commutative:
        \[
            \begin{tikzcd}[baseline=(top.base)]
                |[alias=top]| A
                        \arrow[r,"f"]
                        \arrow[d,"\ev{}"']
                    &Q\\
                \Pi
                        \arrow[ru,"\psi(f)"']
            \end{tikzcd}
            \qquad
            \begin{aligned}[t]
                \psi(f)\circ\ev{}(a)
                    &\equiv (\lambda\pi.\,\pi(Q)(f))(\lambda P.\,\ev a^P)\\
                    &\equiv (\lambda P.\,\ev a^P)(Q)(f)\\
                    &\equiv \ev a^Q(f)\\
                    &\equiv f(a).
            \end{aligned}
        \]

        \item To better understand this part, we need to introduce the following axiom.

        \textbf{Axiom} [Propositional resizing] The canonical map $\iota_i\colon\fun{Prop}_{\univ_i}\to\fun{Prop}_{\univ_{i+1}}$ is an equivalence.

        Using the notation of Exercise~\ref{exr:propositional-resizing}, note that if $\iota_i^{-1}$ is a quasi-inverse of~$\iota_i$, then given $Q'\colon\univ_{i+1}$, the homotopy between $\iota_i\circ\iota_i^{-1}$ and the identity implies that the type $Q\defeq\iota_i^{-1}(Q')$ satisfies $\iota_i(Q)\eq{\univ_{i+1}}Q'$. In particular, $\iota_i(Q)\simeq Q'$.

        It is also worth noting that the universe level of a dependent product $\prod_{x\colon X}Y(x)$ is the maximum of the universe levels of its domain $X$ and its family of codomains $Y(x)$.

        Let $\univ\equiv\univ_i$ and define $\univ'\defeq\univ_{i+1}$. In the case of $\Pi$, since $(A\to P)\to P\colon\univ$ for all $P\colon\fun{Prop}_{\univ}$, and the domain $\fun{Prop}_{\univ}\colon\univ'$, we deduce that $\Pi\colon\univ'$.
        
        By part a) of Theorem~\ref{thm:mere-propositions}, a product of mere propositions is a mere proposition, so $\Pi$ is a mere proposition. This means that it represents an element of $\fun{Prop}_{\univ'}$. 
        
        If propositional resizing is assumed, the equivalence $\fun{Prop}_{\univ}\simeq\fun{Prop}_{\univ'}$ provides a mere proposition $\Pi'\colon\fun{Prop}_{\univ}$ that is equivalent to $\Pi$. Its underlying type belongs to $\univ$. Because $\Pi'$ and $\Pi$ are equivalent, maps out of $\Pi'$ into any proposition $Q\colon\univ$ are equivalent to maps out of $\Pi$, meaning $\Pi'$ satisfies the same recursion principle.
        
        Because the recursion for $\Pi'$ is inherited from its equivalence with $\Pi$, its computation rule only holds propositionally.
        
    \end{enumerate}
    %

\end{solution}    


\begin{exr}\fe
    Assuming $\fun{Lem}$, show that double negation commutes with universal quantification of mere propositions over sets. That is, show that if $X$ is a set and each $C(x)$ is a mere proposition, then $\fun{Lem}$ implies
    \[
        \Bigl(
            \prod_{x\colon X}\neg\neg C(x)
        \Bigr)
        \simeq
        \neg\neg\prod_{x\colon X}C(x). \tag{$\dag$}
    \]
    Observe that if we assume instead that each $C(x)$ is a set, then $(\dag)$ becomes equivalent to the axiom of choice.
\end{exr}

\begin{proof}
    Let $\ell\colon\fun{Lem}$ be a witness of the least excluded middle type. By definition, given a type $A\colon\univ$, we have
    \[
        \ell(A)\colon\fun{isProp}(A)\to(A+\neg A).
    \]
    Applying this to $C(x)$, we obtain a witness
    \[
        \ell(C(x))(\zeta(x))\colon C(x)+\neg C(x),
    \]
    where $\zeta(x)$ inhabits $\fun{isProp}(C(x))$. Applying the map
    \[
        (C(x)+\neg C(x))\to(\neg\neg C(x)\to C(x))
    \]
    from Lemma~\ref{lem:A+negA-to-negnegA-to-A} yields a map
    \[
        e(x)\colon\neg\neg C(x)\to C(x).
    \]
    To define a forward function for $(\dag)$, fix an element $f$ of the \lhs. For $x\colon X$, we have
    \[
        f(x)\colon\neg\neg C(x).
    \]
    Then, to define a map
    \[
        \varphi(f)\colon\Bigl(\neg\prod_{x\colon X}C(x)\Bigr)\to\fun0,
    \]
    for $u\colon\Bigl(\prod_{x\colon X}C(x)\Bigr)\to\fun0$, we specify 
    \[
        \varphi(f)(u)\defeq u\bigl(\lambda x.\,e(x)(f(x))\bigr),
    \]
    which is well-typed because $e(x)(f(x))\colon C(x)$.

    Let $\Pi$ be the dependent product on the \rhs of $(\dag)$. By part a) of Theorem~\ref{thm:mere-propositions}, we know that $\Pi$ is a mere proposition. Hence,
    \[
        \ell(\Pi)(\zeta)\colon\Pi+\neg\Pi,
    \]
    where $\zeta$ is an inhabitant of $\fun{isProp}(\Pi)$. Applying the map
    \[
        (\Pi+\neg\Pi)\to(\neg\neg\Pi\to\Pi)
    \]
    from Lemma~\ref{lem:A+negA-to-negnegA-to-A} yields a map
    \[
        e\colon\Bigl(\neg\neg\prod_{x\colon X}C(x)\Bigr)
            \to\prod_{x\colon X}C(x).
    \]
    Thus, the canonical map $\ev{}\colon C(x)\to\neg\neg C(x)$ [cf.~Exercise~\ref{exr:lem-implies-neg-neg-equals-truncation}] yields a map
    \[
        \psi\colon\Bigl(\neg\neg\prod_{x\colon X}C(x)\Bigr)
            \to\prod_{x\colon X}\neg\neg C(x).
    \]
    Given that both sides of $(\dag)$ are mere propositions, their equivalence follows from Lemma~\ref{lem:mere-proposition-equivalence}.

    As proved in Exercise~\ref{exr:lem-implies-neg-neg-equals-truncation}, double negation and truncation are equivalent in $\fun{Lem}$ is assumed. Therefore, $(\dag)$ is equivalent to
    \[
        \Bigl(
            \prod_{x\colon X}\lVert C(x)\rVert
        \Bigr)
        \simeq
        \Bigl\lVert
            \prod_{x\colon X}C(x)
        \Bigr\rVert.
    \]
    In the case where each $C(x)$ is a set for $x\colon X$, since the canonical backward map
    \[
        \Bigl\lVert\prod_{x\colon X}C(x)\Bigr\rVert
        \to
        \prod_{x\colon X}\lVert C(x)\rVert
    \]
    always exists (even if the types $C(x)$ are not sets), the equivalence $(\dag)$ reduces to the existence of a forward map. By Lemma~\ref{lem:traditional-tac}, the existence of such a map is equivalent to the axiom of choice.
    %
    
\end{proof}

\begin{exr}\fe\label{exr:decidable+mere-proposition-family}
    Suppose\/ $P\colon\N\to\univ$ is a decidable family\footnote{Definition~\ref{defn:lem}.} of mere propositions. Prove that
    \[
        \Bigl\lVert\sum_{n\colon\N}P(n)\Bigr\rVert\to\sum_{n\colon\N}P(n).
    \]
    {\upshape Hint: If $F\colon\N\to\univ$ is a mere proposition family such that $F(n)\eq\univ F(m)$ only when $n\eq\N m$, then $\Sigma_\N F$ is a mere proposition.}
\end{exr}

\begin{solution}
    By hypothesis, there exists
    \[
        \epsilon\colon\prod_{n\colon\N}P(n)+\neg P(n).
    \]
    The $\Pi$-type
    \[
        Q(n)\defeq\prod_{k\colon\N}(k<n)\to\neg P(k)
    \]
    is a mere proposition by parts b) and a) of Theorem~\ref{thm:mere-propositions}.
    The dependent type
    \[
        F(n)\defeq Q(n)\times P(n),
    \]
    which is also a mere proposition by the same theorem, is inhabited when $n$ is the first natural number such that $P(n)$ is true.

    Now consider
    \[
        S\defeq\sum_{n\colon\N}F(n).
    \]
    We claim that $S$ is a mere proposition. Intuitively, it is clear that $S$ can have at most one inhabitant. Take two elements $\dep{n,f}$ and $\dep{k,g}$ of $S$. By Lemma~\ref{lem:total-space-with-mere-proposition-fibers}, to show that they are equal it suffices to show that $n\eq\N k$. If $u\colon k<n$, then
    \[
        \pi_1(f)(u)\colon\neg P(k).
    \]
    However, $\pi_2(g)\colon P(k)$, which yields $\pi_1(f)(u)(\pi_2(g))\colon\fun0$.

    By symmetry, $n<k$ also yields a contradiction. Hence, by part~d) of Proposition~\ref{prop:decidability-of-equality-on-N}, we deduce that $n\eq\N k$, proving our claim that $S$ is a mere proposition.

    Consequently, by recursion on the truncated type, to define a map $\lVert\Sigma_\N P\rVert\to S$, it suffices to define a map $\Sigma_\N P\to S$. And since the second projection induces a map from $S$ to $\Sigma_\N P$, the solution would be complete if we define said map.
    
    The induction principle of $\Sigma$-types reduces our goal further to defining a map $P(n)\to S$ for $n\colon\N$.

    The idea is to construct the desired maps by induction on $n$. However, to enable that, we need an additional property, which we will also prove by induction, namely, that the coproduct
    \[
        S+Q(n)
    \]
    is inhabited for all $n\colon\N$.

    In the case $n\equiv0$, by part g) of Lemma~\ref{lem:order-in-N}, for each $k\colon\N$, there is a witness $\zeta_k\colon(k<0)\to\fun0$. Therefore, the map $\lambda k.\,\lambda u.\,\fun{rec}_{\fun0}(\neg P(k),\zeta_k(u))$ inhabits $Q(0)$.

    In the inductive step, suppose that we have a witness $h\colon S+Q(n)$. To construct a witness of $S+Q(\fun{succ}(n))$ it suffices to define a map from the former type to the latter. We proceed by recursion on the coproduct:
    \begin{itemize}
        \item For the left branch $S\to S+Q(\fun{succ}(n))$, we use the canonical injection $\fun{inl}$.

        \item For the right branch, given $q\colon Q(n)$ we define a function
        \[
            \sigma_n\colon(P(n)+\neg P(n))\to(S+Q(\fun{succ}(n)))
        \]
        using recursion once again:
        
        For the left sub-branch, we define
        \begin{align*}
            P(n)&\to S+Q(\fun{succ}(n))\\
            u&\mapsto\fun{inl}(\dep{n,(q,u)}),
        \end{align*}
        which is well-defined because $(q,u)\colon Q(n)\times P(n)$, the type definitionally equal to $F(n)$, and so $\dep{n,(q,u)}\colon S$.

        For the right sub-branch, if $z\colon\neg P(n)$, using recursion for the third time, we define a map $v_z\colon(k<n)+(k\eq\N n)\to\neg P(k)$ via the functions:
        \begin{align*}
            q(k)&\colon(k<n)\to\neg P(k),\\
            \zeta_z&\colon(k\eq\N n)\to\neg P(k),\quad p\mapsto\fun{transport}^{\neg P}(p^{-1},z).
        \end{align*}
        Thus, precomposing $v_z$ with the canonical map\footnote{This is easily deduced from part e) of Proposition~\ref{prop:decidability-of-equality-on-N}.}
        \[
            (k<\fun{succ}(n))\to(k<n)+(k\eq\N n),
        \]
        we obtain a function $k\mapsto((k<\fun{succ}(n))\to\neg P(k))$, which is exactly an element of $Q(\fun{succ}(n))$. Composing this with 
        \[
            \fun{inr}\colon Q(\fun{succ}(n))\to S+Q(\fun{succ}(n))
        \]
        yields the desired function with domain $\neg P(n)$.        
    \end{itemize}

    This finishes the construction of $\sigma_n$, which proves inductively that $S+Q(n)$ is inhabited for all $n\colon\N$.

    Now we are in a position to construct the desired maps $P(n)\to S$. Indeed, given $u\colon P(n)$, the functions
    \begin{align*}
        \id_S&\colon S\to S\\
        \gamma_{u,n}&\colon Q(n)\to S,\quad
        q\mapsto\dep{n,(q,u)},
    \end{align*}
    define a map $\fun{rec}_{S+Q(n)}(S,\id_S,\gamma_{u,n})\colon S+Q(n)\to S$. Thus, we can define
    \begin{align*}
        P(n)&\to S\\
        u&\mapsto\fun{rec}_{S+Q(n)}(S,\id_S,\gamma_{u,n})(\tau_n),
    \end{align*}
    where $\tau_n\colon S+Q(n)$ is the witness obtained from our previous induction.
    %
    
\end{solution}


\begin{exr}\fe
    Prove that\/ $\fun{isProp}(P)\simeq(P\simeq\lVert P\rVert)$.
\end{exr}

\begin{solution}
    As shown in Lemma~\ref{lem:mere-propositions-are-truncated}, if $P$ is a mere proposition, we have a map $\fun{rec}_{\lVert P\rVert}(P,\id_P)\colon\lVert P\rVert\to P$. Since we also have the canonical injection $P\to\lVert P\rVert$, by Lemma~\ref{lem:mere-proposition-equivalence}, we deduce that there is a map $\fun{isProp}(P)\to(P\simeq\lVert P\rVert)$.

    By Proposition~\ref{prop:isSet-and-isProp-are-mere-propositions} and part~f) of Theorem~\ref{thm:mere-propositions}, both $\fun{isProp}(P)$ and $P\simeq\lVert P\rVert$ are mere propositions. Consequently, by Lemma~\ref{lem:mere-proposition-equivalence}, to prove their equivalence it suffices to construct a backward map.

    Let $(f,\eta,\vartheta)\colon P\simeq\lVert P\rVert$, where $f\colon P \to \lVert P \rVert$ is the forward map and $g\colon \lVert P \rVert \to P$ is its quasi-inverse. Given $x,y\colon P$, we have a path
    \[
        \fun{squash}(f(x),f(y))\colon f(x)\eq{\lVert P\rVert}f(y).
    \]
    Therefore, we can concatenate the following paths:
    \begin{align*}
        x
            &\eq{P} g(f(x))
                &&\text{by } \vartheta(x)^{-1},\\
            &\eq{P} g(f(y))
                &&\text{by } \fun{ap}_g(\fun{squash}(f(x),f(y))),\\
            &\eq{P} y
                &&\text{by } \eta(y),
    \end{align*}
    which proves that $P$ is a mere proposition. Thus, we can define the map
    \[
        (P\simeq\lVert P\rVert)\to\fun{isProp}(P),
            \quad\dep{f,(\eta,\vartheta)}\mapsto\epsilon,
    \]
    where $\epsilon\colon \fun{isProp}(P)$ is the witness constructed from the equality chain above.    
\end{solution}

\begin{exr}\fe\label{exr:finite-axiom-of-choice}
    As in classical set theory, the finite version of the axiom of choice is a theorem. Prove that the axiom of choice \eqref{eq:traditional-tac} holds when\/ $X$ is a finite type\/ $\fun{Fin}(n)$, as defined in\/ {\upshape Exercise~\ref{exr:Fin}}.
\end{exr}

\begin{solution}
    Let us introduce
    \[
        A_n\defeq\prod_{x\colon\fun{Fin}(n)}\lVert C(x)\rVert
        \quad\text{and}\quad
        B_n\defeq\prod_{x\colon\fun{Fin}(n)}C(x).
    \]
    We must construct a map
    \[
        f_n\colon A_n\to\lVert B_n\rVert.
    \]
    We proceed by induction on $n$.
    
    The case $n\equiv0$ is trivial because $\fun{Fin}(0)\eq{\univ}\fun0$, and so $A_0\simeq\fun1$ and $B_0\simeq\fun1$. We can compose the forward map of the first equivalence with the inverse of the second to obtain a map $A_0\to B_0$. Hence, we complete the base case by postcomposing this with the canonical injection $B_0\to\lVert B_0\rVert$.

    For the inductive step, let $C'\defeq C\circ\fun{inl}$ and set
    \[
        A'_n\defeq\prod_{x\colon\fun{Fin}(n)}\lVert C'(x)\rVert,
        \quad
        B'_n\defeq\prod_{x\colon\fun{Fin}(n)}C'(x).
    \]
    Since $\fun{Fin}(\fun{succ}(n))\equiv\fun{Fin}(n)+\fun1$, we can apply the universal property of dependent functions over coproducts, Theorem~\ref{thm:universal-property-of-product-over-coproduct}, to obtain the equivalences
    \[
        A_{\fun{succ}(n)}\simeq A'_n\times\lVert C(\fun{inr}(\ast))\rVert,
        \quad
        B_{\fun{succ}(n)}\simeq B'_n\times C(\fun{inr}(\ast)).
    \]
    Thus, to complete the induction we have to define a map
    \[
        A'_n\times\lVert C(\fun{inr}(\ast))\rVert\to
        \lVert B'_n\times C(\fun{inr}(\ast))\rVert.
    \]
    Since the inductive hypothesis gives us a map $A'_n\to\lVert B'_n\rVert$, the conclusion is a direct consequence of the following

    \textbf{Lemma.} {\itshape If\/ $A$ and\/ $B$ are types, then\/ $\lVert A\times B\rVert\simeq(\lVert A\rVert\times\lVert B\rVert)$.}

    \textsc{Proof:} There is a commutative diagram
    \[
        \begin{tikzcd}
            A\times B
                \arrow[r,"r"]
                \arrow[d,"\lvert-\rvert"']
            &\lVert A\rVert\times\lVert B\rVert
                \arrow[ld,"v",dashed,bend left,shift right]\\
            \lVert A\times B\rVert
                \arrow[ru,"u",shift right]
        \end{tikzcd}
    \]
    where
    \begin{align*}
        r&\defeq\lambda(a,b).\,(\lvert a\rvert,\lvert b\rvert),\\
        u&\defeq\fun{rec}_{\lVert A\times B\rVert}
            (
                \lVert A\rVert\times\lVert B\rVert,r
            ),
    \end{align*}
    which is well-defined because $r$ factors through $\lvert-\rvert$ by part c) of Theorem~\ref{thm:mere-propositions}.

    To construct $v$, define the curried function
    \begin{align*}
        A&\to(B\to\lVert A\times B\rVert)\\
        a&\mapsto(b\mapsto\lvert(a,b)\rvert).
    \end{align*}
    Since the codomain is a function type into a mere proposition, by part~b) of Theorem~\ref{thm:mere-propositions}, it is a mere proposition itself, and so the outer function factors through $\lVert A\rVert$. In turn, for each $a\colon A$, the inner function $b\mapsto\lvert(a,b)\rvert$ factors through $\lVert B\rVert$, yielding a map
    \[
        \lVert A\rVert\to(\lVert B\rVert\to\lVert A\times B\rVert),
    \]
    whose uncurried form gives $v$.
    
    The conclusion follows from Lemma~\ref{lem:mere-proposition-equivalence}.
    %
    
\end{solution}

\begin{exr}\label{exr:decidable-family}
    Show that the conclusion of\/ {\upshape Exercise~\ref{exr:decidable+mere-proposition-family}} if $P\colon\N\to\univ$ is any decidable family.
\end{exr}

\begin{solution}
    We open our reasoning by establishing the following

    \textbf{Lemma.} {\itshape For any type $P$, there is a map $(P+\neg P)\to(\lVert P\rVert+\neg\lVert P\rVert)$. In particular, if $P$ is decidable, then $\lVert P\rVert$ is decidable as well.}

    \textsc{Proof:}
        We proceed by induction on the coproduct $P+\neg P$.
        
        For the left branch, we take the composition $\fun{inl}\circ\lvert-\rvert$.
        
        For the right branch, observe that since $\fun0$ is a mere proposition, every map $u\colon P\to\fun0$ factors through the canonical map $\lvert-\rvert\colon P\to\lVert P\rVert$; i.e., there is a commutative triangle
        \[
            \begin{tikzcd}
                P
                    \arrow[r,"u"]
                    \arrow[d,"\lvert-\rvert"']
                &\fun0\\
                \lVert P\rVert
                    \arrow[ur,"\tilde u"']
            \end{tikzcd}
        \]
        Therefore, we can define the mapping for this branch as $u\mapsto\fun{inr}(\tilde u)$. \localqed
    
    \medskip

    In our case, since the family $P(n)$ is decidable, we deduce that the family of types $\lVert P(n)\rVert$ satisfies the hypothesis of Exercise~\ref{exr:decidable+mere-proposition-family}. Therefore, we have a map
    \[
        \Bigl\lVert
            \sum_{n\colon\N}\lVert P(n)\rVert
        \Bigr\rVert
        \to\sum_{n\colon\N}\lVert P(n)\rVert.
    \]
    Before proceeding, we need a second

    \textbf{Lemma.} {\itshape Let $P\colon A\to\univ$ be any type family. Then, there is an equivalence
    \[
        \Bigl\lVert
            \sum_{x\colon A}\lVert P(x)\rVert
        \Bigr\rVert
        \simeq
        \Bigl\lVert
            \sum_{x\colon A}P(x)
        \Bigr\rVert.
    \]}

    \textsc{Proof:} Since the type on the \lhs is a mere proposition, to construct a forward map, it suffices to define it with domain in the untruncated $\Sigma$-type. The induction principle for $\Sigma$-types reduces the goal to defining, for each $a\colon A$, a map
    \[
        \lVert P(a)\rVert\to\Bigl\lVert\sum_{x\colon A}P(x)\Bigr\rVert.
    \]
    Applying the recursion principle for propositional truncation once again, the goal reduces further to defining a map $P(a)\to\Bigl\lVert\sum_{x\colon A}P(x)\Bigr\rVert$, which is trivially constructed by $u\mapsto\lvert\dep{a,u}\rvert$.
    
    The backward map is trivially derived from the canonical map $P(x)\to\lVert P(x)\rVert$.
    
    Lemma~\ref{lem:mere-proposition-equivalence} completes the proof. \localqed

    \medskip
    As a result, by composing our derived maps, we have a witness
    \[
        \epsilon\colon\Bigl\lVert
            \sum_{n\colon\N}P(n)
        \Bigr\rVert
        \to\sum_{n\colon\N}\lVert P(n)\rVert.
    \]
    Let
    \[
        \dep{n,\zeta}\colon\sum_{n\colon\N}\lVert P(n)\rVert.
    \]
    Then $\zeta\colon\lVert P(n)\rVert$. By hypothesis, we are given an element $\epsilon\colon P(n)+\neg P(n)$. To derive from $\epsilon$ an element of $P(n)$, thereby establishing a map $\lVert P(n)\rVert\to P(n)$, it suffices to construct a map $(P(n)+\neg P(n))\to P(n)$ and evaluate it at $\epsilon$. We proceed by induction on the coproduct:
    
    \begin{description}[font=\normalfont\scshape]
        \item[Left branch:] Here we specify the identity map $\id_{P(n)}$.
    
        \item[Right branch:] Given $u\colon\neg P(n)$, we factor it through $\lVert P(n)\rVert$. This is possible because $u\colon P(n)\to\fun0$ yields an extension $\tilde u\colon\lVert P(n)\rVert\to\fun0$, since $\fun0$ is a mere proposition. Thus, $\tilde u(\zeta)\colon\fun0$, which allows us to specify $\fun{rec}_{\fun0}(P(n),\tilde u(\zeta))$ for the right branch.
    \end{description}
    
    %
    
\end{solution}

\begin{exr}
    Simplify the proof of\/ {\upshape Theorem~\ref{thm:N-equality-encode-decode}} by first proving that\/ $\fun{Code}(n,m)$ is a mere proposition for all\/ $n,m\colon\N$.
\end{exr}

\begin{solution}
    To prove that $\fun{Code}(n,m)$ is a mere proposition for all $n,m\colon\N$, we proceed by double induction on $n$ and $m$.
    \begin{enumerate}[-]
        \item For $n\equiv0$, we use induction on $m$ to show that $\fun{Code}(0,m)$ is a mere proposition:
        \begin{itemize}
            \item For $m\equiv0$, this is trivial because, by definition, $\fun{Code}(0,0)\equiv\fun1$.
            \item For $\fun{succ}(m)$, this is also trivial because $\fun{Code}(0,\fun{succ}(m))\equiv\fun0$ by definition.
        \end{itemize}
        \item For $\fun{succ}(n)$, suppose that we have $h_m\colon\fun{isProp}(\fun{Code}(n,m))$ for every $m\colon\N$. We prove by induction on $m$ that $\fun{isProp}(\fun{Code}(\fun{succ}(n),m))$ is inhabited:
        \begin{itemize}
            \item For $m\equiv0$, this is trivial because $\fun{Code}(\fun{succ}(n),0)\equiv\fun0$.
            \item For $\fun{succ}(m)$, since $\fun{Code}(\fun{succ}(n),\fun{succ}(m))\equiv\fun{Code}(n,m)$, our goal follows from the inductive hypothesis.
        \end{itemize}
    \end{enumerate}
    
    Let $n,m\colon\N$ and $c\colon\fun{Code}(n,m)$. Since both $\fun{encode}(\fun{decode}(c))$ and $c$ are inhabitants of $\fun{Code}(n,m)$, and we just proved that this type is a mere proposition, they must be equal. As a result, the homotopy $\fun{encode}\circ\fun{decode}\sim\id_{\fun{Code}(n,m)}$ is inhabited.

    The proof of the other composition proceeds as presented in Theorem~\ref{thm:N-equality-decode-encode}.
    %
    
\end{solution}
